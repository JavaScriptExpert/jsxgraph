 33 /**
 34  * Construct and handle images
 35  * @class Image:
 36  * It inherits from @see GeometryElement.
 37  * @constructor
 38  * @return A new geometry element Image
 39  */
 40 JXG.Image = function (board, url, coordinates, size, layer, id, name, el) {
 41     //this.constructor();
 42     this.type = JXG.OBJECT_TYPE_IMAGE;
 43     this.elementClass = JXG.OBJECT_CLASS_OTHER;                
 44     this.transformations = [];
 45 
 46     this.init(board, id, name);
 47     this.coords = new JXG.Coords(JXG.COORDS_BY_USER, coordinates, this.board);
 48     this.initialCoords = new JXG.Coords(JXG.COORDS_BY_USER, coordinates, this.board);
 49     this.size = [size[0]*board.stretchX,size[1]*board.stretchY];
 50     //this.imageBase64String = url; //imageBase64String;
 51     this.url = url;
 52     /**
 53      * Set the display layer.
 54      */
 55     if (layer == null) layer = board.options.layer['image'];
 56     this.layer = layer;
 57     this.parent = el;
 58     this.visProp['visible'] = true;
 59     //this.show = true; // noch noetig? BV
 60     this.id = this.board.addImage(this);
 61 };
 62 
 63 JXG.Image.prototype = new JXG.GeometryElement;
 64 
 65 /**
 66  * Empty function (for the moment). It is needed for highlighting, a feature not used for images right now.
 67  * @param {int} x Coordinate in x direction, screen coordinates.
 68  * @param {int} y Coordinate in y direction, screen coordinates.
 69  * @return Always returns false
 70  */
 71 JXG.Image.prototype.hasPoint = function (x,y) {
 72     return false;
 73 };
 74 
 75 /**
 76  * Send an update request to the renderer.
 77  */
 78 JXG.Image.prototype.updateRenderer = function () {
 79     this.updateTransform();
 80     this.board.renderer.updateImage(this);
 81 };
 82 
 83 JXG.Image.prototype.updateTransform = function () {
 84     if (this.transformations.length==0) {
 85         return;
 86     }
 87     for (var i=0;i<this.transformations.length;i++) {
 88         this.transformations[i].update();
 89     }
 90 };
 91 
 92 JXG.Image.prototype.addTransform = function (transform) {
 93     if (JXG.isArray(transform)) {
 94         for (var i=0;i<transform.length;i++) {
 95             this.transformations.push(transform[i]);
 96         }
 97     } else {
 98         this.transformations.push(transform);
 99     }
100 };

102 /**
103  * @class Displays an image. 
104  * @pseudo
105  * @description Shows an imgae. The image can be supplied as an URL or an base64 encoded inline image
106  * like "data:image/png;base64, /9j/4AAQSkZJRgA...".
107  * @constructor
108  * @name Image
109  * @type JXG.Image
110  * @throws {Exception} If the element cannot be constructed with the given parent objects an exception is thrown.
111  * @param {String_Array_Array} url, [position of the top left vertice], [width,height] 
112  * @example
113  * var im = board.create('image', ['http://geonext.uni-bayreuth.de/fileadmin/geonext/design/images/logo.gif', [-3,1],[5,5]]);
114  *
115  * </pre><div id="9850cda0-7ea0-4750-981c-68bacf9cca57" style="width: 400px; height: 400px;"></div>
116  * <script type="text/javascript">
117  *   var image_board = JXG.JSXGraph.initBoard('9850cda0-7ea0-4750-981c-68bacf9cca57', {boundingbox: [-4, 4, 4, -4], axis: false, showcopyright: false, shownavigation: false});
118  *   var image_im = image_board.create('image', ['http://geonext.uni-bayreuth.de/fileadmin/geonext/design/images/logo.gif', [-3,1],[5,5]]);
119  * </script><pre>
120  */
121 JXG.createImage = function(board, parents, atts) {
122     var url;
123     if (atts==null) {
124         atts = {};
125     } else if (atts['imageString']!=null) {
126         url = atts['imageString'];
127     }
128     if (typeof atts['layer'] == 'undefined') {
129         atts['layer'] = null;
130     }
131     return new JXG.Image(board, parents[0], parents[1], parents[2], atts['layer'], false, false, undefined);
132 };
133 
134 JXG.JSXGraph.registerElement('image', JXG.createImage);
