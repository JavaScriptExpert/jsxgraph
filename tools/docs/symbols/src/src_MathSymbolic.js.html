 36 /**
 37  * Generates symbolic coordinates for the part of a construction including all the elements from that
 38  * a specific element depends of. These coordinates will be stored in GeometryElement.symbolic.
 39  * @param {JXG.Board} board The board that's element get some symbolic coordinates.
 40  * @param {JXG.GeometryElement} element All ancestor of this element get symbolic coordinates.
 41  * @param {String} variable Name for the coordinates, e.g. x or u.
 42  * @param {String} append Method for how to append the number of the coordinates. Possible values are
 43  *                        'underscore' (e.g. x_2), 'none' (e.g. x2), 'brace' (e.g. x[2]).
 44  * @type int
 45  * @return Number of coordinates given.
 46  */
 47 JXG.Math.Symbolic.generateSymbolicCoordinatesPartial = function(board, element, variable, append) {
 48     function makeCoords(num) {
 49         if (append == 'underscore')
 50             return '' + variable + '_{' + num + '}';
 51         else if (append == 'brace')
 52             return '' + variable + '[' + num + ']';
 53         else
 54             return '' + variable + '' + num;
 55     }
 56 
 57     var list = element.ancestors;
 58     var count = 0;
 59     var t_num;
 60     for(var t in list) {
 61         t_num = 0;
 62         if (JXG.isPoint(list[t])) {
 63             for(var k in list[t].ancestors) {
 64                 t_num++;
 65             }
 66             if(t_num == 0) {
 67                 list[t].symbolic.x = list[t].coords.usrCoords[1];
 68                 list[t].symbolic.y = list[t].coords.usrCoords[2];
 69             } else {
 70                 count++;
 71                 list[t].symbolic.x = makeCoords(count);
 72                 count++;
 73                 list[t].symbolic.y = makeCoords(count);
 74             }
 75 
 76         }
 77     }
 78 
 79     if(JXG.isPoint(element)) {
 80         element.symbolic.x = 'x';
 81         element.symbolic.y = 'y';
 82     }
 83 
 84     return count;
 85 };

 87 /**
 88  * Clears all .symbolic.x and .symbolic.y members on every point of a given board.
 89  * @param {JXG.Board} board The board that's points get cleared their symbolic coordinates.
 90  */
 91 JXG.Math.Symbolic.clearSymbolicCoordinates = function(board) {
 92     for(var t in board.objects) {
 93         if (JXG.isPoint(board.objects[t])) {
 94             board.objects[t].symbolic.x = '';
 95             board.objects[t].symbolic.y = '';
 96         }
 97     }
 98 };

100 /**
101  * Generates polynomials for the part of a construction including all the points from that
102  * a specific element depends of.
103  * @param {JXG.Board} board The board that's points polynomials will be generated.
104  * @param {JXG.GeometryElement} element All points in the set of ancestors of this element are used to generate the set of polynomials.
105  * @type Array
106  * @return Array of polynomials as strings.
107  */
108 JXG.Math.Symbolic.generatePolynomials = function(board, element, generateCoords) {
109     if(generateCoords)
110         this.generateSymbolicCoordinatesPartial(board, element, 'u', 'brace');
111 
112     var list = element.ancestors,
113         number_of_ancestors,
114         pgs = [],
115         result = [],
116         t, k, i;
117     list[element.id] = element;
118 
119     for(t in list) {
120         number_of_ancestors = 0;
121         pgs = [];
122         if (JXG.isPoint(list[t])) {
123             for(k in list[t].ancestors) {
124                 number_of_ancestors++;
125             }
126             if(number_of_ancestors > 0) {
127                 pgs = list[t].generatePolynomial();
128                 for(i=0; i<pgs.length; i++)
129                     result.push(pgs[i]);
130             }
131         }
132     }
133 
134     if(generateCoords)
135         this.clearSymbolicCoordinates(board);
136 
137     return result;
138 };

140 /**
141  * Calculate geometric locus of a point given on a board. Invokes python script on server.
142  * @param {JXG.Board} board The board on that the point lies.
143  * @param {JXG.Point} point The point that will be traced.
144  * @param {function} callback A callback function that is called after the server request is finished.
145  *    Must take an array of strings as the only parameter.
146  * @type Array
147  * @return Array of points.
148  */
149 JXG.Math.Symbolic.geometricLocusByGroebnerBase = function(board, point, callback) {
150     var numDependent = this.generateSymbolicCoordinatesPartial(board, point, 'u', 'brace'),
151         poly = this.generatePolynomials(board, point);
152     var polyStr = poly.join(','),
153         xsye = new JXG.Coords(JXG.COORDS_BY_USR, [0,0], board),
154         xeys = new JXG.Coords(JXG.COORDS_BY_USR, [board.canvasWidth, board.canvasHeight], board),
155         fileurl;
156 
157     if(typeof JXG.Server.modules.geoloci == 'undefined')
158         JXG.Server.loadModule('geoloci')
159 
160     if(typeof JXG.Server.modules.geoloci == 'undefined')
161         throw new Error("JSXGraph: Unable to load JXG.Server module 'geoloci.py'.");
162 
163     this.cbp = function(data) {
164         //alert(data.exectime);
165         callback(data.datax, data.datay, data.polynomial);
166     };
167 
168     this.cb = JXG.bind(this.cbp, this);
169 
170     JXG.Server.modules.geoloci.lociCoCoA(xsye.usrCoords[1], xeys.usrCoords[1], xeys.usrCoords[2], xsye.usrCoords[2], numDependent, polyStr, this.cb);
171 
172     this.clearSymbolicCoordinates(board);
173 };
