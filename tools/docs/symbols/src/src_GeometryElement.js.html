 53 /**
 54  * Constructs a new GeometryElement object.
 55  * @class This is the basic class for geometry elements like points, circles and lines.
 56  * @constructor
 57  * of identical elements on the board. Is not yet implemented for all elements, only points, lines and circle can be traced.
 58  */
 59 JXG.GeometryElement = function() {
 60     /**
 61      * Reference to board where the element is drawn
 62      * @type JXG.Board
 63      * @default null
 64      * @see JXG.Board
 65      * @private
 66      */
 67     this.board = null;
 68 
 69     /**
 70      * Unique identifier for the element. Equivalent to id-attribute of renderer element.
 71      * @type String
 72      * @default empty string
 73      * @private
 74      */
 75     this.id = '';
 76 
 77     /**
 78      * Controls if updates are necessary
 79      * @type bool
 80      * @default true
 81      * @private
 82      */
 83     this.needsUpdate = true;
 84 
 85     /**
 86      * Not necessarily unique name for the element.
 87      * @type String
 88      * @default Name generated by {@link JXG.Board#generateName}.
 89      * @see JXG.Board#generateName
 90      */
 91     this.name = '';
 92 
 93     /**
 94      * An associative array containing all visual properties.
 95      * @type Object
 96      * @default empty object
 97      * @private
 98      */
 99     this.visProp = {};
100 
101     JXG.clearVisPropOld(this); // create this.visPropOld and set default values
102 
103     /**
104      * If element is in two dimensional real space this is true, else false.
105      * @type boolean
106      * @default true
107      * @private
108      */
109     this.isReal = true;
110 
111     /**
112      * Determines the elements border-style.
113      * Possible values are:
114      * <ul><li>0 for a solid line</li>
115      * <li>1 for a dotted line</li>
116      * <li>2 for a line with small dashes</li>
117      * <li>3 for a line with medium dashes</li>
118      * <li>4 for a line with big dashes</li>
119      * <li>5 for a line with alternating medium and big dashes and large gaps</li>
120      * <li>6 for a line with alternating medium and big dashes and small gaps</li></ul>
121      * @type number
122      * @name JXG.GeometryElement#dash
123      * @default 0
124      */
125     this.visProp['dash'] = 0;
126 
127     /**
128      * Stores all dependent objects to be updated when this point is moved.
129      * @type Object
130      * @private
131      */
132     this.childElements = {};
133 
134     /**
135      * If element has a label subelement then this property will be set to true.
136      * @type boolean
137      * @default false
138      * @private
139      */
140     this.hasLabel = false;
141 
142     /**
143      * display layer which will conting the element.
144      * Controlled in JXG.Options.
145      */
146     this.layer = 9;
147 
148     /**
149      * Stores all Intersection Objects which in this moment are not real and
150      * so hide this element.
151      * @type object
152      * @private
153      */
154     this.notExistingParents = {};
155 
156     /**
157      * If true the element will be traced, i.e. on every movement the element will be copied

159      * @see #clearTrace
160      * @see #traces
161      * @see #numTraces
162      * @type boolean
163      * @default false
164      * @name JXG.GeometryElement#trace
165      */
166     this.traced = false;
167 
168     /**
169      * If true the element is fixed and can not be dragged around. The element 
170      * will be repositioned on zoom and moveOrigin events.
171      * @type boolean
172      * @default false
173      * @name JXG.GeometryElement#fixed
174      */
175     this.fixed = false;
176 
177     /**
178      * If true the element is fixed and can not be dragged around. The element 
179      * will even stay at its position on zoom and moveOrigin events.
180      * Only free elements like points, texts, curves can be frozen.
181      * @type boolean
182      * @default false
183      * @name JXG.GeometryElement#fixed
184      */
185     this.frozen = false;
186 
187     /**
188      * Keeps track of all objects drawn as part of the trace of the element.
189      * @see #traced
190      * @see #clearTrace
191      * @see #numTraces
192      * @type Object
193      * @private
194      */
195     this.traces = {};
196 
197     /**
198      * Counts the number of objects drawn as part of the trace of the element.
199      * @see #traced
200      * @see #clearTrace
201      * @see #traces
202      * @type number
203      * @private
204      */
205     this.numTraces = 0;
206 
207     /**
208      * Stores the  transformations which are applied during update in an array
209      * @type Array
210      * @see JXG.Transformation
211      * @private
212      */
213     this.transformations = [];
214 
215     /** TODO
216      * @type TODO
217      * @default null
218      * @private
219      */
220     this.baseElement = null;
221 
222     /**
223      * Elements depending on this element are stored here.
224      * @type object
225      * @private
226      */
227     this.descendants = {};
228 
229     /**
230      * Elements on which this elements depends on are stored here.
231      * @type object
232      * @private
233      */
234     this.ancestors = {};
235 
236     /**
237      * Stores variables for symbolic computations
238      * @type Object
239      * @private
240      */
241     this.symbolic = {};
242 
243     /**
244      * [c,b0,b1,a,k,r,q0,q1]
245      *
246      * See
247      * A.E. Middleditch, T.W. Stacey, and S.B. Tor:
248      * "Intersection Algorithms for Lines and Circles",
249      * ACM Transactions on Graphics, Vol. 8, 1, 1989, pp 25-40.
250      *
251      * The meaning of the parameters is:
252      * Circle: points p=[p0,p1] on the circle fulfill
253      *  a<p,p> + <b,p> + c = 0
254      * For convenience we also store
255      *  r: radius
256      *  k: discriminant = sqrt(<b,b>-4ac)
257      *  q=[q0,q1] center
258      *
259      * Points have radius = 0.
260      * Lines have radius = infinity.
261      * b: normalized vector, representing the direction of the line.
262      *
263      * Should be put into Coords, when all elements possess Coords.
264      * @type array
265      * @default [1, 0, 0, 0, 1, 1, 0, 0]
266      * @private
267      */
268     this.stdform = [1,0,0,0,1, 1,0,0];
269 
270     /**
271      * Quadratic form representation of circles (and conics)
272      */
273     this.quadraticform = [[1,0,0],[0,1,0],[0,0,1]];
274 
275     /**
276      * If this is set to true, the element is updated in every update
277      * call of the board. If set to false, the element is updated only after
278      * zoom events or more generally, when the bounding box has been changed.
279      * Examples for the latter behaviour should be axes.
280      *
281      * @type boolean
282      * @default true
283      * @private
284      */
285     this.needsRegularUpdate = true;
286

306

309 
310     /* already documented in constructor */

316     } else {

478 /**
479  * Add an element as a child to the current element. Can be used to model dependencies between geometry elements.
480  * @param {JXG.GeometryElement} obj The dependent object.
481  */
482 JXG.GeometryElement.prototype.addChild = function (obj) {
483 	var el, el2;
484 
485     this.childElements[obj.id] = obj;
486 
487     this.addDescendants(obj);
488 
489     obj.ancestors[this.id] = this;
490     for(el in this.descendants) {
491         this.descendants[el].ancestors[this.id] = this;
492         for(el2 in this.ancestors) {
493             this.descendants[el].ancestors[this.ancestors[el2].id] = this.ancestors[el2];
494         }
495     }
496     for(el in this.ancestors) {
497         for(el2 in this.descendants) {
498             this.ancestors[el].descendants[this.descendants[el2].id] = this.descendants[el2];
499         }
500     }
501     return this;
502 };

504 /**
505  * Adds the given object to the descendants list of this object and all its child objects.
506  * @param obj The element that is to be added to the descendants list.
507  * @private
508  * @return
509  */
510 JXG.GeometryElement.prototype.addDescendants = function (/** JXG.GeometryElement */ obj) {
511 	var el;
512 
513     this.descendants[obj.id] = obj;
514     for(el in obj.childElements) {
515         this.addDescendants(obj.childElements[el]);
516     }
517     return this;

531 /**
532  * Animates properties for that object like stroke or fill color, opacity and maybe
533  * even more later.
534  * @param {Object} hash Object containing propiertes with target values for the animation.
535  * @param {number} time Number of milliseconds to complete the animation.
536  * @return A reference to the object
537  * @type JXG.GeometryElement
538  */
539 JXG.GeometryElement.prototype.animate = function(hash, time) {
540     var r, p,
541 	    delay = 35,
542 	    steps = Math.ceil(time/(delay * 1.0)),
543         i, self = this;
544 
545     this.animationData = {};
546 
547     var animateColor = function(startRGB, endRGB, property) {
548         var hsv1, hsv2, sh, ss, sv;
549         hsv1 = JXG.rgb2hsv(startRGB);
550         hsv2 = JXG.rgb2hsv(endRGB);
551 
552         sh = (hsv2[0]-hsv1[0])/(1.*steps);
553         ss = (hsv2[1]-hsv1[1])/(1.*steps);
554         sv = (hsv2[2]-hsv1[2])/(1.*steps);
555         self.animationData[property] = new Array(steps);
556         for(i=0; i<steps; i++) {
557             self.animationData[property][steps-i-1] = JXG.hsv2rgb(hsv1[0]+(i+1)*sh, hsv1[1]+(i+1)*ss, hsv1[2]+(i+1)*sv);
558         }
559     },
560 
561     animateFloat = function(start, end, property) {
562         start = parseFloat(start);
563         end = parseFloat(end);
564 
565         // we can't animate without having valid numbers.
566         // And parseFloat returns NaN if the given string doesn't contain
567         // a valid float number.
568         if(isNaN(start) || isNaN(end))
569             return;
570 
571         var s = (end - start)/(1.*steps);
572         self.animationData[property] = new Array(steps);
573         for(i=0; i<steps; i++) {
574             self.animationData[property][steps-i-1] = start + (i+1)*s;
575         }
576     };
577 
578     for(r in hash) {
579         p = r.toLowerCase();
580         switch(p) {
581             case 'strokecolor':
582                     animateColor(this.visProp['strokeColor'], hash[r], 'strokeColor');
583                 break;
584             case 'strokeopacity':
585                     animateFloat(this.visProp['strokeOpacity'], hash[r], 'strokeOpacity');
586                 break;
587             case 'strokewidth':
588                     animateFloat(this.visProp['strokeWidth'], hash[r], 'strokeWidth');
589                 break;
590             case 'fillcolor':
591                     animateColor(this.visProp['fillColor'], hash[r], 'fillColor');
592                 break;
593             case 'fillopacity':
594                     animateFloat(this.visProp['fillOpacity'], hash[r], 'fillOpacity');
595                 break;
596         }
597     }
598 
599 	this.board.animationObjects[this.id] = this;
600 	if(typeof this.board.animationIntervalCode == 'undefined') {
601 		this.board.animationIntervalCode = window.setInterval('JXG.JSXGraph.boards[\'' + this.board.id + '\'].animate();', delay);
602 	}
603 
604     return this;
605 };

625 /**
626  * Hide the element. It will still exist but not visible on the board.
627  */
628 JXG.GeometryElement.prototype.hideElement = function() {
629     this.visProp['visible'] = false;
630     this.board.renderer.hide(this);
631     if (this.label!=null && this.hasLabel) {
632         this.label.hiddenByParent = true;
633         if(this.label.content.visProp['visible']) {
634             this.board.renderer.hide(this.label.content);
635         }
636     }
637     return this;
638 };
639 
640 /**
641  * Make the element visible.
642  */
643 JXG.GeometryElement.prototype.showElement = function() {
644     this.visProp['visible'] = true;
645     this.board.renderer.show(this);
646     if (this.label!=null && this.hasLabel && this.label.hiddenByParent) {
647         this.label.hiddenByParent = false;
648         if(this.label.content.visProp['visible']) {
649             this.board.renderer.show(this.label.content);
650         }
651     }
652     return this;
653 };

686  * @param % Arbitrary number of strings, containing "key:value" pairs.
687  * The possible key values are the element and class fields in this documentation.
688  * @example
689  * // Set property directly on creation of an element using the attributes object parameter
690  * var board = JXG.JSXGraph.initBoard('jxgbox', {boundingbox: [-1, 5, 5, 1]};
691  * var p = board.createElement('point', [2, 2], {visible: false});
692  *
693  * // Now make this point visible and fixed:
694  * p.setProperty('fixed:true', 'visible:true');
695  *
696  * // Alternatively you can use #hideElement resp. #showElement:
697  * p.hideElement();
698  */
699 JXG.GeometryElement.prototype.setProperty = function () {
700     var i, key, color, pairRaw,
701         opacity,
702         pair;
703 
704     for (i=0; i<arguments.length; i++) {
705         pairRaw = arguments[i];
706         if (typeof pairRaw == 'string') {    // pairRaw is string of the form 'key:value'
707             pair = pairRaw.split(':');
708             // trim pair[0] and pair[1]
709             pair[0] = pair[0].replace (/^\s+/, '').replace (/\s+$/, '');
710             pair[1] = pair[1].replace (/^\s+/, '').replace (/\s+$/, '');
711         } else if (!JXG.isArray(pairRaw)) {    // pairRaw consists of objects of the form {key1:value1,key2:value2,...}
712             /*
713             for (var i=0; i<Object.keys(pairRaw).length;i++) {  // Here, the prototype lib is used (Object.keys, Object.isArray)
714                 var key = Object.keys(pairRaw)[i];
715                 this.setProperty([key,pairRaw[key]]);
716             }
717             */
718             for (key in pairRaw) {
719                 this.setProperty([key,pairRaw[key]]);
720             }
721             return this;
722         } else {                             // pairRaw consists of array [key,value]
723             pair = pairRaw;
724         }
725         if (pair[1]==null) continue;
726         switch(pair[0].replace(/\s+/g).toLowerCase()) {   // Whitespace entfernt und in Kleinbuchstaben umgewandelt.
727             case 'strokewidth':
728                 this.visProp['strokeWidth'] = pair[1];
729                 this.visProp['highlightStrokeWidth'] = pair[1];
730                 this.board.renderer.setObjectStrokeWidth(this, this.visProp['strokeWidth']);
731                 break;
732             case 'strokecolor':
733                 color = pair[1];
734                 if (color.length=='9' && color.substr(0,1)=='#') {
735                     opacity = color.substr(7,2);
736                     color = color.substr(0,7);
737                 }
738                 else {
739                     opacity = 'FF';
740                 }
741                 this.visProp['strokeColor'] = color;
742                 this.visProp['strokeOpacity'] = parseInt(opacity.toUpperCase(),16)/255;
743                 this.board.renderer.setObjectStrokeColor(this, this.visProp['strokeColor'], this.visProp['strokeOpacity']);
744                 break;
745             case 'fillcolor':
746                 color = pair[1];
747                 if (color.length=='9' && color.substr(0,1)=='#') {
748                     opacity = color.substr(7,2);
749                     color = color.substr(0,7);
750                 }
751                 else {
752                     opacity = 'FF';
753                 }
754                 this.visProp['fillColor'] = color;
755                 this.visProp['fillOpacity'] = parseInt(opacity.toUpperCase(),16)/255;
756                 this.board.renderer.setObjectFillColor(this, this.visProp['fillColor'], this.visProp['fillOpacity']);
757                 break;
758             case 'highlightstrokewidth':
759                 this.visProp['highlightStrokeWidth'] = pair[1];
760                 break;                
761             case 'highlightstrokecolor':
762                 color = pair[1];
763                 if (color.length=='9' && color.substr(0,1)=='#') {
764                     opacity = color.substr(7,2);
765                     color = color.substr(0,7);
766                 }
767                 else {
768                     opacity = 'FF';
769                 }
770                 this.visProp['highlightStrokeColor'] = color;
771                 this.visProp['highlightStrokeOpacity'] = parseInt(opacity.toUpperCase(),16)/255;
772                 break;
773             case 'highlightfillcolor':
774                 color = pair[1];
775                 if (color.length=='9' && color.substr(0,1)=='#') {
776                     opacity = color.substr(7,2);
777                     color = color.substr(0,7);
778                 }
779                 else {
780                     opacity = 'FF';
781                 }
782                 this.visProp['highlightFillColor'] = color;
783                 this.visProp['highlightFillOpacity'] = parseInt(opacity.toUpperCase(),16)/255;
784                 break;
785             case 'fillopacity':
786                 this.visProp['fillOpacity'] = pair[1];
787                 this.board.renderer.setObjectFillColor(this, this.visProp['fillColor'], this.visProp['fillOpacity']);
788                 break;
789             case 'strokeopacity':
790                 this.visProp['strokeOpacity'] = pair[1];
791                 this.board.renderer.setObjectStrokeColor(this, this.visProp['strokeColor'], this.visProp['strokeOpacity']);
792                 break;
793             case 'highlightfillopacity':
794                 this.visProp['highlightFillOpacity'] = pair[1];
795                 break;
796             case 'highlightstrokeopacity':
797                 this.visProp['highlightStrokeOpacity'] = pair[1];
798                 break;
799             case 'labelcolor':
800                 color = pair[1];
801                 if (color.length=='9' && color.substr(0,1)=='#') {
802                     opacity = color.substr(7,2);
803                     color = color.substr(0,7);
804                 }
805                 else {
806                     opacity = 'FF';
807                 }
808                 if(opacity == '00') {
809                     if (this.label!=null && this.hasLabel) {
810                         this.label.content.hideElement();
811                     }
812                 }
813                 if(this.label!=null && this.hasLabel) {
814                     this.label.color = color;
815                     this.board.renderer.setObjectStrokeColor(this.label.content, color, opacity);
816                 }
817                 if(this.type == JXG.OBJECT_TYPE_TEXT) {
818                     this.visProp['strokeColor'] = color;
819                     this.board.renderer.setObjectStrokeColor(this, this.visProp['strokeColor'], 1);
820                 }
821                 break;
822             case 'infoboxtext':
823                 if(typeof(pair[1]) == 'string') {
824                     this.infoboxText = pair[1];
825                 }
826                 else this.infoboxText = false;
827                 break;
828             case 'showinfobox':
829                 if(pair[1] == 'false' || pair[1] == false) {
830                     this.showInfobox = false;
831                 }
832                 else if(pair[1] == 'true' || pair[1] == true) {
833                     this.showInfobox = true;
834                 }
835                 break;
836             case 'visible':
837                 if(pair[1] == 'false' || pair[1] == false) {
838                     this.visProp['visible'] = false;
839                     this.hideElement();
840                 }
841                 else if(pair[1] == 'true' || pair[1] == true) {
842                     this.visProp['visible'] = true;

844                 }
845                 break;
846             case 'dash':
847                 this.setDash(pair[1]);
848                 break;
849             case 'trace':
850                 if(pair[1] == 'false' || pair[1] == false) {
851                     this.traced = false;
852                 }
853                 else if(pair[1] == 'true' || pair[1] == true) {
854                     this.traced = true;
855                 }
856                 break;
857             case 'style':
858                 this.setStyle(1*pair[1]);
859                 break;
860             case 'face':
861             	if(this.elementClass == JXG.OBJECT_CLASS_POINT)
862             		this.setFace(pair[1]);
863                 break;
864             case 'size':
865             	if(this.elementClass == JXG.OBJECT_CLASS_POINT) {
866             		this.visProp['size'] = 1*pair[1];
867                 	this.board.renderer.updatePoint(this);
868         		}
869                 break;
870             case 'fixed':
871                 this.fixed = ((pair[1]=='false') || (pair[1]==false)) ? false : true;
872                 break;
873             case 'frozen':
874                 this.frozen = ((pair[1]=='false') || (pair[1]==false)) ? false : true;
875                 break;
876             case 'shadow':
877                 if(pair[1] == 'false' || pair[1] == false) {
878                     this.visProp['shadow'] = false;
879                 }
880                 else if(pair[1] == 'true' || pair[1] == true) {
881                     this.visProp['shadow'] = true;
882                 }
883                 this.board.renderer.setShadow(this);
884                 break;
885             case 'gradient':
886                 this.visProp['gradient'] = pair[1];
887                 this.board.renderer.setGradient(this);
888                 break;
889             case 'gradientsecondcolor':
890                 color = pair[1];
891                 if (color.length=='9' && color.substr(0,1)=='#') {
892                     opacity = color.substr(7,2);
893                     color = color.substr(0,7);
894                 }
895                 else {
896                     opacity = 'FF';
897                 }
898                 this.visProp['gradientSecondColor'] = color;
899                 this.visProp['gradientSecondOpacity'] = parseInt(opacity.toUpperCase(),16)/255;
900                 this.board.renderer.updateGradient(this);
901                 break;
902             case 'gradientsecondopacity':
903                 this.visProp['gradientSecondOpacity'] = pair[1];
904                 this.board.renderer.updateGradient(this);
905                 break;
906             case 'draft':
907                 if(pair[1] == 'false' || pair[1] == false) {
908                     if(this.visProp['draft'] == true) {
909                         this.visProp['draft'] = false;
910                         this.board.renderer.removeDraft(this);
911                     }
912                 }
913                 else if(pair[1] == 'true' || pair[1] == true) {
914                     this.visProp['draft'] = true;
915                     this.board.renderer.setDraft(this);
916                 }
917                 break;
918             case 'straightfirst':
919                 if(pair[1] == 'false' || pair[1] == false) {
920                     this.visProp['straightFirst'] = false;
921                 }
922                 else if(pair[1] == 'true' || pair[1] == true) {
923                     this.visProp['straightFirst'] = true;
924                 }
925                 this.setStraight(this.visProp['straightFirst'], this.visProp['straightLast']);
926                 break;
927             case 'straightlast':
928                 if(pair[1] == 'false' || pair[1] == false) {
929                     this.visProp['straightLast'] = false;
930                 }
931                 else if(pair[1] == 'true' || pair[1] == true) {
932                     this.visProp['straightLast'] = true;
933                 }
934                 this.setStraight(this.visProp['straightFirst'], this.visProp['straightLast']);
935                 break;
936             case 'firstarrow':
937                 if(pair[1] == 'false' || pair[1] == false) {
938                     this.visProp['firstArrow'] = false;
939                 }
940                 else if(pair[1] == 'true' || pair[1] == true) {
941                     this.visProp['firstArrow'] = true;
942                 }
943                 this.setArrow(this.visProp['firstArrow'], this.visProp['lastArrow']);
944                 break;
945             case 'lastarrow':
946                 if(pair[1] == 'false' || pair[1] == false) {
947                     this.visProp['lastArrow'] = false;
948                 }
949                 else if(pair[1] == 'true' || pair[1] == true) {
950                     this.visProp['lastArrow'] = true;
951                 }
952                 this.setArrow(this.visProp['firstArrow'], this.visProp['lastArrow']);
953                 break;
954             case 'curvetype':
955                 this.curveType = pair[1];
956                 break;
957             case 'fontsize':
958                 this.visProp['fontSize'] = pair[1];
959                 break;
960             case 'insertticks':
961                 if(this.type == JXG.OBJECT_TYPE_TICKS) {
962                     var old = this.insertTicks;
963                     this.insertTicks = true;
964                     if(pair[1] == 'false' || pair[1] == false) {
965                         this.insertTicks = false;
966                     }
967                     if(old != this.insertTicks) this.calculateTicksCoordinates();
968                 }
969                 break;
970             case 'drawlabels':
971                 if(this.type == JXG.OBJECT_TYPE_TICKS) {
972                     var old = this.drawLabels;
973                     this.drawLabels = true;
974                     if(pair[1] == 'false' || pair[1] == false) {
975                         this.drawLabels = false;
976                     }
977                     if(old != this.drawLabels) this.calculateTicksCoordinates();
978                 }
979                 break;
980             case 'drawzero':
981                 if(this.type == JXG.OBJECT_TYPE_TICKS) {
982                     var old = this.drawZero;
983                     this.drawZero = true;
984                     if(pair[1] == 'false' || pair[1] == false) {
985                         this.drawZero = false;
986                     }
987                     if(old != this.drawZero) this.calculateTicksCoordinates();
988                 }
989                 break;
990             case 'minorticks':
991                 if(this.type == JXG.OBJECT_TYPE_TICKS) {
992                     var old = this.minorTicks;
993                     if((pair[1] != null) && (pair[1] > 0))
994                         this.minorTicks = pair[1];
995                     if(old != this.minorTicks) this.calculateTicksCoordinates();
996                 }
997                 break;
998             case 'majortickheight':
999                 if(this.type == JXG.OBJECT_TYPE_TICKS) {
1000                     var old = this.majorHeight;
1001                     if((pair[1] != null) && (pair[1] > 0))
1002                         this.majorHeight = pair[1];
1003                     if(old != this.majorHeight) this.calculateTicksCoordinates();
1004                 }
1005                 break;
1006             case 'minortickheight':
1007                 if(this.type == JXG.OBJECT_TYPE_TICKS) {
1008                     var old = this.minorHeight;
1009                     if((pair[1] != null) && (pair[1] > 0))
1010                         this.minorHeight = pair[1];
1011                     if(old != this.minorHeight) this.calculateTicksCoordinates();
1012                 }
1013                 break;
1014             case 'snapwidth':
1015                 if(this.type == JXG.OBJECT_TYPE_GLIDER) {
1016                     this.snapWidth = pair[1];
1017                 }
1018                 break;
1019             case 'withlabel':
1020                 if(!pair[1]) {
1021                     if (this.label!=null && this.hasLabel) {
1022                         this.label.content.hideElement();
1023                     }
1024                 } 
1025                 else {
1026                     if (this.label!=null && this.hasLabel) {
1027                         if(this.visProp['visible']) {
1028                             this.label.content.showElement();
1029                         }
1030                         
1031                     }
1032                     else {
1033                         this.addLabelToElement();
1034                         if(!this.visProp['visible']) {
1035                             this.label.content.hideElement();
1036                         }                        
1037                     }
1038                 }
1039                 this.hasLabel = pair[1];
1040         }
1041     }
1042     return this;
1043 };

1148 /**

1155 JXG.GeometryElement.prototype.createLabel = function(withLabel,coords) {
1156     var isTmpId = false;
1157     if (typeof coords=='undefined' || coords==null) {
1158         coords = [10,10];
1159     }
1160     this.nameHTML = this.board.algebra.replaceSup(this.board.algebra.replaceSub(this.name));

1169             [this.label.relativeCoords[0],-this.label.relativeCoords[1]], this.id+"Label", "", null, true, this.board.options.text.defaultDisplay);
1170         if (isTmpId) delete(this.board.objects[this.id]);

1173             this.label.hiddenByParent = true;
1174             this.label.content.visProp['visible'] = false;
1175         }
1176         this.hasLabel = true;
