 31 /**
 32  * Construct and handle texts.
 33  * @class Text: On creation the GEONExT syntax
 34  * of <value>-terms 
 35  * are converted into JavaScript syntax.
 36  * The coordinates can be relative to the coordinates of an element "element".
 37  * @constructor
 38  * @return A new geometry element Text
 39  */
 40 JXG.Text = function (board, contentStr, element, coords, id, name, digits, isLabel, display, layer) {
 41     this.constructor();
 42 
 43     this.type = JXG.OBJECT_TYPE_TEXT;
 44     this.elementClass = JXG.OBJECT_CLASS_OTHER;                
 45 
 46     this.init(board, id, name);
 47     this.contentStr = contentStr;
 48     this.plaintextStr = '';
 49 
 50     /**
 51      * Set the display layer.
 52      */
 53     if (layer == null) layer = board.options.layer['text'];
 54     this.layer = layer;
 55 
 56     /**
 57      * There is choice between 'html' and 'internal'
 58      * 'internal' is the text element of SVG and the textpath element 
 59      * of VML.
 60      */
 61     this.display = display || 'html'; 
 62     
 63     if((typeof isLabel != 'undefined') && (isLabel != null)) {
 64         this.isLabel = isLabel;
 65     }
 66     else {
 67         this.isLabel = false;
 68     }
 69     
 70     /**
 71      * The text color of the given text.
 72      * @type {string}
 73      * @name JXG.Text#strokeColor
 74      */
 75     this.visProp['strokeColor'] = this.board.options.text.strokeColor;
 76     /**
 77      * The text opacity of the given text.
 78      * @type {string}
 79      * @name JXG.Text#strokeOpacity
 80      */
 81      /**
 82      * The font size of the given text.
 83      * @type {string}
 84      * @name JXG.Text#fontSize
 85      * @default {@link JXG.Options.fontSize}
 86      */
 87     //this.visProp['fontSize'] = this.board.options.text.fontSize;
 88 
 89     this.visProp['visible'] = true;
 90     //this.show = true; // noch noetig? BV
 91 
 92     if (digits!=null) {
 93         this.digits = digits;
 94     } else {
 95         this.digits = 2;
 96     }
 97 
 98     /**
 99      * Coordinates of the text.
100      * @ private
101      * @type JXG.Coords
102      */
103     if ((this.element = this.board.objects[element])){
104         var anchor;
105         if(!this.isLabel) {
106             anchor = this.element.getTextAnchor();
107         }
108         else {
109             anchor = this.element.getLabelAnchor();
110         }      
111         this.element.addChild(this);
112         this.relativeCoords = new JXG.Coords(JXG.COORDS_BY_SCREEN, [parseFloat(coords[0]),parseFloat(coords[1])],this.board);     
113         this.coords = new JXG.Coords(JXG.COORDS_BY_SCREEN, 
114             [this.relativeCoords.scrCoords[1]+anchor.scrCoords[1],
115              this.relativeCoords.scrCoords[2]+anchor.scrCoords[2]], this.board);
116     } else {
117         this.X = JXG.createFunction(coords[0],this.board,'');
118         this.Y = JXG.createFunction(coords[1],this.board,'');
119         this.coords = new JXG.Coords(JXG.COORDS_BY_USER, [this.X(),this.Y()], this.board);
120         var fs = 'this.coords.setCoordinates(JXG.COORDS_BY_USER,[this.X(),this.Y()]);';
121         this.updateCoords = new Function('',fs);
122     }
123 
124     if (typeof this.contentStr=='function') {
125         this.updateText = function() { this.plaintextStr = this.contentStr(); };
126     } else {
127         var plaintext;
128         if (typeof this.contentStr=='number') {
129             plaintext = (this.contentStr).toFixed(this.digits);  
130         } else {
131             if (this.board.options.text.useASCIIMathML) {
132                 plaintext = "'`"+this.contentStr+"`'";              // Convert via ASCIIMathML
133             } else {
134                 plaintext = this.generateTerm(this.contentStr);   // Converts GEONExT syntax into JavaScript string
135             }
136         }
137         this.updateText = new Function('this.plaintextStr = ' + plaintext + ';');
138     }
139     //this.updateText();                    // First evaluation of the string    
140     if(!this.isLabel) {
141         this.id = this.board.addText(this);
142     }
143     if (typeof this.contentStr=='string') {
144         this.notifyParents(this.contentStr);
145     }
146 };
147 JXG.Text.prototype = new JXG.GeometryElement();

160 /**
161  * Overwrite the text.
162  * @param {string,function} str
163  * @return {object} reference to the text object.
164  */
165 JXG.Text.prototype.setText = function(text) {
166     var plaintext;
167     if (typeof text=='number') {
168         plaintext = (text).toFixed(this.digits);  
169     } else {
170         plaintext = this.generateTerm(text);   // Converts GEONExT syntax into JavaScript string
171     }
172     this.updateText = new Function('this.plaintextStr = ' + plaintext + ';');
173     this.updateText();
174     return this;
175 };

177 /**
178  * Set the text to new, fixed coordinates.
179  * @param {number} x
180  * @param {number} y
181  * @return {object} reference to the text object.
182  */
183 JXG.Text.prototype.setCoords = function (x,y) {
184     this.X = function() { return x; };
185     this.Y = function() { return y; };
186     this.coords = new JXG.Coords(JXG.COORDS_BY_USER, [x,y], this.board);
187     return this;
188 };

190 /**
191  * Evaluates the text.
192  * Then, the update function of the renderer
193  * is called. 
194  */
195 JXG.Text.prototype.update = function () {
196     if (this.needsUpdate && !this.frozen) {
197         if (this.relativeCoords){
198             var anchor;
199             if(!this.isLabel) {
200                 anchor = this.element.getTextAnchor();
201             }
202             else {
203                 anchor = this.element.getLabelAnchor();
204             }
205             this.coords.setCoordinates(JXG.COORDS_BY_SCREEN, 
206                 [this.relativeCoords.scrCoords[1]+anchor.scrCoords[1],
207                  this.relativeCoords.scrCoords[2]+anchor.scrCoords[2]]);
208         } else {
209             this.updateCoords();
210         }
211     }   
212     if (this.needsUpdate) {
213         this.updateText();
214     }
215     return this;
216 };
217 
218 /**
219  * The update function of the renderer
220  * is called. 
221  * @private
222  */
223 JXG.Text.prototype.updateRenderer = function () {
224     if (this.needsUpdate) {
225         this.board.renderer.updateText(this);
226         this.needsUpdate = false;
227     }
228     return this;
229 };

231 /**
232  * Converts the GEONExT syntax of the <value> terms into JavaScript.
233  * Also, all Objects whose name appears in the term are searched and
234  * the text is added as child to these objects.
235  * @private
236  * @see Algebra
237  * @see #geonext2JS.
238  */
239 JXG.Text.prototype.generateTerm = function (contentStr) {
240     var res = null;
241     var elements = this.board.elementsByName;
242     var plaintext = '""';
243     contentStr = contentStr.replace(/\r/g,''); 
244     contentStr = contentStr.replace(/\n/g,''); 
245     contentStr = contentStr.replace(/\"/g,'\\"'); 
246     contentStr = contentStr.replace(/\'/g,"\\'"); 
247     contentStr = contentStr.replace(/&arc;/g,'∠'); 
248     contentStr = contentStr.replace(/<arc\s*\/>/g,'∠'); 
249     contentStr = contentStr.replace(/<sqrt\s*\/>/g,'√'); 
250 
251     // Convert GEONExT syntax into  JavaScript syntax
252     var i;
253     //var i = contentStr.indexOf('<mp>');
254     //contentStr = contentStr.slice(i+4);
255     //i = contentStr.indexOf('</mp>');
256     //contentStr = contentStr.slice(0,i);
257 
258     i = contentStr.indexOf('<value>');
259     var j = contentStr.indexOf('</value>');
260     if (i>=0) {
261         while (i>=0) {
262             plaintext += ' + "'+ JXG.GeonextParser.replaceSub(JXG.GeonextParser.replaceSup(contentStr.slice(0,i))) + '"';
263             var term = contentStr.slice(i+7,j);
264             var res = JXG.GeonextParser.geonext2JS(term, this.board); 
265             res = res.replace(/\\"/g,'"');
266             res = res.replace(/\\'/g,"'");
267             if (res.indexOf('toFixed')<0) {  // GEONExT-Hack: apply rounding once only.  
268                 plaintext += '+('+ res + ').toFixed('+(this.digits)+')';
269             } else {
270                 plaintext += '+('+ res + ')';
271             }
272             contentStr = contentStr.slice(j+8);
273             i = contentStr.indexOf('<value>');
274             j = contentStr.indexOf('</value>');
275         }
276     } //else {
277     plaintext += ' + "' + JXG.GeonextParser.replaceSub(JXG.GeonextParser.replaceSup(contentStr)) + '"';
278     //}
279     plaintext = plaintext.replace(/<overline>/g,'<span style=text-decoration:overline>');
280     plaintext = plaintext.replace(/<\/overline>/g,'</span>');
281     plaintext = plaintext.replace(/<arrow>/g,'<span style=text-decoration:overline>');
282     plaintext = plaintext.replace(/<\/arrow>/g,'</span>');
283 /*    i = plaintext.indexOf('<name>');
284     j = plaintext.indexOf('</name>');
285     while (i>=0) {
286         var head = plaintext.slice(0,i+6);
287         var mid = plaintext.slice(i+6,j);
288         var tail = plaintext.slice(j);
289         mid = JXG.GeonextParser.replaceSub(JXG.GeonextParser.replaceSup(mid));
290         plaintext = head + mid + tail;
291         i = plaintext.indexOf('<name>',i+7);
292         j = plaintext.indexOf('</name>',i+7);
293     }
294 */
295     plaintext = plaintext.replace(/&/g,'&'); // This should replace &pi; by π
296     return plaintext;
297 };

299 /**
300  * Finds dependencies in a given term and notifies the parents by adding the
301  * dependent object to the found objects child elements.
302  * @param {String} term String containing dependencies for the given object.
303  * @private
304  */
305 JXG.Text.prototype.notifyParents = function (contentStr) {
306     var res = null;
307     var elements = this.board.elementsByName;
308 
309     do {
310         var search = /<value>([\w\s\*\/\^\-\+\(\)\[\],<>=!]+)<\/value>/;
311         res = search.exec(contentStr);
312         if (res!=null) {
313             JXG.GeonextParser.findDependencies(this,res[1],this.board);
314             contentStr = contentStr.substr(res.index);
315             contentStr = contentStr.replace(search,'');
316         }
317     } while (res!=null);
318     return this;
319 };

321 /**
322  * @class This element is used to provide a constructor for text, which is just a wrapper for element {@link Text}. 
323  * @pseudo
324  * @description
325  * @name Text
326  * @augments JXG.GeometryElement
327  * @constructor
328  * @type JXG.Text
329  *
330  * @param {number,function_number,function_String,function} x,y,str Parent elements for text elements.
331  *                     <p>
332  *                     x and y are the coordinates of the lower left corner of the text box. The position of the text is fixed, 
333  *                     x and y are numbers. The position is variable if x or y are functions.
334  *                     <p>
335  *                     The text to display may be given as string or as function returning a string.
336  *
337  * There is the attribute 'display' which takes the values 'html' or 'internal'. In case of 'html' a HTML division tag is created to display
338  * the text. In this case it is also possible to use ASCIIMathML. Incase of 'internal', a SVG or VML text element is used to display the text.
339  * @see JXG.Text
340  * @example
341  * // Create a fixed text at position [0,1].
342  *   var t1 = board.create('text',[0,1,"Hello World"]); 
343  * </pre><div id="896013aa-f24e-4e83-ad50-7bc7df23f6b7" style="width: 300px; height: 300px;"></div>
344  * <script type="text/javascript">
345  *   var t1_board = JXG.JSXGraph.initBoard('896013aa-f24e-4e83-ad50-7bc7df23f6b7', {boundingbox: [-3, 6, 5, -3], axis: true, showcopyright: false, shownavigation: false});
346  *   var t1 = t1_board.create('text',[0,1,"Hello World"]);
347  * </script><pre>
348  * @example
349  * // Create a variable text at a variable position.
350  *   var s = board.create('slider',[[0,4],[3,4],[-2,0,2]]);
351  *   var graph = board.create('text', 
352  *                        [function(x){ return s.Value();}, 1,
353  *                         function(){return "The value of s is"+s.Value().toFixed(2);}
354  *                        ]
355  *                     );
356  * </pre><div id="5441da79-a48d-48e8-9e53-75594c384a1c" style="width: 300px; height: 300px;"></div>
357  * <script type="text/javascript">
358  *   var t2_board = JXG.JSXGraph.initBoard('5441da79-a48d-48e8-9e53-75594c384a1c', {boundingbox: [-3, 6, 5, -3], axis: true, showcopyright: false, shownavigation: false});
359  *   var s = t2_board.create('slider',[[0,4],[3,4],[-2,0,2]]);
360  *   var t2 = t2_board.create('text',[function(x){ return s.Value();}, 1, function(){return "The value of s is "+s.Value().toFixed(2);}]);
361  * </script><pre>
362  */
363 JXG.createText = function(board, parentArr, atts) {
364     atts = JXG.checkAttributes(atts,{layer:null,display:board.options.text.defaultDisplay,parent:null});  // 'html' or 'internal'
365     if(atts['parent'] != null) { atts['parent'] = atts['parent'].id;}
366     return new JXG.Text(board, parentArr[parentArr.length-1], atts['parent'], parentArr, atts['id'], atts['name'], atts['digits'], false, atts['display'],atts['layer']);
367 };
368 
369 JXG.JSXGraph.registerElement('text', JXG.createText);
