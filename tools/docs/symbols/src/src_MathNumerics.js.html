 80 /**
 81  * Solves a system of linear equations given by the right triangular matrix R and vector b.
 82  * @param R Right triangular matrix. All entries a_(i,j) with i < j are ignored.
 83  * @param b Right hand side of the linear equation system.
 84  * @return A vector that solves the system of linear equations.
 85  * @private
 86  */ 
 87 JXG.Math.Numerics.backwardSolve = function(/** JXG.Math.Matrix */ R, /** JXG.Math.Vector */ b) /** JXG.Math.Vector */ {
 88     var x = b,
 89         m, n,
 90         i, j;
 91 
 92     // m: number of rows of R
 93     // n: number of columns of R
 94     // Relaxation: R may be of type JXG.Math.Matrix or Array
 95     //             b may be of type JXG.Math.Vector or Array
 96     if (R.m) { // R is of type JXG.Math.Matrix
 97         m = R.m();
 98         n = R.n();
 99     } else {   // R is of type array
100         m = R.length;
101         n = (R.length>0)?R[0].length:0;
102     }
103     for (i = m-1; i >= 0; i--) {
104         for (j = n-1; j > i; j--) {
105             x[i] -= R[i][j] * x[j];
106         }
107         x[i] /= R[i][i];
108     }
109    
110     return x;
111 };

113 /**
114  * Solves a system of linear equations given by A and b using the Gauss-Jordan-elimination.
115  * The algorithm runs in-place. I.e. the entries of A and b are changed.
116  * @param A Square matrix containing the coefficients of the lineare equation system.
117  * @param b A vector containing the linear equation system's right hand side. 
118  * @throws {JXG.DimensionMismatchException} If a non-square-matrix is given or the b has not the right length.
119  * @throws {JXG.SingularMatrixException} If A's rank is not full.
120  * @return A vector that solves the linear equation system.
121  */
122 JXG.Math.Numerics.Gauss = function(/** JXG.Math.Matrix */ A, /** JXG.Math.Vector */ b) /** JXG.Math.Vector */ {
123     var eps = JXG.Math.eps,
124         n,                 
125         i, j, k, P,
126         x, y;
127     
128     // n: number of columns of A
129     // Relaxation: A may be of type JXG.Math.Matrix or Array
130     //             b may be of type JXG.Math.Vector or Array
131     if (A.n) { // A is of type JXG.Math.Matrix
132         n = A.n();
133     } else {   // A is of type array
134         n = (A.length>0)?A[0].length:0;
135     }
136     
137     /* vector to keep track of permutations caused by pivotion */
138     P = new JXG.Math.Vector();
139     for (i = 0; i < n; i++) {
140         P.push(i);
141     }
142     
143    /* Gauss-Jordan-elimination */
144     for (j=0; j < n; j++)
145     {
146         for (i = n-1; i > j; i--) {
147             /* Is the element which is to eliminate greater than zero? */
148             if (Math.abs(A[i][j]) > JXG.Math.eps) {
149                 /* Equals pivot element zero? */
150                 if (Math.abs(A[j][j]) < JXG.Math.eps) {
151                     /* Yeah, so we have to exchange the rows */
152                     A.exchangeRows(i, j);
153                     b.exchange(i, j);
154                     P.exchange(i, j);
155                 }
156                 else {
157                     /* Saves the L matrix of the LR-decomposition. unneeded. */
158                     A[i][j] /= A[j][j];
159                     /* Transform right-hand-side b */
160                     b[i] -= A[i][j] * b[j];
161                     /* subtract the multiple of A[i][j] / A[j][j] of the j-th row from the i-th. */
162                     for (k = j + 1; k < n; k ++) {
163                         A[i][k] -= A[i][j] * A[j][k];
164                     }
165                 }
166             }
167             if (Math.abs(A[j][j]) < JXG.Math.eps) { // The absolute values of all coefficients below the j-th row in the j-th column are smaller than JXG.Math.eps.
168                 throw new SingularMatrixException();
169             }
170         }
171     }
172    
173     return JXG.Math.Numerics.backwardSolve(A, b); // return Array
174     /* 
175     y = JXG.Math.Numerics.backwardSolve(A, b);
176     x = new JXG.Math.Vector();
177     for (i = 0; i < n ; i++) { // y.n()
178         x.push(y[P[i]]);
179     }
180    
181     return x; // return JXG.Math.Vector
182     */
183 };

185 /**
186  * Compute the inverse of an nxn matrix with Gauss elimination.
187  * @param {JXG.Math.Matrix} Ain
188  */
189 JXG.Math.Numerics.Inverse = function(Ain) {
190     var i,j,k,s,ma,r,swp,
191         n = Ain.length,
192         A = [],
193         p = [],
194         hv = [];
195         
196     for (i=0;i<n;i++) {
197         A[i] = [];
198         for (j=0;j<n;j++) { A[i][j] = Ain[i][j]; }
199         p[i] = i;
200     }
201     
202     for (j=0;j<n;j++) {
203         // pivot search:
204         ma = Math.abs(A[j][j]);
205         r = j;
206         for (i=j+1;i<n;i++) {
207             if (Math.abs(A[i][j])>ma) {
208                 ma = Math.abs(A[i][j]);
209                 r = i;
210             }
211         }
212         if (ma<=JXG.Math.eps) { // Singular matrix
213             return false;  
214         }
215         // swap rows:
216         if (r>j) {
217             for (k=0;k<n;k++) { 
218                 swp = A[j][k]; A[j][k] = A[r][k]; A[r][k] = swp;
219             }
220             swp = p[j]; p[j] = p[r]; p[r] = swp;
221         }
222         // transformation:
223         s = 1.0/A[j][j];
224         for (i=0;i<n;i++) {
225             A[i][j] *= s;
226         }
227         A[j][j] = s;
228         for (k=0;k<n;k++) if (k!=j) {
229             for (i=0;i<n;i++) if (i!=j) {
230                 A[i][k] -= A[i][j]*A[j][k];
231             }
232             A[j][k] = -s*A[j][k];
233         }
234     }
235     // swap columns:
236     for (i=0;i<n;i++) {
237         for (k=0;k<n;k++) { hv[p[k]] = A[i][k]; }
238         for (k=0;k<n;k++) { A[i][k] = hv[k]; }
239     }
240     return A;
241 };

255 /**
256  * Compute the Eigenvalues and Eigenvectors of a symmetric 3x3 matrix with the Jacobi method
257  * Adaption of a FORTRAN program by Ed Wilson, Dec. 25, 1990
258  * @param {JXG.Math.Matrix} Ain A symmetric 3x3 matrix.
259  * @type Object
260  * @throws {Exception} If A's rank is not full.
261  * @return [A,V] the matrices A and V. The diagonal of A contains the Eigenvalues,
262  * V contains the Eigenvectors.
263  */
264 JXG.Math.Numerics.Jacobi = function(Ain) {
265     var i,j,k,ih,aa,si,co,tt,eps=JXG.Math.eps, //Math.sqrt(JXG.Math.eps),
266         sum = 0.0,
267         ssum, amax,
268         n = Ain.length,
269         V = [[0,0,0],[0,0,0],[0,0,0]],
270         A = [[0,0,0],[0,0,0],[0,0,0]];
271     
272     // Initialization. Set initial Eigenvectors.
273     for (i=0;i<n;i++) {
274         for (j=0;j<n;j++) { 
275             V[i][j] = 0.0;
276             A[i][j] = Ain[i][j];
277             sum += Math.abs(A[i][j]);
278         }
279         V[i][i] = 1.0;
280     }
281     // Trivial problems
282     if (n==1) { return [A,V]; }
283     if (sum<=0.0) { return [A,V]; }
284     
285     sum /= (n*n);
286 
287     // Reduce matrix to diagonal
288     do {
289         ssum = 0.0;
290         amax = 0.0;
291         for (j=1;j<n;j++) {
292             for (i=0;i<j;i++) {
293                 // Check if A[i][j] is to be reduced
294                 aa = Math.abs(A[i][j]);
295                 if (aa>amax) { amax = aa; }
296                 ssum += aa;
297                 if (aa<eps/*0.1*amax*/) {
298                     continue;
299                 } else {
300                     // calculate rotation angle
301                     aa = Math.atan2(2.0*A[i][j],A[i][i]-A[j][j])*0.5;
302                     si = Math.sin(aa);
303                     co = Math.cos(aa);
304                     // Modify 'i' and 'j' columns
305                     for (k=0;k<n;k++) {
306                         tt = A[k][i];
307                         A[k][i] =  co*tt+si*A[k][j];
308                         A[k][j] = -si*tt+co*A[k][j];
309                         tt = V[k][i];
310                         V[k][i] =  co*tt+si*V[k][j];
311                         V[k][j] = -si*tt+co*V[k][j];
312                     }
313                     // Modify diagonal terms
314                     A[i][i] =  co*A[i][i]+si*A[j][i];
315                     A[j][j] = -si*A[i][j]+co*A[j][j];
316                     A[i][j] = 0.0;
317                     // Make 'A' matrix symmetrical
318                     for (k=0;k<n;k++) {
319                         A[i][k] = A[k][i];
320                         A[j][k] = A[k][j];
321                     }
322                     // A[i][j] made zero by rotation
323                 }
324             }
325         }
326     } while(Math.abs(ssum)/sum>eps);
327     return [A,V];
328 };

330 /**
331  * Calculates the integral of function f over interval using Newton-Cotes-algorithm.
332  * @param interval The integration interval, e.g. [0, 3]. 
333  * @param f A function which takes one argument of type number and returns a number.
334  * @return Integral value of f over interval interval
335  * @throws {Exception} If {@link JXG.Math.Numerics.number_of_nodes} doesn't match
336  * {@link JXG.Math.Numerics.integration_type} an exception is thrown. If you want to use
337  * simpson rule respectively milne rule {@link JXG.Math.Numerics.number_of_nodes} must be dividable by
338  * 2 respectively 4.
339  * @example
340  * function f(x) {
341  *   return x*x;
342  * }
343  * 
344  * // calculates integral of <tt>f</tt> from 0 to 2.
345  * var area1 = JXG.Math.Numerics.NewtonCotes([0, 2], f);
346  * 
347  * // the same with anonymous function
348  * var area2 = JXG.Math.Numerics.NewtonCotes([0, 2], function (x) { return x*x; });
349  */
350 JXG.Math.Numerics.NewtonCotes = function(/** array */ interval, /** function */ f) /** float */ {
351     var integral_value = 0.0,
352         step_size = (interval[1] - interval[0]) / this.number_of_nodes,
353         evaluation_point, i, number_of_intervals;
354 
355     switch(this.integration_type) {
356         case JXG.INT_TRAPEZ:
357             integral_value = (f(interval[0]) + f(interval[1])) * 0.5;
358     
359             evaluation_point = interval[0];
360             for (i = 0; i < this.number_of_nodes - 1; i++)
361             {
362                 evaluation_point += step_size;
363                 integral_value   += f(evaluation_point);
364             }
365             integral_value *= step_size;
366 
367             break;
368         case JXG.INT_SIMPSON:
369             if (this.number_of_nodes%2 > 0) {
370                 throw new Error("JSXGraph:  INT_SIMPSON requires JXG.Math.Numerics.number_of_nodes dividable by 2.");
371             }
372             number_of_intervals = this.number_of_nodes / 2.0;
373             integral_value = f(interval[0]) + f(interval[1]);
374             evaluation_point = interval[0];
375             for (i = 0; i < number_of_intervals - 1; i++)
376             {
377                 evaluation_point += 2.0 * step_size;
378                 integral_value   += 2.0 * f(evaluation_point);
379             }
380             evaluation_point = interval[0] - step_size;
381             for (i = 0; i < number_of_intervals; i++)
382             {
383                 evaluation_point += 2.0 * step_size;
384                 integral_value   += 4.0 * f(evaluation_point);
385             }
386             integral_value *= step_size / 3.0;
387             break;
388         default:
389             if (this.number_of_nodes%4 > 0) {
390                 throw new Error("JSXGraph: Error in INT_MILNE: JXG.Math.Numerics.number_of_nodes must be a multiple of 4");
391             }
392             number_of_intervals = this.number_of_nodes * 0.25;
393             integral_value = 7.0 * (f(interval[0]) + f(interval[1]));
394             evaluation_point = interval[0];
395             for (i = 0; i < number_of_intervals - 1; i++)
396             {
397                 evaluation_point += 4.0 * step_size;
398                 integral_value   += 14.0 * f(evaluation_point);
399             }
400             evaluation_point = interval[0] - 3.0 * step_size;
401             for (i = 0; i < number_of_intervals; i++)
402             {
403                 evaluation_point += 4.0 * step_size;
404                 integral_value   += 32.0 * (f(evaluation_point) + f(evaluation_point + 2 * step_size));
405             }
406             evaluation_point = interval[0] - 2.0 * step_size;
407             for (i = 0; i < number_of_intervals; i++)
408             {
409                 evaluation_point += 4.0 * step_size;
410                 integral_value   += 12.0 * f(evaluation_point);
411             }
412             integral_value *= 2.0 * step_size / 45.0; /* todo */
413     }
414     return integral_value;
415 };

417 /**
418  * Calculates second derivatives at the knots.
419  * @param x x values of knots
420  * @param y y values of knots
421  * @return Second derivatives of interpolated function at the knots.
422  */
423 JXG.Math.Numerics.splineDef = function(/** JXG.Math.Vector */ x, /** JXG.Math.Vector */ y) /** JXG.Math.Vector */ {
424     var n = x.length,
425         pair, i, diag, z, l,
426         data = new Array(),
427         dx = [], 
428         delta = [],
429         F;
430         
431     if (x.length != y.length)
432         throw new Error("JSXGraph: Error in JXG.Math.Numerics.splineDef: Input vector dimensions do not match.");
433     
434     for (i=0; i<n; i++) {
435         pair = {X: x[i], Y: y[i]};
436         data.push(pair);
437     }
438     data.sort(function (a,b) { return a.X - b.X; });
439     for (i=0; i<n; i++) {
440         x[i] = data[i].X;
441         y[i] = data[i].Y;
442     }
443     
444     for (i=0; i<n-1; i++) {
445         dx.push(x[i+1] - x[i]);
446     }
447     for (i=0; i<n-2; i++) {
448         delta.push(6 * (y[i+2] - y[i+1])/(dx[i+1]) - 6 * (y[i+1] - y[i])/(dx[i]));
449     }
450 
451     // ForwardSolve
452     diag = new Array();
453     z = new Array();
454     diag.push(2*(dx[0] + dx[1]));
455     z.push(delta[0]);
456 
457     for (i=0; i<n-3; i++) {
458         l = dx[i+1]/diag[i];
459         diag.push(2 * (dx[i+1] + dx[i+2]) - l*dx[i+1]);
460         z.push(delta[i+1] - l*z[i]);
461     }
462 
463     // BackwardSolve
464     F = new Array();
465     F[n-3] = z[n-3]/diag[n-3];
466     for(i=n-4; i>=0; i--) {
467         F[i] = (z[i] - (dx[i+1]*F[i+1]))/diag[i];
468     }
469     
470     // Generate f''-Vector
471     for(i=n-3; i>=0; i--)
472         F[i+1] = F[i];
473     
474     // natural cubic spline
475     F[0] = 0;
476     F[n-1] = 0;
477     return F;
478     //return new JXG.Math.Vector(F);
479 };

481 /**
482  * Evaluate points on spline.
483  * @param {float,Array} x0 A single float value or an array of values to evaluate
484  * @param {JXG.Math.Vector} x x values of knots
485  * @param {JXG.Math.Vector} y y values of knots
486  * @param {JXG.Math.Vector} F Second derivatives at knots, calculated by #splineDef
487  * @see splineDef
488  * @type float,Array
489  * @return A single value
490  */
491 JXG.Math.Numerics.splineEval = function(x0, x, y, F) {
492     var n = x.length,
493         l = 1,
494         asArray = false,
495         y0, i, j, a, b, c, d, x_;
496 
497     if (n != y.length)
498         throw new Error("JSXGraph: Error in JXG.Math.Numerics.splineEval: Defining vector dimensions do not match.");
499     
500     // number of points to be evaluated
501     if(JXG.isArray(x0)) {
502         l = x0.length;
503         asArray = true;
504     } else
505         x0 = [x0];
506     
507     y0 = new Array();
508     
509     for (i=0; i<l; i++) {
510         // is x0 in defining interval?
511         if( (x0[i] < x[0]) || (x[i] > x[n-1]))
512             return 'NaN';
513 //            throw new Error("JSXGraph: Error in JXG.Math.Numerics.splineEval: Evaluation point outside spline interval.");
514         
515         // determine part of spline in which x0 lies
516         j;
517         for (j=1; j<n; j++) {
518             if (x0[i] <= x[j])
519                 break;
520         }
521         j--;
522         
523         // we're now in the j-th partial interval, i.e. x[j] < x0[i] <= x[j+1];
524         // determine the coefficients of the polynomial in this interval
525         a = y[j];
526         b = (y[j+1]-y[j])/(x[j+1]-x[j]) - (x[j+1]-x[j])/6 * (F[j+1]+2*F[j]);
527         c = F[j]/2;
528         d = (F[j+1]-F[j])/(6*(x[j+1]-x[j]));
529         // evaluate x0[i]
530         x_ = x0[i]-x[j];
531         //y0.push(a + b*x_ + c*x_*x_ + d*x_*x_*x_);
532         y0.push(a + (b + (c+ d*x_)*x_)*x_);
533     }
534 
535     if (asArray)
536         return y0;
537     else
538         return y0[0];
539 };

542   * Generate a string containing the function term of a polynomial.
543   * @param {Array} coeffs Coefficients of the polynomial. The position i belongs to x^i.
544   * @param {int} deg Degree of the polynomial
545   * @param {String} varname Name of the variable (usually 'x')

550     var t = '', i;
551     for (i=deg;i>=0;i--) {
552         t += '('+coeffs[i].toPrecision(prec)+')';
553         if (i>1) { t+='*'+varname+'<sup>'+i+'</sup> + '; }
554         else if (i==1) { t+='*'+varname+' + '; }
555     }
556     return t;
557 }

901 /**
902  * Newton's method to find roots of a funtion in one variable.
903  * @param {function} f We search for a solution of f(x)=0.
904  * @param {float} x initial guess for the root, i.e. staring value.
905  * @param {object} obj optional object that is treated as "this" in the function body. This is useful, if the function is a 
906  *                 method of an object and contains a reference to its parent object via "this".
907  * @return {float} root of the function f.
908  */
909 JXG.Math.Numerics.newton = function(/** function */ f, /** number */ x, /** object */ obj) /** number */ {
910     var i = 0,
911         h = 0.000001,
912         newf = f.apply(obj,[x]), // set "this" to "obj" in f 
913         df;
914     while (i<50 && Math.abs(newf)>h) {
915         df = this.D(f,obj)(x);
916         if (Math.abs(df)>h) {
917             x -= newf/df;
918         } else {
919             x += (Math.random()*0.2-1.0);
920         }
921         newf = f.apply(obj,[x]);
922         i++;
923     }
924     return x;
925 };

941  * Compute coordinates for the rectangles showing the Riemann sum.
942  * @param {function} f Function f, whose integral is approximated by the Riemann sum.
943  * @param {int} n number of rectangles.
944  * @param {String} type Type of approximation. Possible values are: 'left', 'right', 'middle', 'lower', 'upper', or 'trapezodial'.
945  * @param {float} start Left border of the approximation interval
946  * @param {float} end Right border of the approximation interval
947  * @return {array} An array of two arrays containing the x and y coordinates for the rectangles showing the Riemann sum. This array may be used as
948  *                 parent array of a JXG.Curve.
949  */
950 JXG.Math.Numerics.riemann = function(/** function */ f, /** type */ n,  /** type */ type,  /** type */ start,  /** type */ end)  /** array */ {
951     var xarr,yarr,i,delta,j,x,y,x1,delta1,y1;
952     
953     xarr = [];
954     yarr = [];
955     j = 0;
956     x = start;

958     xarr[j] = x; yarr[j] = 0.0;
959     
960     if (n>0) {
961         delta = (end-start)/n;
962         delta1 = delta*0.01; // for 'lower' and 'upper'
963         
964         for (i=0;i<n;i++) {
965             if (type=='right') {
966                 y = f(x+delta);
967             } else if (type=='middle') {
968                 y = f(x+delta*0.5);
969             } else if ((type=='left') || (type=='trapezodial')) {
970                 y = f(x);
971             } else if (type=='lower') {
972                 y = f(x);
973                 for (x1=x+delta1;x1<=x+delta;x1+=delta1) {
974                     y1 = f(x1);
975                     if (y1<y) { y = y1; };
976                 }
977             } else { // (type=='upper')
978                 y = f(x);
979                 for (x1=x+delta1;x1<=x+delta;x1+=delta1) {
980                     y1 = f(x1);
981                     if (y1>y) { y = y1; };
982                 }
983             }
984             
985             j++;
986             xarr[j] = x; yarr[j] = y;
987             j++; x+=delta;
988             if (type=='trapezodial') {
989                 y = f(x);
990             }
991             xarr[j] = x; yarr[j] = y;
992             j++;
993             xarr[j] = x; yarr[j] = 0.0;
994          }
995     }
996     return [xarr,yarr];
