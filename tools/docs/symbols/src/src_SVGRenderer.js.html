112 JXG.SVGRenderer.prototype.setShadow = function(el) {
113     if (el.visPropOld['shadow']==el.visProp['shadow']) {
114         return;
115     }
116     if(el.rendNode != null) {
117         if(el.visProp['shadow']) {
118             el.rendNode.setAttributeNS(null,'filter','url(#f1)');
119         }
120         else {
121             el.rendNode.removeAttributeNS(null,'filter');
122         }    
123     }
124     el.visPropOld['shadow']=el.visProp['shadow'];
125 }

127 JXG.SVGRenderer.prototype.setGradient = function(el) {
128     var fillNode = el.rendNode, col, op,
129         node,node,node3,x1,x2,y1,y2;
130     
131     if (typeof el.visProp['fillOpacity']=='function') {
132         op = el.visProp['fillOpacity']();
133     } else {
134         op = el.visProp['fillOpacity'];
135     }
136     op = (op>0)?op:0;
137     if (typeof el.visProp['fillColor']=='function') {
138         col = el.visProp['fillColor']();
139     } else {
140         col = el.visProp['fillColor'];
141     }
142 
143     if(el.visProp['gradient'] == 'linear') {
144         node = this.createPrim('linearGradient',el.id+'_gradient');
145         x1 = '0%'; // TODO: get x1,x2,y1,y2 from el.visProp['angle']
146         x2 = '100%';
147         y1 = '0%';
148         y2 = '0%'; //means 270 degrees
149 
150         node.setAttributeNS(null,'x1',x1);
151         node.setAttributeNS(null,'x2',x2);
152         node.setAttributeNS(null,'y1',y1);
153         node.setAttributeNS(null,'y2',y2);
154         node2 = this.createPrim('stop',el.id+'_gradient1');
155         node2.setAttributeNS(null,'offset','0%');
156         node2.setAttributeNS(null,'style','stop-color:'+col+';stop-opacity:'+op);     
157         node3 = this.createPrim('stop',el.id+'_gradient2');
158         node3.setAttributeNS(null,'offset','100%');
159         node3.setAttributeNS(null,'style','stop-color:'+el.visProp['gradientSecondColor']+';stop-opacity:'+el.visProp['gradientSecondOpacity']);
160         node.appendChild(node2);
161         node.appendChild(node3);     
162         this.defs.appendChild(node);
163         fillNode.setAttributeNS(null, 'style', 'fill:url(#'+el.id+'_gradient)');      
164         el.gradNode1 = node2;
165         el.gradNode2 = node3;
166     }
167     else if (el.visProp['gradient'] == 'radial') {
168         node = this.createPrim('radialGradient',el.id+'_gradient');
169 
170         node.setAttributeNS(null, 'cx', '50%')
171         node.setAttributeNS(null, 'cy', '50%')
172         node.setAttributeNS(null, 'r', '50%')
173         node.setAttributeNS(null, 'fx', el.visProp['gradientPositionX']*100+'%')
174         node.setAttributeNS(null, 'fy', el.visProp['gradientPositionY']*100+'%')
175 
176         node2 = this.createPrim('stop',el.id+'_gradient1');
177         node2.setAttributeNS(null,'offset','0%');
178         node2.setAttributeNS(null,'style','stop-color:'+el.visProp['gradientSecondColor']+';stop-opacity:'+el.visProp['gradientSecondOpacity']);
179         node3 = this.createPrim('stop',el.id+'_gradient2');
180         node3.setAttributeNS(null,'offset','100%');
181         node3.setAttributeNS(null,'style','stop-color:'+col+';stop-opacity:'+op);         
182 
183         node.appendChild(node2);
184         node.appendChild(node3);     
185         this.defs.appendChild(node);
186         fillNode.setAttributeNS(null, 'style', 'fill:url(#'+el.id+'_gradient)'); 
187         el.gradNode1 = node2;
188         el.gradNode2 = node3;
189     }
190     else {
191         fillNode.removeAttributeNS(null,'style');
192     }
193 };

195 JXG.SVGRenderer.prototype.updateGradient = function(el) {
196     var node2 = el.gradNode1, 
197         node3 = el.gradNode2, 
198         col, op;
199 
200     if (node2==null || node3==0) {
201         return;
202     }
203     if (typeof el.visProp['fillOpacity']=='function') {
204         op = el.visProp['fillOpacity']();
205     } else {
206         op = el.visProp['fillOpacity'];
207     }
208     op = (op>0)?op:0;
209     if (typeof el.visProp['fillColor']=='function') {
210         col = el.visProp['fillColor']();
211     } else {
212         col = el.visProp['fillColor'];
213     }
214     
215     if(el.visProp['gradient'] == 'linear') {
216         node2.setAttributeNS(null,'style','stop-color:'+col+';stop-opacity:'+op);     
217         node3.setAttributeNS(null,'style','stop-color:'+el.visProp['gradientSecondColor']+';stop-opacity:'+el.visProp['gradientSecondOpacity']);
218     } else if (el.visProp['gradient'] == 'radial') {
219         node2.setAttributeNS(null,'style','stop-color:'+el.visProp['gradientSecondColor']+';stop-opacity:'+el.visProp['gradientSecondOpacity']);
220         node3.setAttributeNS(null,'style','stop-color:'+col+';stop-opacity:'+op);         
221     }

263 JXG.SVGRenderer.prototype.updateTicks = function(axis,dxMaj,dyMaj,dxMin,dyMin) {
264     var tickStr = "",
265         i, c, node, 
266         len = axis.ticks.length;
267         
268     for (i=0; i<len; i++) {
269         c = axis.ticks[i].scrCoords;
270         if (axis.ticks[i].major) {
271             if (axis.labels[i].visProp['visible']) this.drawText(axis.labels[i]);
272             tickStr += "M " + (c[1]+dxMaj) + " " + (c[2]-dyMaj) + " L " + (c[1]-dxMaj) + " " + (c[2]+dyMaj) + " ";
273         }
274         else
275             tickStr += "M " + (c[1]+dxMin) + " " + (c[2]-dyMin) + " L " + (c[1]-dxMin) + " " + (c[2]+dyMin) + " ";
276     }
277     
278     node = this.getElementById(axis.id);
279     if(node == null) {
280         node = this.createPrim('path', axis.id);
281         //node.setAttributeNS(null, 'shape-rendering', 'crispEdges');
282         this.appendChildPrim(node,axis.layer);
283         this.appendNodesToElement(axis,'path');
284     }
285     node.setAttributeNS(null, 'stroke', axis.visProp['strokeColor']);    
286     node.setAttributeNS(null, 'stroke-opacity', axis.visProp['strokeOpacity']);
287     node.setAttributeNS(null, 'stroke-width', axis.visProp['strokeWidth']);
288     this.updatePathPrim(node, tickStr, axis.board);
289 };

310 JXG.SVGRenderer.prototype.joinTransforms = function(el,t) {
311     var str = '', i, s,
312         len = t.length;
313         
314     for (i=0;i<len;i++) {
315         s = t[i].matrix[1][1]+','+t[i].matrix[2][1]+','+t[i].matrix[1][2]+','+t[i].matrix[2][2]+','+t[i].matrix[1][0]+','+t[i].matrix[2][0];
316         str += 'matrix('+s+') ';
317     }
318     return str;
319 };

327         str = '';

329     el.rendNode.setAttributeNS(null, 'transform', str);

350 JXG.SVGRenderer.prototype.setObjectStrokeColor = function(el, color, opacity) {
351     var c = this.eval(color), 
352         o = this.eval(opacity), 
353         node;
354 
355     o = (o>0)?o:0;
356 
357     if (el.visPropOld['strokeColor']==c && el.visPropOld['strokeOpacity']==o) {
358         return;
359     }
360     node = el.rendNode;
361     if(el.type == JXG.OBJECT_TYPE_TEXT) {
362         if (el.display=='html') {
363             node.style.color = c; // Schriftfarbe
364         } else {
365             node.setAttributeNS(null, "style", "fill:"+ c); 
366         }
367     }
368     else {
369         node.setAttributeNS(null, 'stroke', c);
370         node.setAttributeNS(null, 'stroke-opacity', o);          
371     }
372     if(el.type == JXG.OBJECT_TYPE_ARROW) {
373          this.setArrowAtts(el.rendNodeTriangle,c,o);
374     } else if (el.elementClass == JXG.OBJECT_CLASS_CURVE || el.elementClass == JXG.OBJECT_CLASS_LINE) {
375         if(el.visProp['firstArrow']) {
376             this.setArrowAtts(el.rendNodeTriangleStart,c,o);
377         }
378         if(el.visProp['lastArrow']) {
379             this.setArrowAtts(el.rendNodeTriangleEnd,c,o);
380         }                
381     }     
382     el.visPropOld['strokeColor'] = c;
383     el.visPropOld['strokeOpacity'] = o;
384 };

526 JXG.SVGRenderer.prototype.makeArrows = function(el) {
527     var node2;
528     if (el.visPropOld['firstArrow']==el.visProp['firstArrow'] && el.visPropOld['lastArrow']==el.visProp['lastArrow']) {
529         return;
530     }
531     if(el.visProp['firstArrow']) {
532         node2 = el.rendNodeTriangleStart;
533         if(node2 == null) {
534             node2 = this.createArrowHead(el,'End');
535             this.defs.appendChild(node2);            
536             el.rendNodeTriangleStart = node2;
537             el.rendNode.setAttributeNS(null, 'marker-start', 'url(#'+this.container.id+'_'+el.id+'TriangleEnd)');    
538         }    
539     }
540     else {
541         node2 = el.rendNodeTriangleStart;
542         if(node2 != null) {
543             this.remove(node2);
544         }
545     }
546     if(el.visProp['lastArrow']) {
547         node2 = el.rendNodeTriangleEnd;
548         if(node2 == null) {
549             node2 = this.createArrowHead(el,'Start');
550             this.defs.appendChild(node2);            
551             el.rendNodeTriangleEnd = node2;

602 JXG.SVGRenderer.prototype.updatePathStringPrim = function(el) {
603     var symbm = ' M ',
604         symbl = ' L ',
605         nextSymb = symbm,
606         maxSize = 5000.0,
607         pStr = '',
608         //h = 3*el.board.canvasHeight,
609         //w = 100*el.board.canvasWidth,
610         i, scr, 
611         isNoPlot = (el.curveType!='plot'),
612         //isFunctionGraph = (el.curveType=='functiongraph'),
613         len;
614 
615     if (el.numberPoints<=0) { return ''; }
616     
617     if (isNoPlot && el.board.options.curve.RDPsmoothing) {
618         el.points = this.RamenDouglasPeuker(el.points,0.5);
619     }
620     len = Math.min(el.points.length,el.numberPoints);
621     for (i=0; i<len; i++) {
622         scr = el.points[i].scrCoords;
623         //if (isNaN(scr[1]) || isNaN(scr[2]) /*|| Math.abs(scr[1])>w || (isFunctionGraph && (scr[2]>h || scr[2]<-0.5*h))*/ ) {  // PenUp
624         if (isNaN(scr[1]) || isNaN(scr[2])) {  // PenUp
625             nextSymb = symbm;
626         } else {
627             // Chrome has problems with values  being too far away.
628             if (scr[1]>maxSize) { scr[1] = maxSize; }
629             else if (scr[1]<-maxSize) { scr[1] = -maxSize; }
630             if (scr[2]>maxSize) { scr[2] = maxSize; }
631             else if (scr[2]<-maxSize) { scr[2] = -maxSize; }
632             
633             pStr += [nextSymb,scr[1],' ',scr[2]].join(''); // Attention: first coordinate may be inaccurate if far way
634             nextSymb = symbl;
635         }
636     }
637     return pStr;
638 };

640 JXG.SVGRenderer.prototype.updatePathStringPoint = function(el, size, type) {
641     var s = '',
642         scr = el.coords.scrCoords,
643         sqrt32 = size*Math.sqrt(3)*0.5,
644         s05 = size*0.5;
645         
646     if(type == 'x') {
647         s = 'M ' + (scr[1]-size) + ' ' + (scr[2]-size) + ' L ' + 
648         (scr[1]+size) + ' ' + (scr[2]+size) + ' M ' + 
649         (scr[1]+size) + ' ' + (scr[2]-size) + ' L ' +
650         (scr[1]-size) + ' ' + (scr[2]+size);
651     }
652     else if(type == '+') {
653         s = 'M ' + (scr[1]-size) + ' ' + (scr[2]) + ' L ' + 
654         (scr[1]+size) + ' ' + (scr[2]) + ' M ' + 
655         (scr[1]) + ' ' + (scr[2]-size) + ' L ' +
656         (scr[1]) + ' ' + (scr[2]+size);    
657     }
658     else if(type == 'diamond') {
659         s = 'M ' + (scr[1]-size) + ' ' + (scr[2]) + ' L ' + 
660         (scr[1]) + ' ' + (scr[2]+size) + ' L ' + 
661         (scr[1]+size) + ' ' + (scr[2]) + ' L ' +
662         (scr[1]) + ' ' + (scr[2]-size) + ' Z ';
663     }
664     else if(type == 'A') {
665         s = 'M ' + (scr[1]) + ' ' + (scr[2]-size) + ' L ' + 
666         (scr[1]-sqrt32) + ' ' + (scr[2]+s05) + ' L ' + 
667         (scr[1]+sqrt32) + ' ' + (scr[2]+s05) + ' Z ';
668     } 
669     else if(type == 'v') {
670         s = 'M ' + (scr[1]) + ' ' + (scr[2]+size) + ' L ' + 
671         (scr[1]-sqrt32) + ' ' + (scr[2]-s05) + ' L ' + 
672         (scr[1]+sqrt32) + ' ' + (scr[2]-s05) + ' Z ';
673     }   
674     else if(type == '>') {
675         s = 'M ' + (scr[1]+size) + ' ' + (scr[2]) + ' L ' + 
676         (scr[1]-s05) + ' ' + (scr[2]-sqrt32) + ' L ' + 
677         (scr[1]-s05) + ' ' + (scr[2]+sqrt32) + ' Z ';
678     }
679     else if(type == '<') {
680         s = 'M ' + (scr[1]-size) + ' ' + (scr[2]) + ' L ' + 
681         (scr[1]+s05) + ' ' + (scr[2]-sqrt32) + ' L ' + 
682         (scr[1]+s05) + ' ' + (scr[2]+sqrt32) + ' Z ';
683     }
684     return s;
685 }

717 JXG.SVGRenderer.prototype.drawVerticalGrid = function(topLeft, bottomRight, gx, board) {
718     var node = this.createPrim('path', 'gridx'),
719         gridArr = '';
720         
721     while(topLeft.scrCoords[1] < bottomRight.scrCoords[1] + gx - 1) { 
722         gridArr += ' M ' + topLeft.scrCoords[1] + ' ' + 0 + ' L ' + topLeft.scrCoords[1] + ' ' + board.canvasHeight+' ';
723         topLeft.setCoordinates(JXG.COORDS_BY_SCREEN, [topLeft.scrCoords[1] + gx, topLeft.scrCoords[2]]);   
724     }
