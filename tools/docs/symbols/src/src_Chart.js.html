 32 JXG.Chart = function(board, parents, attributes) {
 33     this.constructor();
 34     if (parents.length==0) { return; }  // No input data in parentArr
 35     
 36     /**
 37      * Contains lpointers to the various displays.
 38      */
 39     this.elements = [];
 40     
 41     var id = attributes['id'] || '';
 42     var name = attributes['name'] || '';
 43     this.init(board, id, name);
 44     
 45     var x,y,i;
 46     if (parents.length>0 && (typeof parents[0]=='number')) { // parents looks like [a,b,c,..]
 47                                                                 // x has to be filled
 48         y = parents;
 49         x = [];
 50         for (i=0;i<y.length;i++) {
 51             x[i] = i+1;
 52         }
 53     } else {
 54         if (parents.length==1) { // parents looks like [[a,b,c,..]]
 55                                 // x has to be filled
 56             y = parents[0];
 57             x = [];
 58             var len;
 59             if (JXG.isFunction(y)) {
 60                 len = y().length;
 61             } else {
 62                 len = y.length;
 63             }
 64             for (i=0;i<len;i++) {
 65                 x[i] = i+1;
 66             }
 67         }
 68         if (parents.length==2) { // parents looks like [[x0,x1,x2,...],[y1,y2,y3,...]]
 69             x = parents[0];
 70             y = parents[1];
 71         }
 72     }
 73     if (attributes==undefined) attributes = {};
 74     var style = attributes['chartStyle'] || 'line';
 75     style = style.replace(/ /g,'');
 76     style = style.split(',');
 77     var c;
 78     for (i=0;i<style.length;i++) {
 79         switch (style[i]) {
 80             case 'bar':
 81                 c = this.drawBar(board,[x,y],attributes);
 82                 break;
 83             case 'line':
 84                 c = this.drawLine(board, [x, y], attributes);
 85                 break;
 86             case 'fit':
 87                 c = this.drawFit(board, [x, y], attributes);
 88                 break;
 89             case 'spline':
 90                 c = this.drawSpline(board, [x, y], attributes);
 91                 break;
 92             case 'pie':
 93                 c = this.drawPie(board,[y],attributes);
 94                 break;
 95             case 'point':
 96                 c = this.drawPoints(board,[x,y],attributes);
 97                 break;
 98             case 'radar':
 99                 c = this.drawRadar(board,parents,attributes);
100                 break;
101         };
102         this.elements.push(c);
103     };
104     this.id = this.board.addChart(this);
105     return this.elements;
106 };

169 JXG.Chart.prototype.drawBar = function(board, parents, attributes) {
170     var i, pols = [], x = parents[0], y = parents[1], w, xp0,xp1,xp2, yp, ypL, colorArray, p = [], fill;
171     if (attributes['fillOpacity'] == undefined) {
172         attributes['fillOpacity'] = 0.6;
173     }
174     
175     // Determine the width of the bars
176     if (attributes && attributes['width']) {  // width given
177         w = attributes['width'];
178     } else {
179         if (x.length<=1) {
180             w = 1;
181         } else {
182             // Find minimum distance between to bars.
183             w = x[1]-x[0];
184             for (i=1;i<x.length-1;i++) {  
185                 w = (x[i+1]-x[i]<w)?(x[i+1]-x[i]):w;
186             }
187         }
188         w *=0.8;
189     }
190 
191     fill = attributes['fillColor']
192     for (i=0;i<x.length;i++) {        
193         if (JXG.isFunction(x[i])) {  // Not yet
194             xp0 = function() { return x[i]()-w*0.5; };
195             xp1 = function() { return x[i](); };
196             xp2 = function() { return x[i]()+w*0.5; };
197         } else {
198             xp0 = x[i]-w*0.5;
199             xp1 = x[i];
200             xp2 = x[i]+w*0.5;
201         }
202         if (JXG.isFunction(y[i])) {  // Not yet
203             ypL = yp; //function() { return y[i]()*1.1; };
204         } else {
205             ypL = y[i]+0.2;
206         }
207         yp = y[i];
208        
209         if (attributes['dir']=='horizontal') {  // horizontal bars
210             p[0] = board.create('point',[0,xp0], {name:'',fixed:true,visible:false});
211             p[1] = board.create('point',[yp,xp0], {name:'',fixed:true,visible:false});
212             p[2] = board.create('point',[yp,xp2], {name:'',fixed:true,visible:false});
213             p[3] = board.create('point',[0,xp2], {name:'',fixed:true,visible:false});
214             if (attributes['labels'] && attributes['labels'][i]) {
215                 board.create('text',[yp,xp2,attributes['labels'][i]],attributes);
216             }
217         } else { // vertical bars
218             p[0] = board.create('point',[xp0,0], {name:'',fixed:true,visible:false});
219             p[1] = board.create('point',[xp0,yp], {name:'',fixed:true,visible:false});
220             p[2] = board.create('point',[xp2,yp], {name:'',fixed:true,visible:false});
221             p[3] = board.create('point',[xp2,0], {name:'',fixed:true,visible:false});
222             if (attributes['labels'] && attributes['labels'][i]) {
223                 board.create('text',[xp2,yp,attributes['labels'][i]],attributes);
224             }
225         }
226         attributes['withLines'] = false;
227 
228         if(!fill) {
229             colorArray = attributes['colorArray'] || ['#B02B2C','#3F4C6B','#C79810','#D15600','#FFFF88','#C3D9FF','#4096EE','#008C00'];
230             attributes['fillColor'] = colorArray[i%colorArray.length];
231         }
232         pols[i] = board.create('polygon',p,attributes);
233     }
234     this.rendNode = pols[0].rendNode;  // This is needed in setProperty
235 
236     return pols; //[0];  // Not enough! We need pols, but this gives an error in board.setProperty.
237 };

256 JXG.Chart.prototype.drawPie = function(board, parents, attributes) {  // Only 1 array possible as argument 
257     var y = parents[0];
258     if (y.length<=0) { return; }
259     if (typeof y[0] == 'function') { return; } // functions not yet possible
260 
261     var i;
262     var p = [];
263     var arc = [];
264     var s = JXG.Math.Statistics.sum(y);
265     var colorArray = attributes['colorArray'] || ['#B02B2C','#3F4C6B','#C79810','#D15600','#FFFF88','#C3D9FF','#4096EE','#008C00'];
266     var highlightColorArray = attributes['highlightColorArray'] || ['#FF7400'];
267     var la = new Array(y.length);
268     for(i=0; i<y.length; i++) {
269         la[i] = '';
270     }
271     var labelArray = attributes['labelArray'] || la;
272     var radius = attributes['radius'] || 4;
273     var myAtts = {};
274     if (typeof attributes['highlightOnSector']  =='undefined') {
275         attributes['highlightOnSector'] = false;
276     }    
277     myAtts['name'] = attributes['name'];
278     myAtts['id'] = attributes['id'];
279     myAtts['strokeWidth'] = attributes['strokeWidth'] || 1;
280     myAtts['strokeColor'] = attributes['strokeColor'] || 'none';
281     myAtts['straightFirst'] = false;
282     myAtts['straightLast'] = false;
283     myAtts['fillColor'] = attributes['fillColor'] || '#FFFF88';
284     myAtts['fillOpacity'] = attributes['fillOpacity'] || 0.6;
285     myAtts['highlightFillColor'] = attributes['highlightFillColor'] || '#FF7400';
286     myAtts['highlightStrokeColor'] = attributes['highlightStrokeColor'] || '#FFFFFF';
287     myAtts['gradient'] = attributes['gradient'] || 'none';
288     var cent = attributes['center'] || [0,0];
289     var xc = cent[0];
290     var yc = cent[1];
291 
292     var center = board.create('point',[xc,yc], {name:'',fixed:true,visible:false});
293     p[0] = board.create('point',[radius+xc,0+yc], {name:'',fixed:true,visible:false});
294     var rad = 0.0;
295     for (i=0;i<y.length;i++) {
296         rad += (s!=0)?(2*Math.PI*y[i]/s):0;
297         var xcoord = radius*Math.cos(rad)+xc;
298         var ycoord = radius*Math.sin(rad)+yc;
299         p[i+1] = board.create('point',[xcoord,ycoord], {name:'',fixed:true,visible:false,withLabel:false});
300         myAtts['fillColor'] = colorArray[i%colorArray.length];
301         myAtts['name'] = labelArray[i];
302         if(myAtts['name'] != '') {
303             myAtts['withLabel'] = true;
304         }
305         else {
306             myAtts['withLabel'] = false;
307         }
308         myAtts['labelColor'] = colorArray[i%colorArray.length];
309         myAtts['highlightfillColor'] = highlightColorArray[i%highlightColorArray.length];
310         arc[i] = board.create('sector',[center,p[i],p[i+1]], myAtts);
311         
312         if(attributes['highlightOnSector']) {
313             arc[i].hasPoint = arc[i].hasPointSector; // overwrite hasPoint so that the whole sector is used for highlighting
314         }
315 
316     }
317     this.rendNode = arc[0].rendNode;
318     return {arcs:arc, points:p, midpoint:center}; //[0];  // Not enough! We need points, but this gives an error in board.setProperty.
319 };

326 JXG.Chart.prototype.drawRadar = function(board, parents, attributes) { 
327     if (parents.length<=0) { alert("No data"); return; } 
328     // labels for axes
329     var paramArray = attributes['paramArray'];
330     if (paramArray == undefined){ alert("Need paramArray attribute"); return; }
331     var numofparams=paramArray.length;
332     if (numofparams<=1) { alert("Need more than 1 param"); return; }
333 
334     var i,j;
335     for(i=0; i<parents.length; i++) {
336         if (numofparams!=parents[i].length) { alert("Use data length equal to number of params (" + parents[i].length + " != " + numofparams + ")"); return; }
337     }
338     maxes=new Array(numofparams);
339     mins=new Array(numofparams);
340     for(j=0; j<numofparams; j++) {
341         maxes[j] = parents[0][j];
342         mins[j] = maxes[j];
343     }
344     for(i=1; i<parents.length; i++) {
345         for(j=0;j<numofparams; j++) {
346             if (parents[i][j]>maxes[j])
347                 maxes[j] = parents[i][j];
348             if (parents[i][j]<mins[j])
349                 mins[j] = parents[i][j];
350         }
351     }
352     //alert("maxes " + maxes + ", mins " + mins);
353     var la = new Array(parents.length);
354     var pdata = new Array(parents.length);
355     for(i=0; i<parents.length; i++) {
356         la[i] = '';
357         pdata[i] = [];
358     }
359 
360     var ssa = new Array(numofparams);
361     var esa = new Array(numofparams);
362     // 0 <= Offset from chart center <=1
363     var ssratio = attributes['startShiftRatio'] || 0.0;
364     // 0 <= Offset from chart radius <=1
365     var esratio = attributes['endShiftRatio'] || 0.0;
366     for(i=0; i<numofparams; i++) {
367         ssa[i] = (maxes[i]-mins[i])*ssratio;
368         esa[i] = (maxes[i]-mins[i])*esratio;
369     }
370     // Adjust offsets per each axis
371     var sshifts = attributes['startShiftArray'] || ssa;
372     var eshifts = attributes['endShiftArray'] || esa;
373     // Values for inner circle, minimums by default
374     var starts = attributes['startArray'] || mins;
375     if (attributes['start'] != undefined)
376         for(var i=0;i<numofparams; i++) starts[i] = attributes['start'];
377     // Values for outer circle, maximums by default
378     var ends = attributes['endArray'] || maxes;
379     if (attributes['end'] != undefined)
380         for(var i=0;i<numofparams; i++) ends[i] = attributes['end'];
381 
382     if(sshifts.length != numofparams) { alert("Start shifts length is not equal to number of parameters"); return; }
383     if(eshifts.length != numofparams) { alert("End shifts length is not equal to number of parameters"); return; }
384     if(starts.length != numofparams) { alert("Starts length is not equal to number of parameters"); return; }
385     if(ends.length != numofparams) { alert("Ends length is not equal to number of parameters"); return; }
386 
387     //for(i=0; i<numofparams; i++) {
388         //starts[i]-=sshifts[i];
389         //ends[i]+=eshifts[i];
390     //}
391     //alert(starts + ' ' + ends);
392     // labels for legend
393     var labelArray = attributes['labelArray'] || la;
394     var colorArray = attributes['colorArray'] || ['#B02B2C','#3F4C6B','#C79810','#D15600','#FFFF88','#C3D9FF','#4096EE','#008C00'];
395     var highlightColorArray = attributes['highlightColorArray'] || ['#FF7400'];
396     var radius = attributes['radius'] || 10;
397     var myAtts = {};
398     if (typeof attributes['highlightOnSector']  =='undefined') {
399         attributes['highlightOnSector'] = false;
400     }    
401     myAtts['name'] = attributes['name'];
402     myAtts['id'] = attributes['id'];
403     myAtts['strokeWidth'] = attributes['strokeWidth'] || 1;
404     myAtts['polyStrokeWidth'] = attributes['polyStrokeWidth'] || 2*myAtts['strokeWidth'];
405     myAtts['strokeColor'] = attributes['strokeColor'] || 'black';
406     myAtts['straightFirst'] = false;
407     myAtts['straightLast'] = false;
408     myAtts['fillColor'] = attributes['fillColor'] || '#FFFF88';
409     myAtts['fillOpacity'] = attributes['fillOpacity'] || 0.4;
410     myAtts['highlightFillColor'] = attributes['highlightFillColor'] || '#FF7400';
411     myAtts['highlightStrokeColor'] = attributes['highlightStrokeColor'] || 'black';
412     myAtts['gradient'] = attributes['gradient'] || 'none';
413 
414     var cent = attributes['center'] || [0,0];
415     var xc = cent[0];
416     var yc = cent[1];
417 
418     var center = board.createElement('point',[xc,yc], {name:'',fixed:true, withlabel:false,visible:false});
419     var start_angle = Math.PI/2 - Math.PI/numofparams;
420     if(attributes['startAngle'] || attributes['startAngle'] === 0) start_angle = attributes['startAngle'];
421     var rad = start_angle;
422     var p = [];
423     var line = [];
424     var get_anchor =  function() {
425         var x1,x2,y1,y2,relCoords = [].concat(this.labelOffsets);
426         x1 = this.point1.X();
427         x2 = this.point2.X();
428         y1 = this.point1.Y();
429         y2 = this.point2.Y();
430         if(x2<x1)
431             relCoords[0] = -relCoords[0];
432         if(y2<y1)
433             relCoords[1] = -relCoords[1];
434 
435         this.setLabelRelativeCoords(relCoords);
436         return new JXG.Coords(JXG.COORDS_BY_USER, [this.point2.X(),this.point2.Y()],this.board);
437     }
438 
439     var get_transform = function(angle,i) {
440         var t;
441         var tscale;
442         var trot;
443         t = board.createElement('transform', [-(starts[i]-sshifts[i]), 0],{type:'translate'}); 
444         tscale = board.createElement('transform', [radius/((ends[i]+eshifts[i])-(starts[i]-sshifts[i])), 1],{type:'scale'}); 
445         t.melt(tscale);
446         trot = board.createElement('transform', [angle],{type:'rotate'}); 
447         t.melt(trot);
448         return t;
449     }
450     var t;
451     for (i=0;i<numofparams;i++) {
452         rad += 2*Math.PI/numofparams;
453         var xcoord = radius*Math.cos(rad)+xc;
454         var ycoord = radius*Math.sin(rad)+yc;
455 
456         p[i] = board.createElement('point',[xcoord,ycoord], {name:'',fixed:true,withlabel:false,visible:false});
457         line[i] = board.createElement('line',[center,p[i]], 
458             {name:paramArray[i], 
459             strokeColor:myAtts['strokeColor'], strokeWidth:myAtts['strokeWidth'], strokeOpacity:1.0,
460             straightFirst:false, straightLast:false, withLabel:true,
461             highlightStrokeColor:myAtts['highlightStrokeColor']
462         });
463         line[i].getLabelAnchor = get_anchor;
464         t = get_transform(rad,i);
465 
466         for(j=0; j<parents.length; j++) {
467             var data=parents[j][i];
468             pdata[j][i] = board.createElement('point',[data,0], {name:'',fixed:true,withlabel:false,visible:false});
469             pdata[j][i].addTransform(pdata[j][i], t);
470         }
471         //board.unsuspendUpdate();
472         //alert('');
473         /*
474         if(myAtts['name'] != '') {
475             myAtts['withLabel'] = true;
476         }
477         else {
478             myAtts['withLabel'] = false;
479         }
480         */
481 
482     }
483     var polygons = new Array(parents.length);
484     for(i=0;i<parents.length;i++) {
485         myAtts['labelColor'] = colorArray[i%colorArray.length];
486         myAtts['strokeColor'] = colorArray[i%colorArray.length];
487         myAtts['fillColor'] = colorArray[i%colorArray.length];
488         polygons[i] = board.createElement('polygon',pdata[i], 
489                 {withLines:true,
490                 withLabel:false,
491                 fillColor:myAtts['fillColor'],
492                 fillOpacity:myAtts['fillOpacity']
493                 });
494         for(j=0;j<numofparams;j++) {
495             polygons[i].borders[j].setProperty('strokeColor:' + colorArray[i%colorArray.length]);
496             polygons[i].borders[j].setProperty('strokeWidth:' + myAtts['polyStrokeWidth']);
497         }
498     }
499 
500     var legend_position = attributes['legendPosition'] || 'none';
501     switch(legend_position) {
502         case 'right':
503         var lxoff = attributes['legendLeftOffset'] || 2;
504             var lyoff = attributes['legendTopOffset'] || 1;
505             this.legend = board.createElement('legend', [xc+radius+lxoff,yc+radius-lyoff],
506             {labelArray:labelArray,
507             colorArray: colorArray
508             });
509             break;
510         case 'none':
511             break;
512         default:
513             alert('Unknown legend position');
514     }
515 
516     var circles = [];
517     if (attributes['showCircles'] != false) {
518         var cla = [];
519         for(i=0;i<6;i++)
520             cla[i]=20*i;
521         cla[0] = "0";
522         var clabelArray = attributes['circleLabelArray'] || cla;
523         var ncircles = clabelArray.length;
524         if (ncircles<2) {alert("Too less circles"); return; }
525         var pcircles = [];
526         var angle=start_angle + Math.PI/numofparams
527         t = get_transform(angle,0);
528         myAtts['fillColor'] = 'none';
529         myAtts['highlightFillColor'] = 'none';
530         myAtts['strokeColor'] = attributes['strokeColor'] || 'black';
531         myAtts['strokeWidth'] = attributes['circleStrokeWidth'] || 0.5;
532         // we have ncircles-1 intervals between ncircles circles
533         var dr = (ends[0]-starts[0])/(ncircles-1);
534         for(i=0;i<ncircles;i++) {
535             pcircles[i] = board.createElement('point', [starts[0]+i*dr,0],{name:clabelArray[i], size:0,withLabel:true, visible:true});
536             pcircles[i].addTransform(pcircles[i],t);
537             circles[i] = board.createElement('circle', [center,pcircles[i]], myAtts);
538         }
539 
540     }
541     this.rendNode = polygons[0].rendNode;
542     return {circles:circles, lines:line, points:pdata, midpoint:center,polygons:polygons}; //[0];  // Not enough! We need points, but this gives an error in board.setProperty.
543 };

572 JXG.createChart = function(board, parents, attributes) {
573     if((parents.length == 1) && (typeof parents[0] == 'string')) {
574         var table = document.getElementById(parents[0]),
575             data, row, i, j, col, cell, charts = [], w, x, showRows,
576             originalWidth, name, strokeColor, fillColor, hStrokeColor, hFillColor, len;
577         if(typeof table != 'undefined') {
578             // extract the data
579             attributes = JXG.checkAttributes(attributes,{withHeader:true});
580             
581             table = (new JXG.DataSource()).loadFromTable(parents[0], attributes['withHeader'], attributes['withHeader']);
582             data = table.data;
583             col = table.columnHeader;
584             row = table.rowHeader;
585 
586             originalWidth = attributes['width'];
587             name = attributes['name'];
588             strokeColor = attributes['strokeColor'];
589             fillColor = attributes['fillColor'];
590             hStrokeColor = attributes['highlightStrokeColor'];
591             hFillColor = attributes['highlightFillColor'];
592 
593             board.suspendUpdate();
594 
595             len = data.length;
596             showRows = [];
597             if (attributes['rows'] && JXG.isArray(attributes['rows'])) {
598                 for(i=0; i<len; i++) {
599                     for(j=0; j<attributes['rows'].length; j++) {
600                         if((attributes['rows'][j] == i) || (attributes['withHeaders'] && attributes['rows'][j] == row[i])) {
601                             showRows.push(data[i]);
602                             break;
603                         }
604                     }
605                 }
606             } else {
607                 showRows = data;
608             }
609 
610             len = showRows.length;
611 
612             for(i=0; i<len; i++) {
613 
614                 x = [];
615                 if(attributes['chartStyle'] && attributes['chartStyle'].indexOf('bar') != -1) {
616                     if(originalWidth) {
617                         w = originalWidth;
618                     } else {
619                         w = 0.8;
620                     }
621                     x.push(1 - w/2. + (i+0.5)*w/(1.0*len));
622                     for(j=1; j<showRows[i].length; j++) {
623                         x.push(x[j-1] + 1);
624                     }
625                     attributes['width'] = w/(1.0*len);
626                 }
627                 
628                 if(name && name.length == len)
629                     attributes['name'] = name[i];
630                 else if(attributes['withHeaders'])
631                     attributes['name'] = col[i];
632                 
633                 if(strokeColor && strokeColor.length == len)
634                     attributes['strokeColor'] = strokeColor[i];
635                 else
636                     attributes['strokeColor'] = JXG.hsv2rgb(((i+1)/(1.0*len))*360,0.9,0.6);
637                 
638                 if(fillColor && fillColor.length == len)
639                     attributes['fillColor'] = fillColor[i];
640                 else
641                     attributes['fillColor'] = JXG.hsv2rgb(((i+1)/(1.0*len))*360,0.9,1.0);
642                 
643                 if(hStrokeColor && hStrokeColor.length == len)
644                     attributes['highlightStrokeColor'] = hStrokeColor[i];
645                 else
646                     attributes['highlightStrokeColor'] = JXG.hsv2rgb(((i+1)/(1.0*len))*360,0.9,1.0);
647                 
648                 if(hFillColor && hFillColor.length == len)
649                     attributes['highlightFillColor'] = hFillColor[i];
650                 else
651                     attributes['highlightFillColor'] = JXG.hsv2rgb(((i+1)/(1.0*len))*360,0.9,0.6);
652                 
653                 if(attributes['chartStyle'] && attributes['chartStyle'].indexOf('bar') != -1) {
654                     charts.push(new JXG.Chart(board, [x, showRows[i]], attributes));
655                 } else
656                     charts.push(new JXG.Chart(board, [showRows[i]], attributes));
657             }
658 
659             board.unsuspendUpdate();
660 
661         }
662         return charts;
663     } else     
664         return new JXG.Chart(board, parents, attributes);
665 };

667 JXG.JSXGraph.registerElement('chart', JXG.createChart);

673 JXG.Legend = function(board, coords, attributes) {
674     /* Call the constructor of GeometryElement */
675     this.constructor();
676     this.board = board;
677     this.coords = new JXG.Coords(JXG.COORDS_BY_USER, coords, this.board);
678     this.myAtts = {};
679     this.label_array = attributes['labelArray'] || ['1','2','3','4','5','6','7','8'];
680     this.color_array = attributes['colorArray'] || ['#B02B2C','#3F4C6B','#C79810','#D15600','#FFFF88','#C3D9FF','#4096EE','#008C00'];
681     var i;
682     this.lines = [];
683     this.myAtts['strokeWidth'] = attributes['strokeWidth'] || 5;
684     this.myAtts['straightFirst'] = false;
685     this.myAtts['straightLast'] = false;
686     this.myAtts['withLabel'] = true;
687     this.style = attributes['legendStyle'] || 'vertical';
688 
689     switch(this.style) {
690         case 'vertical':
691             this.drawVerticalLegend(attributes); 
692             break;
693         default:
694             alert('Unknown legend style' + this.style);
695             break;
696     }
697 }
698 JXG.Legend.prototype = new JXG.GeometryElement;
699 
700 JXG.Legend.prototype.drawVerticalLegend = function(attributes) {
701     var line_length = attributes['lineLength'] || 1;
702     var offy = (attributes['rowHeight'] || 20)/this.board.stretchY;
703 
704     for(i=0;i<this.label_array.length;i++) {
705         this.myAtts['strokeColor'] = this.color_array[i];
706         this.myAtts['highlightStrokeColor'] = this.color_array[i];
707         this.myAtts['name'] = this.label_array[i];
708         this.myAtts['labelOffsets'] = [10, 0];
709         this.lines[i] = board.createElement('line', 
710                 [[this.coords.usrCoords[1],this.coords.usrCoords[2] - i*offy],
711                 [this.coords.usrCoords[1] + line_length,this.coords.usrCoords[2] - i*offy]],
712                 this.myAtts
713                 );
714         this.lines[i].getLabelAnchor = function() {
715             this.setLabelRelativeCoords(this.labelOffsets);
716             return new JXG.Coords(JXG.COORDS_BY_USER, [this.point2.X(),this.point2.Y()],this.board);
717         }
718     }
719 }

721 JXG.createLegend = function(board, parents, attributes) {
722     //parents are coords of left top point of the legend
723     var start_from = [0,0];
724     if(parents != undefined)
725         if(parents.length == 2) {
726             start_from = parents;
727         }
728     return new JXG.Legend(board, start_from, attributes);
729 }
