 33 /**
 34  * Creates ticks for an axis.
 35  * @class Ticks provides methods for creation and management
 36  * of ticks on an axis.
 37  * @param {JXG.Line} line Reference to the axis the ticks are drawn on.
 38  * @param {Number,Array,Function} ticks Number, array or function defining the ticks.
 39  * @param {int} major Every major-th tick is drawn with heightmajorHeight, the other ones are drawn with height minorHeight.
 40  * @param {int} majorHeight The height used to draw major ticks.
 41  * @param {int} minorHeight The height used to draw minor ticks.
 42  * @param {String} id Unique identifier for this object.  If null or an empty string is given,
 43  * an unique id will be generated by Board.
 44  * @param {String} name Not necessarily unique name, won't be visible or used by this object.
 45  * @see JXG.Board#addTicks
 46  * @constructor
 47  * @extends JXG.GeometryElement
 48  */
 49 JXG.Ticks = function (line, ticks, minor, majorHeight, minorHeight, id, name, layer) {
 50     /* Call the constructor of GeometryElement */
 51     this.constructor();
 52 
 53     /**
 54      * Type of GeometryElement, value is OBJECT_TYPE_ARC.
 55      * @final
 56      * @type int
 57      */
 58     this.type = JXG.OBJECT_TYPE_TICKS;
 59 
 60     /**
 61      * Class of the element, value is OBJECT_CLASS_CIRCLE.
 62      * @final
 63      * @type int
 64      */
 65     this.elementClass = JXG.OBJECT_CLASS_OTHER;
 66 
 67     /**
 68      * Set the display layer.
 69      */
 70     //if (layer == null) layer = board.options.layer['line']; // no board available
 71     //this.layer = layer;
 72 
 73     /**
 74      * The line the ticks belong to.
 75      * @type JXG.Line
 76      */
 77     this.line = line;
 78 
 79     /**
 80      * The board the ticks line is drawn on.
 81      * @type JXG.Board
 82      */
 83     this.board = this.line.board;
 84 
 85     /**
 86      * A function calculating ticks delta depending on the ticks number.
 87      * @type Function
 88      */
 89     this.ticksFunction = null;
 90 
 91     /**
 92      * Array of fixed ticks.
 93      * @type Array
 94      */
 95     this.fixedTicks = null;
 96 
 97     /**
 98      * Equidistant ticks. Distance is defined by ticksFunction
 99      * @type bool
100      */
101     this.equidistant = false;
102 
103     if(JXG.isFunction(ticks)) {
104         this.ticksFunction = ticks;
105     	throw new Error("Function arguments are no longer supported.");
106     } else if(JXG.isArray(ticks))
107         this.fixedTicks = ticks;
108     else {
109         if(Math.abs(ticks) < JXG.Math.eps)
110             ticks = this.board.options.line.ticks.defaultDistance;
111         this.ticksFunction = function (i) { return ticks; };
112         this.equidistant = true;
113     }
114 
115     /**
116      * minorTicks is the number of minor ticks between two major ticks.
117      * @type int
118      */
119     this.minorTicks = ( (minor == null)? this.board.options.line.ticks.minorTicks : minor);
120     if(this.minorTicks < 0)
121         this.minorTicks = -this.minorTicks;
122 
123     /**
124      * Total height of a major tick.
125      * @type int
126      */
127     this.majorHeight = ( (majorHeight == null) || (majorHeight == 0) ? this.board.options.line.ticks.majorHeight : majorHeight);
128     if(this.majorHeight < 0)
129         this.majorHeight = -this.majorHeight;
130 
131     /**
132      * Total height of a minor tick.
133      * @type int
134      */
135     this.minorHeight = ( (minorHeight == null) || (minorHeight == 0) ? this.board.options.line.ticks.minorHeight : minorHeight);
136     if(this.minorHeight < 0)
137         this.minorHeight = -this.minorHeight;
138 
139     /**
140      * Least distance between two ticks, measured in pixels.
141      * @type int
142      */
143     this.minTicksDistance = this.board.options.line.ticks.minTicksDistance;
144 
145     /**
146      * Maximum distance between two ticks, measured in pixels. Is used only when insertTicks
147      * is set to true.
148      * @type int
149      * @see #insertTicks
150      */
151     this.maxTicksDistance = this.board.options.line.ticks.maxTicksDistance;
152 
153     /**
154      * If the distance between two ticks is too big we could insert new ticks. If insertTicks
155      * is <tt>true</tt>, we'll do so, otherwise we leave the distance as is.
156      * This option is ignored if equidistant is false.
157      * @type bool
158      * @see #equidistant
159      * @see #maxTicksDistance
160      */
161     this.insertTicks = this.board.options.line.ticks.insertTicks;
162 
163     /**
164      * Draw the zero tick, that lies at line.point1?
165      * @type bool
166      */
167     this.drawZero = this.board.options.line.ticks.drawZero;
168 
169     /**
170      * Draw labels yes/no
171      * @type bool
172      */
173     this.drawLabels = this.board.options.line.ticks.drawLabels;
174 
175     /**
176      * Array where the labels are saved. There is an array element for every tick,
177      * even for minor ticks which don't have labels. In this case the array element
178      * contains just <tt>null</tt>.
179      * @type array
180      */
181     this.labels = [];
182 
183     /* Call init defined in GeometryElement to set board, id and name property */
184     this.init(this.board, id, name);
185 
186     this.visProp['visible'] = true;
187 
188     this.visProp['fillColor'] = this.line.visProp['fillColor'];
189     this.visProp['highlightFillColor'] = this.line.visProp['highlightFillColor'];
190     this.visProp['strokeColor'] = this.line.visProp['strokeColor'];
191     this.visProp['highlightStrokeColor'] = this.line.visProp['highlightStrokeColor'];
192     this.visProp['strokeWidth'] = this.line.visProp['strokeWidth'];
193 
194     /* Register ticks at line. */
195     this.id = this.line.addTicks(this);
196 };

206 JXG.Ticks.prototype.hasPoint = function (x, y) {

210 /**
211  * (Re-)calculates the ticks coordinates.
212  */
213 JXG.Ticks.prototype.calculateTicksCoordinates = function() {
214 	
215 	/*
216 	 * 
217 	 * It's all new in here but works pretty well.
218 	 * Known bugs:
219 	 *   * Special ticks behave oddly. See example ticked_lines.html and drag P2 around P1.
220 	 * 
221 	 */
222 	
223 	
224 	    // Point 1 of the line
225 	var p1 = this.line.point1,
226 		// Point 2 of the line
227 		p2 = this.line.point2,
228 		// Distance between the two points from above
229 		distP1P2 = p1.coords.distance(JXG.COORDS_BY_USER, p2.coords),
230 		// Distance of X coordinates of two major ticks
231 		// Initialized with the distance of Point 1 to a point between Point 1 and Point 2 on the line and with distance 1
232 		deltaX = (p2.coords.usrCoords[1] - p1.coords.usrCoords[1])/distP1P2,
233 		// The same thing for Y coordinates
234 		deltaY = (p2.coords.usrCoords[2] - p1.coords.usrCoords[2])/distP1P2,
235 		// Distance of p1 to the unit point in screen coordinates
236 		distScr = p1.coords.distance(JXG.COORDS_BY_SCREEN, new JXG.Coords(JXG.COORDS_BY_USER, [p1.coords.usrCoords[1] + deltaX, p1.coords.usrCoords[2] + deltaY], this.board)),
237 		// Distance between two major ticks in user coordinates
238 		ticksDelta = (this.equidistant ? this.ticksFunction(1) : 1),
239 		// This factor is for enlarging ticksDelta and it switches between 5 and 2
240 		// Hence, if two major ticks are too close together they'll be expanded to a distance of 5
241 		// if they're still too close together, they'll be expanded to a distance of 10 etc
242 		factor = 5,
243 		// Edge points: This is where the display of the line starts and ends, e.g. the intersection points
244 		// of the line with the edges of the viewing area if the line is a straight.
245 		e1, e2,
246 		// Which direction do we go? Plus or Minus
247 		dir = 1,
248 		// what's the first/last tick to draw?
249 		begin, end,
250 		// Coordinates of the current tick
251 		tickCoords,
252 		// Coordinates of the first drawn tick
253 		startTick,
254 		// a counter
255 		i,
256 		// the distance of the tick to p1. Is displayed on the board using a label
257 		// for majorTicks
258 		tickPosition,
259 		// creates a label
260 		makeLabel = function(pos, newTick, board, drawLabels, id) {
261 			var labelText, label;
262 			
263         	labelText = pos.toString();
264         	if(labelText.length > 5)
265             	labelText = pos.toPrecision(3).toString();
266         	
267         	label = new JXG.Text(board, labelText, null, [newTick.usrCoords[1], newTick.usrCoords[2]], id+i+"Label", '', null, true, board.options.text.defaultDisplay);
268         	label.distanceX = 0;
269         	label.distanceY = -10;
270         	label.setCoords(newTick.usrCoords[1]*1+label.distanceX/(board.stretchX), 
271                         	newTick.usrCoords[2]*1+label.distanceY/(board.stretchY));
272         	
273         	if (drawLabels) {
274             	label.visProp['visible'] = true; 
275         	} else {
276             	label.visProp['visible'] = false;
277         	}
278         	return label;
279 		},
280 		
281 		respDelta = function(val) {
282 			return Math.floor(val) - (Math.floor(val) % ticksDelta);
283 		},
284 		
285 		// the following variables are used to define ticks height and slope
286 	    eps = JXG.Math.eps,
287 	    slope = -this.line.getSlope(),
288 	    distMaj = this.majorHeight/2,
289 	    distMin = this.minorHeight/2,
290 	    dxMaj = 0, dyMaj = 0,
291 	    dxMin = 0, dyMin = 0;
292 		
293 	// END OF variable declaration
294 		
295 
296     // this piece of code used to be in AbstractRenderer.updateAxisTicksInnerLoop
297     // and has been moved in here to clean up the renderers code.
298     //
299     // The code above only calculates the position of the ticks. The following code parts
300     // calculate the dx and dy values which make ticks out of this positions, i.e. from the
301     // position (p_x, p_y) calculated above we have to draw a line from
302     // (p_x - dx, py - dy) to (p_x + dx, p_y + dy) to get a tick.
303 
304     if(Math.abs(slope) < eps) {
305         // if the slope of the line is (almost) 0, we can set dx and dy directly
306         dxMaj = 0;
307         dyMaj = distMaj;
308         dxMin = 0;
309         dyMin = distMin;
310     } else if((Math.abs(slope) > 1/eps) || (isNaN(slope))) {
311         // if the slope of the line is (theoretically) infinite, we can set dx and dy directly
312         dxMaj = distMaj;
313         dyMaj = 0;
314         dxMin = distMin;
315         dyMin = 0;
316     } else {
317         // here we have to calculate dx and dy depending on the slope and the length of the tick (dist)
318         // if slope is the line's slope, the tick's slope is given by
319         //
320         //            1          dy
321         //     -   -------  =   ----                 (I)
322         //          slope        dx
323         //
324         // when dist is the length of the tick, using the pythagorean theorem we get
325         //
326         //     dx*dx + dy*dy = dist*dist             (II)
327         //
328         // dissolving (I) by dy and applying that to equation (II) we get the following formulas for dx and dy
329         dxMaj = -distMaj/Math.sqrt(1/(slope*slope) + 1);
330         dyMaj = dxMaj/slope;
331         dxMin = -distMin/Math.sqrt(1/(slope*slope) + 1);
332         dyMin = dxMin/slope;
333     }
334 
335     // BEGIN: clean up the mess we left from our last run through this function
336     // remove existing ticks
337     if(this.ticks != null) {
338         for(var j=0; j<this.ticks.length; j++) {
339             if(this.labels[j] != null) {
340                 if (this.labels[j].visProp['visible']) this.board.renderer.remove(this.labels[j].rendNode);
341             }
342         }
343     }
344 
345     // initialize storage arrays
346     // ticks stores the ticks coordinates
347     this.ticks = new Array();
348     
349     // labels stores the text to display beside the ticks
350     this.labels = new Array();
351     // END cleanup
352     
353 	// calculate start (e1) and end (e2) points
354     // for that first copy existing lines point coordinates...
355     e1 = new JXG.Coords(JXG.COORDS_BY_USER, [p1.coords.usrCoords[1], p1.coords.usrCoords[2]], this.board);
356     e2 = new JXG.Coords(JXG.COORDS_BY_USER, [p2.coords.usrCoords[1], p2.coords.usrCoords[2]], this.board);
357 		
358 	// ... and calculate the drawn start and end point
359     this.board.renderer.calcStraight(this.line, e1, e2);
360 	    
361 	if(!this.equidistant) {
362         // we have an array of fixed ticks we have to draw
363         var dx_minus = p1.coords.usrCoords[1]-e1.usrCoords[1];
364         var dy_minus = p1.coords.usrCoords[2]-e1.usrCoords[2];
365         var length_minus = Math.sqrt(dx_minus*dx_minus + dy_minus*dy_minus);
366 
367         var dx_plus = p1.coords.usrCoords[1]-e2.usrCoords[1];
368         var dy_plus = p1.coords.usrCoords[2]-e2.usrCoords[2];
369         var length_plus = Math.sqrt(dx_plus*dx_plus + dy_plus*dy_plus);
370 
371         // new ticks coordinates
372         var nx = 0;
373         var ny = 0;
374 
375         for(var i=0; i<this.fixedTicks.length; i++) {
376             // is this tick visible?
377             if((-length_minus <= this.fixedTicks[i]) && (this.fixedTicks[i] <= length_plus)) {
378                 if(this.fixedTicks[i] < 0) {
379                     nx = Math.abs(dx_minus) * this.fixedTicks[i]/length_minus;
380                     ny = Math.abs(dy_minus) * this.fixedTicks[i]/length_minus;
381                 } else {
382                     nx = Math.abs(dx_plus) * this.fixedTicks[i]/length_plus;
383                     ny = Math.abs(dy_plus) * this.fixedTicks[i]/length_plus;
384                 }
385 
386                 tickCoords = new JXG.Coords(JXG.COORDS_BY_USER, [p1.coords.usrCoords[1] + nx, p1.coords.usrCoords[2] + ny], this.board);
387                 this.ticks.push(tickCoords);
388                 this.ticks[this.ticks.length-1].major = true;
389                 
390         		this.labels.push(makeLabel(this.fixedTicks[i], tickCoords, this.board, this.drawLabels, this.id));
391             }
392         }
393         
394         this.board.renderer.updateTicks(this, dxMaj, dyMaj, dxMin, dyMin);
395     	return;
396 	} // ok, we have equidistant ticks and not special ticks, so we continue here with generating them:
397 	
398 	// adjust distances
399     while(distScr > 4*this.minTicksDistance) {
400         ticksDelta /= 10;
401         deltaX /= 10;
402         deltaY /= 10;
403 
404         distScr = p1.coords.distance(JXG.COORDS_BY_SCREEN, new JXG.Coords(JXG.COORDS_BY_USER, [p1.coords.usrCoords[1] + deltaX, p1.coords.usrCoords[2] + deltaY], this.board));
405     }
406 
407     // If necessary, enlarge ticksDelta
408     while(distScr < this.minTicksDistance) {
409         ticksDelta *= factor;
410         deltaX *= factor;
411         deltaY *= factor;
412 
413         factor = (factor == 5 ? 2 : 5);
414         distScr = p1.coords.distance(JXG.COORDS_BY_SCREEN, new JXG.Coords(JXG.COORDS_BY_USER, [p1.coords.usrCoords[1] + deltaX, p1.coords.usrCoords[2] + deltaY], this.board));
415     }
416 
417     /*
418      * In the following code comments are sometimes talking about "respect ticksDelta". this could be done
419      * by calculating the modulus of the distance wrt to ticksDelta and add resp. subtract a ticksDelta from that.
420      */
421 
422     // p1 is outside the visible area or the line is a segment
423     if(this.board.renderer.isSameDirection(p1.coords, e1, e2)) {
424     	// calculate start and end points
425 		begin = respDelta(p1.coords.distance(JXG.COORDS_BY_USER, e1));
426 		end = p1.coords.distance(JXG.COORDS_BY_USER, e2);
427 		
428 		if(this.board.renderer.isSameDirection(p1.coords, p2.coords, e1)) {
429             if(this.line.visProp.straightFirst)
430     			begin -=  2*ticksDelta;
431 		} else {
432 			end = -1*end;
433 			begin = -1*begin;
434             if(this.line.visProp.straightFirst)
435                 begin -= 2*ticksDelta
436 		}
437     	
438     	// TODO: We should check here if the line is visible at all. If it's not visible but
439     	// close to the viewport there may be drawn some ticks without a line visible.
440     	
441     } else {
442     	// p1 is inside the visible area and direction is PLUS
443 
444     	// now we have to calculate the index of the first tick
445         if(!this.line.visProp.straightFirst) {
446         	begin = 0; 
447         } else {
448         	begin = -respDelta(p1.coords.distance(JXG.COORDS_BY_USER, e1)) - 2*ticksDelta;
449         }
450         
451         if(!this.line.visProp.straightLast) {
452         	end = distP1P2;
453         } else {
454         	end = p1.coords.distance(JXG.COORDS_BY_USER, e2);
455         }
456     }
457 
458     startTick = new JXG.Coords(JXG.COORDS_BY_USER, [p1.coords.usrCoords[1] + begin*deltaX/ticksDelta, p1.coords.usrCoords[2] + begin*deltaY/ticksDelta], this.board);
459     tickCoords = new JXG.Coords(JXG.COORDS_BY_USER, [p1.coords.usrCoords[1] + begin*deltaX/ticksDelta, p1.coords.usrCoords[2] + begin*deltaY/ticksDelta], this.board);
460     
461     deltaX /= this.minorTicks+1;
462     deltaY /= this.minorTicks+1;
463     
464 //    JXG.debug('begin: ' + begin + '; e1: ' + e1.usrCoords[1] + ', ' + e1.usrCoords[2]);
465 //    JXG.debug('end: ' + end + '; e2: ' + e2.usrCoords[1] + ', ' + e2.usrCoords[2]);
466     
467     
468     // After all the precalculations from above here finally comes the tick-production:
469     i = 0;
470     tickPosition = begin;
471     while(startTick.distance(JXG.COORDS_BY_USER, tickCoords) < Math.abs(end - begin) + JXG.Math.eps) {
472     	if(i % (this.minorTicks+1) == 0) {
473     		tickCoords.major = true;
474     		this.labels.push(makeLabel(tickPosition, tickCoords, this.board, this.drawLabels, this.id));
475     		tickPosition += ticksDelta;
476     	} else {
477     		tickCoords.major = false;
478     		this.labels.push(null);
479     	}
480     	i++;
481 
482     	this.ticks.push(tickCoords);
483     	tickCoords = new JXG.Coords(JXG.COORDS_BY_USER, [tickCoords.usrCoords[1] + deltaX, tickCoords.usrCoords[2] + deltaY], this.board);
484     	if(!this.drawZero && tickCoords.distance(JXG.COORDS_BY_USER, p1.coords) <= JXG.Math.eps) {
485     		// zero point is always a major tick. hence, we have to set i = 0;
486     		i++;
487     		tickPosition += ticksDelta;
488         	tickCoords = new JXG.Coords(JXG.COORDS_BY_USER, [tickCoords.usrCoords[1] + deltaX, tickCoords.usrCoords[2] + deltaY], this.board);
489     	}
490     }
491 
492     this.board.renderer.updateTicks(this, dxMaj, dyMaj, dxMin, dyMin);
493 	return;
494 };

496 /**
497  * Uses the boards renderer to update the arc.
498  * update() is not needed for arc.
499  */
500 JXG.Ticks.prototype.updateRenderer = function () {
501     if (this.needsUpdate) {
502         this.calculateTicksCoordinates();
503         this.needsUpdate = false;
504     }

507 /**
508  * Creates new ticks.
509  * @param {JXG.Board} board The board the ticks are put on.
510  * @param {Array} parents Array containing a line and an array of positions, where ticks should be put on that line or
511  *   a function that calculates the distance based on the ticks number that is given as a parameter. E.g.:<br />
512  *   <tt>var ticksFunc = function(i) {</tt><br />
513  *   <tt>    return 2;</tt><br />
514  *   <tt>}</tt><br />
515  *   for ticks with distance 2 between each tick.
516  * @param {Object} attributs Object containing properties for the element such as stroke-color and visibility. See @see JXG.GeometryElement#setProperty
517  * @type JXG.Ticks
518  * @return Reference to the created ticks object.
519  */
520 JXG.createTicks = function(board, parents, attributes) {
521     var el;
522     attributes = JXG.checkAttributes(attributes,{layer:null});
523     if ( (parents[0].elementClass == JXG.OBJECT_CLASS_LINE) && (JXG.isFunction(parents[1]) || JXG.isArray(parents[1]) || JXG.isNumber(parents[1]))) {
524         el = new JXG.Ticks(parents[0], parents[1], attributes['minorTicks'], attributes['majHeight'], attributes['minHeight'], attributes['id'], attributes['name'], attributes['layer']);
525     } else
526         throw new Error("JSXGraph: Can't create Ticks with parent types '" + (typeof parents[0]) + "' and '" + (typeof parents[1]) + "' and '" + (typeof parents[2]) + "'.");
527 
528     return el;
529 };
