 30 /**
 31  * Curves are the common object for function graphs, parametric curves, polar curves, adn data plots.
 32  * @class Creates a new curve object. Do not use this constructor to create a curve. Use {@link JXG.Board#create} with
 33  * type {@link Curve}, or {@link Functiongraph} instead.  
 34  * @augments JXG.GeometryElement
 35  * @param {string,JXG.Board} board The board the new curve is drawn on.
 36  * @param {Array} defining terms An array with the functon terms, data points of the curve.
 37  * @param {String} id Unique identifier for the point. If null or an empty string is given,
 38  *  an unique id will be generated by Board
 39  * @param {String} name Not necessarily unique name for the point. If null or an
 40  *  empty string is given, an unique name will be generated
 41  * @param {boolean} show False if the point is invisible, True otherwise
 42  * @see JXG.Board#generateName
 43  * @see JXG.Board#addCurve
 44   */
 45 JXG.Curve = function (board, parents, id, name, withLabel, layer) {
 46     this.constructor();
 47  
 48     this.points = []; 
 49 
 50     this.type = JXG.OBJECT_TYPE_CURVE;
 51     this.elementClass = JXG.OBJECT_CLASS_CURVE;                
 52     
 53     this.init(board, id, name);
 54 
 55     /**
 56      * Set the display layer.
 57      */
 58     if (layer == null) layer = board.options.layer['curve'];
 59     this.layer = layer;
 60 
 61     /** Use the algorithm by Gillam and Hohenwarter for plotting.
 62       * If false the naive algorithm is used.
 63       * It is much slower, but the result is better.
 64       */
 65     this.doAdvancedPlot = this.board.options.curve.doAdvancedPlot;
 66     
 67     /** 
 68       * Number of points on curves after mouseUp, i.e. high quality output.
 69       * Only used if this.doAdvancedPlot==false
 70       * May be overwritten.
 71       **/
 72     this.numberPointsHigh = this.board.options.curve.numberPointsHigh;
 73     /** 
 74       * Number of points on curves after mousemove, i.e. low quality output.
 75       * Only used if this.doAdvancedPlot==false
 76       * May be overwritten.
 77       **/
 78     this.numberPointsLow = this.board.options.curve.numberPointsLow;
 79     /** 
 80       * Number of points on curves. This value changes
 81       * between numberPointsLow and numberPointsHigh.
 82       * It is set in {@link #updateCurve}.
 83       */
 84     this.numberPoints = this.numberPointsHigh; 
 85 
 86     this.visProp['strokeWidth'] = this.board.options.curve.strokeWidth;
 87     this.visProp['highlightStrokeWidth'] = this.visProp['strokeWidth'];
 88 
 89     this.visProp['visible'] = true;
 90     this.dataX = null;
 91     this.dataY = null;
 92 
 93     /**
 94      * This is just for the hasPoint() method.
 95      * @type int
 96      */
 97     //this.r = this.board.options.precision.hasPoint;
 98     
 99     /**
100      * The curveType is set in @see generateTerm and used in 
101      * {@link updateCurve}
102      * Possible values are:
103      * 'none'
104      * 'plot': Data plot
105      * 'parameter': we can not distinguish function graphs and parameter curves
106      * 'functiongraph': function graph
107      * 'polar'
108      * 'implicit' (not yet)
109      *
110      * Only parameter and plot are set directly.
111      * polar is set with setProperties only.
112      **/
113     // this.curveType = 'none';
114     this.curveType = null;
115 
116     if (parents[0]!=null) {
117         this.varname = parents[0];
118     } else {
119         this.varname = 'x';
120     }
121     this.xterm = parents[1];  // function graphs: "x"
122     this.yterm = parents[2];  // function graphs: e.g. "x^2"
123     this.generateTerm(this.varname,this.xterm,this.yterm,parents[3],parents[4]);  // Converts GEONExT syntax into JavaScript syntax
124     this.updateCurve();                        // First evaluation of the curve
125     
126     this.createLabel(withLabel);
127     this.id = this.board.addCurve(this);
128     
129     if (typeof this.xterm=='string') {
130         this.notifyParents(this.xterm);
131     }
132     if (typeof this.yterm=='string') {
133         this.notifyParents(this.yterm);
134     }
135 };

138 /**
139  * Gives the default value of the left bound for the curve.
140  * May be overwritten in @see generateTerm.
141  */
142 JXG.Curve.prototype.minX = function () {
143     if (this.curveType=='polar') {
144         return 0.0;
145     } else {
146         var leftCoords = new JXG.Coords(JXG.COORDS_BY_SCREEN, [0, 0], this.board);
147         return leftCoords.usrCoords[1];
148     }

151 /**
152  * Gives the default value of the right bound for the curve.
153  * May be overwritten in @see generateTerm.
154  */
155 JXG.Curve.prototype.maxX = function () {
156     var rightCoords;
157     if (this.curveType=='polar') {
158         return 2.0*Math.PI;
159     } else {
160         rightCoords = new JXG.Coords(JXG.COORDS_BY_SCREEN, [this.board.canvasWidth, 0], this.board);
161         return rightCoords.usrCoords[1];
162     }

165 /**
166  * Checks whether (x,y) is near the curve.
167  * @param {int} x Coordinate in x direction, screen coordinates.
168  * @param {int} y Coordinate in y direction, screen coordinates.
169  * @param {y} Find closest point on the curve to (x,y)
170  * @return {bool} True if (x,y) is near the curve, False otherwise.
171  */
172 JXG.Curve.prototype.hasPoint = function (x,y) {
173     var t, dist = Infinity, 
174         c, trans, i, j, tX, tY,
175         xi, xi1, yi, yi1,
176         lbda, x0, y0, x1, y1, xy, den,
177         steps = this.numberPointsLow, 
178         d = (this.maxX()-this.minX())/steps,
179         prec = this.board.options.precision.hasPoint/(this.board.unitX*this.board.zoomX),
180         checkPoint, len,
181         suspendUpdate = true;
182 
183     prec = prec*prec;
184     checkPoint = new JXG.Coords(JXG.COORDS_BY_SCREEN, [x,y], this.board);
185     x = checkPoint.usrCoords[1];
186     y = checkPoint.usrCoords[2];
187     if (this.curveType=='parameter' || this.curveType=='polar' || this.curveType=='functiongraph') { 
188         // Brute fore search for a point on the curve close to the mouse pointer
189         len = this.transformations.length;
190         for (i=0,t=this.minX(); i<steps; i++) {
191             tX = this.X(t,suspendUpdate);
192             tY = this.Y(t,suspendUpdate);
193             for (j=0; j<len; j++) {
194                 trans = this.transformations[j];
195                 trans.update();
196                 c = JXG.Math.matVecMult(trans.matrix,[1,tX,tY]);
197                 tX = c[1];
198                 tY = c[2];
199             }
200             dist = (x-tX)*(x-tX)+(y-tY)*(y-tY);
201             if (dist<prec) { return true; }
202             t+=d;
203         }  
204     } else if (this.curveType == 'plot') {
205         //$('debug').innerHTML +='. ';
206         len = this.numberPoints; // Rough search quality
207         for (i=0;i<len-1;i++) {
208             xi = this.X(i);
209             xi1 = this.X(i+1);
210             //if (i!=xi) {
211             //    yi = this.Y(xi);
212             //    yi1 = this.Y(xi1);
213             //} else {
214                 yi = this.Y(i);
215                 yi1 = this.Y(i+1);
216                // $('debug').innerHTML = this.Y.toString();
217             //}
218             x1 = xi1 - xi;
219             y1 = yi1-yi;
220             
221             x0 = x-xi; //this.X(i);
222             y0 = y-yi; //this.Y(i);
223             den = x1*x1+y1*y1;
224             
225             if (den>=JXG.Math.eps) {
226                 xy = x0*x1+y0*y1;
227                 lbda = xy/den;
228                 dist = x0*x0+y0*y0 - lbda*xy;
229             } else {
230                 lbda = 0.0;
231                 dist = x0*x0+y0*y0;
232             }
233             if (lbda>=0.0 && lbda<=1.0 && dist<prec) { 
234                 return true; 
235             } 
236         }
237         return false;
238     } 
239     return (dist<prec);
240 };

246     var i, len;
247     len = this.numberPoints;
248     if (this.points.length<this.numberPoints) {
249         for (i=this.points.length; i<len; i++) {
250             this.points[i] = new JXG.Coords(JXG.COORDS_BY_USER, [0,0], this.board);
251         }
252     }

298 /**
299  * Computes for equidistant points on the x-axis the values
300  * of the function. @see #update
301  * If the mousemove event triggers this update, we use only few
302  * points. Otherwise, e.g. on mouseup, many points are used.
303  */
304 JXG.Curve.prototype.updateCurve = function () {
305     var len, mi, ma, x, y, i,
306         suspendUpdate = false;
307     
308     this.updateDataArray();
309     mi = this.minX();
310     ma = this.maxX();
311 
312     // Discrete data points
313     if (this.dataX!=null) { // x-coordinates are in an array
314         this.numberPoints = this.dataX.length;
315         len = this.numberPoints;
316         this.allocatePoints();  // It is possible, that the array length has increased.
317         for (i=0; i<len; i++) {
318             x = i;
319             if (this.dataY!=null) { // y-coordinates are in an array
320                 y = i;
321             } else {
322                 y = this.X(x); // discrete x data, continuous y data
323             }
324             this.points[i].setCoordinates(JXG.COORDS_BY_USER, [this.X(x,suspendUpdate),this.Y(y,suspendUpdate)], false); // The last parameter prevents rounding in usr2screen().
325             this.updateTransform(this.points[i]);
326             suspendUpdate = true;
327         }
328     } else { // continuous x data
329         if (this.doAdvancedPlot) {
330             this.updateParametricCurve(mi,ma,len);
331         } else {
332             if (this.board.updateQuality==this.board.BOARD_QUALITY_HIGH) {
333                 this.numberPoints = this.numberPointsHigh;
334             } else {
335                 this.numberPoints = this.numberPointsLow;
336             }
337             len = this.numberPoints;
338             this.allocatePoints();  // It is possible, that the array length has increased.
339             this.updateParametricCurveNaive(mi,ma,len);
340         }
341     }
342     this.getLabelAnchor();
343     return this;
344 };

346 JXG.Curve.prototype.updateParametricCurveNaive = function(mi,ma,len) {
347     var i, t,
348         suspendUpdate = false,
349         stepSize = (ma-mi)/len;
350         
351     for (i=0; i<len; i++) {
352         t = mi+i*stepSize;
353         this.points[i].setCoordinates(JXG.COORDS_BY_USER, [this.X(t,suspendUpdate),this.Y(t,suspendUpdate)], false); // The last parameter prevents rounding in usr2screen().
354         this.updateTransform(this.points[i]);
355         suspendUpdate = true;
356     }
357     return this;
358 };

360 JXG.Curve.prototype.updateParametricCurve = function(mi,ma,len) {
361     var i, t, t0,
362         suspendUpdate = false,
363         po = new JXG.Coords(JXG.COORDS_BY_USER, [0,0], this.board),
364         x, y, x0, y0, top, depth,
365         MAX_DEPTH,
366         MAX_XDIST,
367         MAX_YDIST,
368         dyadicStack = [],
369         depthStack = [],
370         pointStack = [],
371         divisors = [], 
372         //xd_ = NaN, yd_ = NaN,
373         distOK = false,
374         j = 0;
375 
376     
377     if (this.board.updateQuality==this.board.BOARD_QUALITY_LOW) {
378         MAX_DEPTH = 12;
379         MAX_XDIST = 12;
380         MAX_YDIST = 12;
381     } else {
382         MAX_DEPTH = 20;
383         MAX_XDIST = 2;
384         MAX_YDIST = 2;
385     }
386     
387     divisors[0] = ma-mi;
388     for (i=1;i<MAX_DEPTH;i++) {
389         divisors[i] = divisors[i-1]*0.5;
390     }
391     
392     i = 1;
393     dyadicStack[0] = 1;
394     depthStack[0] = 0;
395     t = mi;
396     po.setCoordinates(JXG.COORDS_BY_USER, [this.X(t,suspendUpdate),this.Y(t,suspendUpdate)], false);
397     suspendUpdate = true;
398     x0 = po.scrCoords[1];
399     y0 = po.scrCoords[2];
400     t0 = t;
401     
402     t = ma;
403     po.setCoordinates(JXG.COORDS_BY_USER, [this.X(t,suspendUpdate),this.Y(t,suspendUpdate)], false);
404     x = po.scrCoords[1];
405     y = po.scrCoords[2];
406     
407     pointStack[0] = [x,y];
408     
409     top = 1;
410     depth = 0;
411 
412     this.points = [];
413     this.points[j++] = new JXG.Coords(JXG.COORDS_BY_SCREEN, [x0, y0], this.board);
414     
415     do {
416         distOK = this.isDistOK(x0,y0,x,y,MAX_XDIST,MAX_YDIST)||this.isSegmentOutside(x0,y0,x,y);
417         while ( depth<MAX_DEPTH &&
418                (!distOK || depth<3 /*|| (j>1 &&!this.bendOK(xd_,yd_,x-x0,y-y0))*/) &&
419                !(!this.isSegmentDefined(x0,y0,x,y) && depth>8)
420             ) {
421             dyadicStack[top] = i;
422             depthStack[top] = depth;
423             pointStack[top] = [x,y];
424             top++;
425             
426             i = 2*i-1;
427             depth++;
428             t = mi+i*divisors[depth];
429             po.setCoordinates(JXG.COORDS_BY_USER, [this.X(t,suspendUpdate),this.Y(t,suspendUpdate)], false);
430             x = po.scrCoords[1];
431             y = po.scrCoords[2];
432             distOK = this.isDistOK(x0,y0,x,y,MAX_XDIST,MAX_YDIST)||this.isSegmentOutside(x0,y0,x,y);
433         }
434         /*
435         if (this.board.updateQuality==this.board.BOARD_QUALITY_HIGH && !this.isContinuous(t0,t,MAX_DEPTH)) {
436             //$('debug').innerHTML += 'x ';
437             this.points[j] = new JXG.Coords(JXG.COORDS_BY_SCREEN, [NaN, NaN], this.board);
438             //this.points[j] = new JXG.Coords(JXG.COORDS_BY_SCREEN, [1, 1], this.board);
439             j++;
440         }
441         */
442         this.points[j] = new JXG.Coords(JXG.COORDS_BY_SCREEN, [x, y], this.board);
443         this.updateTransform(this.points[j]);
444         j++;
445         //xd_ = x-x0;
446         //yd_ = x-y0;
447         x0 = x;
448         y0 = y;
449         t0 = t;
450         
451         top--;
452         x = pointStack[top][0];
453         y = pointStack[top][1];
454         depth = depthStack[top]+1;
455         i = dyadicStack[top]*2;
456         
457     } while (top != 0);
458     this.numberPoints = this.points.length;
459     //$('debug').innerHTML = ' '+this.numberPoints;
460     return this;
461         
462 };

464 JXG.Curve.prototype.isSegmentOutside = function (x0,y0,x1,y1) {
465     if (y0<0 && y1<0) { return true; }
466     else if (y0>this.board.canvasHeight && y1>this.board.canvasHeight) { return true; }
467     else if (x0<0 && x1<0) { return true; }
468     else if (x0>this.board.canvasWidth && x1>this.board.canvasWidth) { return true; }
469     return false;
470 };

473     if (isNaN(x0+y0+x1+y1)) { return false; }
474     return (Math.abs(x1-x0)<MAXY && Math.abs(y1-y0)<MAXY);

535 JXG.Curve.prototype.updateTransform = function (p) {
536     var t, c, i, 
537         len = this.transformations.length;
538     if (len==0) {
539         return p;
540     }
541     for (i=0; i<len; i++) {
542         t = this.transformations[i];
543         t.update();
544         c = JXG.Math.matVecMult(t.matrix,p.usrCoords);
545         p.setCoordinates(JXG.COORDS_BY_USER,[c[1],c[2]]);
546     }
547     return p;
548 };

550 JXG.Curve.prototype.addTransform = function (transform) {
551     var list, i, len;
552     if (JXG.isArray(transform)) {
553         list = transform;
554     } else {
555         list = [transform];
556     }
557     len = list.length;
558     for (i=0; i<len; i++) {
559         this.transformations.push(list[i]);
560     }
561     return this;
562 };

579 /**
580  * Converts the GEONExT syntax of the defining function term into JavaScript.
581  * New methods X() and Y() for the Curve object are generated, further
582  * new methods for minX() and maxX().
583  *
584  * Also, all objects whose name appears in the term are searched and
585  * the curve is added as child to these objects. (Commented out!!!!)
586  * @see Algebra
587  * @see #geonext2JS.
588  */
589 JXG.Curve.prototype.generateTerm = function (varname, xterm, yterm, mi, ma) {
590     var fx, fy;
591 
592     // Generate the methods X() and Y()
593     if (JXG.isArray(xterm)) {
594         this.dataX = xterm;
595         this.X = function(i) { return this.dataX[i]; };
596         this.curveType = 'plot';
597         this.numberPoints = this.dataX.length;
598     } else {
599         this.X = JXG.createFunction(xterm,this.board,varname);
600         if (JXG.isString(xterm)) { 
601             this.curveType = 'functiongraph'; 
602         } else if (JXG.isFunction(xterm) || JXG.isNumber(xterm)) {
603             this.curveType = 'parameter';
604         }
605     }
606 
607     if (JXG.isArray(yterm)) {
608         this.dataY = yterm;
609         this.Y = function(i) {
