 26 /**
 27  * Creates a new instance of Polygon.
 28  * @class Polygon stores all style and functional properties that are required
 29  * to draw a polygon on a board.
 30  * @param {JXG.Board} board Reference to the board the polygon is drawn on.
 31  * @param {Array} vertices Unique identifiers for the points defining the polygon.
 32  * Last point must be first point.
 33  * @param {Array} borders Unique identifiers for the derived borderlines of the polygon
 34  * @param {String} id Unique identifier for this object.  If null or an empty string is given,
 35  * an unique id will be generated by Board
 36  * @param {String} name Not necessarily unique name, displayed on the board.  If null or an
 37  * empty string is given, an unique name will be generated.
 38  * @see JXG.Board#addPolygon
 39  * @constructor
 40  * @extends JXG.GeometryElement
 41  */
 42 
 43 JXG.Polygon = function (board, vertices, borders, id, name, withLines, withLabel, lineLabels, layer) {
 44     /* Call the constructor of GeometryElement */
 45     this.constructor();
 46     /**
 47      * Sets type of GeometryElement, value is OBJECT_TYPE_POLYGON.
 48      * @final
 49      * @type int
 50      */ 
 51     this.type = JXG.OBJECT_TYPE_POLYGON;
 52     this.elementClass = JXG.OBJECT_CLASS_AREA;                
 53     
 54     this.init(board, id, name);
 55     /**
 56      * Set the display layer.
 57      */
 58     if (layer == null) layer = board.options.layer['polygon'];
 59     this.layer = layer;
 60     
 61     if( (typeof withLines == 'undefined') || (withLines == null) ) {
 62         withLines = true;
 63     }
 64     if( (typeof lineLabels == 'undefined') || (lineLabels == null) ) {
 65         lineLabels = false;
 66     }    
 67         
 68     /**
 69      * Is the polygon bordered by lines?
 70      * @type bool
 71      */
 72     this.withLines = withLines;
 73 
 74     /**
 75      * References to the points defining the polygon.
 76      * Last vertex is the same as first vertex.
 77      * 
 78      * @type Array
 79      */    
 80     this.vertices = [];    
 81     for(var i=0; i<vertices.length; i++) {
 82        var vertex = JXG.getReference(this.board, vertices[i]);
 83        this.vertices[i] = vertex;
 84     }
 85     
 86     if((typeof borders == 'undefined') || (borders == null)) {
 87         borders = [];
 88         for(var i=0; i<vertices.length-1; i++) {
 89             borders[i] = {};
 90         }
 91     }
 92     
 93     if(this.vertices[this.vertices.length-1] != this.vertices[0]) {
 94         this.vertices.push(this.vertices[0]);
 95         borders.push({});
 96     }
 97     
 98     this.visProp['fillColor'] = this.board.options.polygon.fillColor;
 99     this.visProp['highlightFillColor'] = this.board.options.polygon.highlightFillColor;
100     this.visProp['fillOpacity'] = this.board.options.polygon.fillOpacity;
101     this.visProp['highlightFillOpacity'] = this.board.options.polygon.highlightFillOpacity;
102     
103     var l;
104  
105     /**
106      * References to the borderlines of the polygon.
107      * 
108      * @type Array
109      */  
110     this.borders = [];
111     if(withLines) {
112         for(var i=0; i<this.vertices.length-1; i++) {
113             /* create the borderlines */
114             l = new JXG.Line(board, this.vertices[i], this.vertices[i+1], borders[i].id, borders[i].name, lineLabels, this.layer+1); // keine Labels?
115                                                                                                                         // Layer +1, da sonst die Linien  teils
116                                                                                                                         // hinter dem Polygon verschwinden
117             l.setStraight(false,false); // Strecke
118             this.borders[i] = l;
119             l.parentPolygon = this;
120         }
121     }
122     
123     /* Add polygon as child to defining points */
124     for(var i=0; i<this.vertices.length-1; i++) { // last vertex is first vertex
125         var vertex = JXG.getReference(this.board, this.vertices[i]);
126         vertex.addChild(this);
127     }
128     
129     // create label 
130     this.createLabel(withLabel,[0,0]);
131     
132     /* Register polygon at board */
133     this.id = this.board.addPolygon(this);
134 };
135 JXG.Polygon.prototype = new JXG.GeometryElement;

147 /**
148  * Uses the boards renderer to update the polygon.
149  */
150 JXG.Polygon.prototype.updateRenderer = function () {
151     if (this.needsUpdate) {
152         this.board.renderer.updatePolygon(this);
153         this.needsUpdate = false;
154     }
155     if(this.hasLabel && this.label.content.visProp['visible']) {
156         //this.label.setCoordinates(this.coords);
157         this.label.content.update();
158         //this.board.renderer.updateLabel(this.label);
159         this.board.renderer.updateText(this.label.content);
160     }    
161 };

163 /**
164  * return TextAnchor
165  */
166 JXG.Polygon.prototype.getTextAnchor = function() {
167     var a = 0;
168     var b = 0;
169     var x = 0;
170     var y = 0;
171     a = x = this.vertices[0].X();
172     b = y = this.vertices[0].Y();
173     for (var i = 0; i < this.vertices.length; i++) {
174         if (this.vertices[i].X() < a)
175             a = this.vertices[i].X();
176         if (this.vertices[i].X() > x)
177             x = this.vertices[i].X();
178         if (this.vertices[i].Y() > b)
179             b = this.vertices[i].Y();
180         if (this.vertices[i].Y() < y)
181             y = this.vertices[i].Y();
182     }
183     return new JXG.Coords(JXG.COORDS_BY_USER, [(a + x)*0.5, (b + y)*0.5], this.board);
184 };

206 /**
207  * Copy the element to the background.
208  */
209 JXG.Polygon.prototype.cloneToBackground = function(addToTrace) {
210     var copy = {};
211     copy.id = this.id + 'T' + this.numTraces;
212     this.numTraces++;
213     copy.vertices = this.vertices;
214     copy.visProp = this.visProp;
215     JXG.clearVisPropOld(copy);
216     
217     this.board.renderer.drawPolygon(copy);
218 
219     this.traces[copy.id] = $(copy.id);
220 
221     delete copy;
222 };

224 JXG.createPolygon = function(board, parents, atts) {
225     var el, i;
226 
227     atts = JXG.checkAttributes(atts,{withLabel:JXG.readOption(board.options,'polygon','withLabel'), layer:null});
228     // Sind alles Punkte?
229     for(i=0; i<parents.length; i++) {
230         parents[i] = JXG.getReference(board, parents[i]);
231         if(!JXG.isPoint(parents[i]))
232             throw new Error("JSXGraph: Can't create polygon with parent types other than 'point'.");
233     }
234     
235     el = new JXG.Polygon(board, parents, atts["borders"], atts["id"],atts["name"],atts["withLines"],
236                         atts['withLabel'],atts['lineLabels'],atts['layer']);
237     
238     if(atts["withLines"] || true) {
239     	for(i=0; i<el.borders.length; i++) {
240     		el.borders[i].setProperty(atts);
241     	}
242     }
243 
244     return el;
245 };
246 
247 JXG.JSXGraph.registerElement('polygon', JXG.createPolygon);

249 JXG.Polygon.prototype.hideElement = function() {
250     this.visProp['visible'] = false;
251     this.board.renderer.hide(this);
252 
253     if(this.withLines) {
254         for(var i=0; i<this.borders.length; i++) {
255             this.borders[i].hideElement();
256         }
257     }
258     
259     if (this.hasLabel && this.label!=null) {
260         this.label.hiddenByParent = true;
261         if(this.label.content.visProp['visible']) {
262             this.board.renderer.hide(this.label.content);
263         }
264     }    
265 };
266 
267 JXG.Polygon.prototype.showElement = function() {
268     this.visProp['visible'] = true;
269     this.board.renderer.show(this);
270 
271     if(this.withLines) {
272         for(var i=0; i<this.borders.length; i++) {
273             this.borders[i].showElement();
274         }
275     }
276 };

278 JXG.Polygon.prototype.Area = function() {
279     //Surveyor's Formula
280     var area=0, i;
281     for(i=0; i<this.vertices.length-1; i++) {
282         area += (this.vertices[i].X()*this.vertices[i+1].Y()-this.vertices[i+1].X()*this.vertices[i].Y()); // last vertex is first vertex
283     }
284     area /= 2.0;
285     return Math.abs(area);
286 };

288 /**
289  * @class Constructs a regular polygon. It needs two points which define the base line and the number of vertices.
290  * @pseudo
291  * @description Constructs a regular polygon. It needs two points which define the base line and the number of vertices, or a set of points.
292  * @constructor
293  * @name RegularPolygon
294  * @type JXG.Polygon
295  * @augments JXG.Polygon
296  * @throws {Exception} If the element cannot be constructed with the given parent objects an exception is thrown.
297  * @param {JXG.Point_JXG.Point_Number} p1,p2,n The constructed regular polygon has n vertices and the base line defined by p1 and p2.
298  * @example
299  * var p1 = board.create('point', [0.0, 2.0]);
300  * var p2 = board.create('point', [2.0, 1.0]);
301  *
302  * var pol = board.create('regularpolygon', [p1, p2, 5]);
303  * </pre><div id="682069e9-9e2c-4f63-9b73-e26f8a2b2bb1" style="width: 400px; height: 400px;"></div>
304  * <script type="text/javascript">
305  *   var ccmex1_board = JXG.JSXGraph.initBoard('682069e9-9e2c-4f63-9b73-e26f8a2b2bb1', {boundingbox: [-1, 9, 9, -1], axis: false, showcopyright: false, shownavigation: false});
306  *   var regpol_p1 = ccmex1_board.create('point', [0.0, 2.0]);
307  *   var regpol_p2 = ccmex1_board.create('point', [2.0, 1.0]);
308  *   var regpol_cc1 = ccmex1_board.create('regularpolygon', [ccmex1_p1, ccmex1_p2, 5]);
309  * </script><pre>
310  * @example
311  * var p1 = board.create('point', [0.0, 2.0]);
312  * var p2 = board.create('point', [0.0,-2.0]);
313  * var p3 = board.create('point', [-2.0,0.0]);
314  *
315  * var pol = board.create('regularpolygon', [p1, p2, p3]);
316  * </pre><div id="096a78b3-bd50-4bac-b958-3be5e7df17ed" style="width: 400px; height: 400px;"></div>
317  * <script type="text/javascript">
318  *   var ccmex2_board = JXG.JSXGraph.initBoard('096a78b3-bd50-4bac-b958-3be5e7df17ed', {boundingbox: [-1, 9, 9, -1], axis: false, showcopyright: false, shownavigation: false});
319  *   var ccmex2_p1 = ccmex2_board.create('point', [0.0, 2.0]);
320  *   var ccmex2_p2 = ccmex2_board.create('point', [0.0, -2.0]);
321  *   var ccmex2_p3 = ccmex2_board.create('point', [-2.0,0.0]);
322  *   var ccmex2_cc1 = ccmex2_board.create('regularpolygon', [ccmex2_p1, ccmex2_p2, ccmex2_p3]);
323  * </script><pre>
324  */
325 JXG.createRegularPolygon = function(board, parents, atts) {
326     var el, i, n, p = [], rot, c, len, pointsExist;
327 
328     atts = JXG.checkAttributes(atts,{withLabel:JXG.readOption(board.options,'polygon','withLabel'), layer:null});
329     if (JXG.isNumber(parents[parents.length-1]) && parents.length!=3) {
330         throw new Error("JSXGraph: A regular polygon needs two point and a number as input.");
331     }
332 
333     len = parents.length;
334     n = parents[len-1];
335     if ((!JXG.isNumber(n) && !JXG.isPoint(JXG.getReference(board, n))) || n<3) {
336         throw new Error("JSXGraph: The third parameter has to be number greater than 2 or a point.");
337     }
338     
339     if (JXG.isPoint(JXG.getReference(board, n))) {  // Regular polygon given by n points
340         n = len;
341         pointsExist = true;
342     } else {
343         len--;
344         pointsExist = false;
345     }
346     // Sind alles Punkte? 
347     for(i=0; i<len; i++) {
348         parents[i] = JXG.getReference(board, parents[i]);
349         if(!JXG.isPoint(parents[i]))
350             throw new Error("JSXGraph: Can't create regular polygon if the first two parameters aren't points.");
351     }
352 
353     p[0] = parents[0];
354     p[1] = parents[1];
355     for (i=2;i<n;i++) {
356         rot = board.create('transform', [Math.PI*(2.0-(n-2)/n),p[i-1]], {type:'rotate'});
357         if (pointsExist) {
358             p[i] = parents[i];
359             p[i].addTransform(parents[i-2],rot);
360         } else {
361             p[i] = board.create('point',[p[i-2],rot],{name:'', withLabel:false,fixed:true,face:'o',size:1});
362         }
363     }
364     el = board.create('polygon',p,atts);
365 
366     return el;
367 };
