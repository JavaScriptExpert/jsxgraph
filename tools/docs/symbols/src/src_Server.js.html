 33 /**
 34  * @namespace
 35  * JXG.Server namespace holding functions to load JXG server modules.
 36  */
 37 JXG.Server = function(){};
 38 
 39 /**
 40  * This is where all of a module's handlers are accessed from. If you're loading a module named JXGModule which
 41  * provides a handler called ImaHandler, then this handler can be called by invoking JXG.Server.modules.JXGModule.ImaHandler().
 42  * @namespace
 43  */
 44 JXG.Server.modules = function(){};
 45 
 46 /**
 47  * Stores all asynchronous calls to server which aren't finished yet.
 48  * @private
 49  */
 50 JXG.Server.runningCalls = {};
 51 
 52 /**
 53  * Handles errors, just a default implementation, can be overwritten by you, if you want to handle errors by yourself.
 54  * @param {object} data An object holding a field of type string named message handling the error described in the message string.
 55  */
 56 JXG.Server.handleError = function(data) {
 57 	alert('error occured, server says: ' + data.message);
 58 };
 59 
 60 /**
 61  * The main method of JXG.Server. Actually makes the calls to the server and parses the feedback.
 62  * @param {string} action Can be 'load' or 'exec'.
 63  * @param {function} callback Function pointer or anonymous function which takes as it's only argument an
 64  * object containing the data from the server. The fields of this object depend on the reply of the server
 65  * module. See the correspondings server module readme.
 66  * @param {object} data What is to be sent to the server.
 67  * @param {boolean} sync If the call should be synchronous or not.
 68  */
 69 JXG.Server.callServer = function(action, callback, data, sync) {
 70 	var fileurl, passdata, AJAX,
 71 	params, id, dataJSONStr,
 72 	k;
 73 
 74     if(typeof sync == 'undefined' || sync == null)
 75         sync = false;
 76 
 77 	params = '';
 78 	for(k in data) {
 79 		params += '&' + escape(k) + '=' + escape(data[k]);
 80 	}
 81 
 82 	dataJSONStr = JXG.toJSON(data);
 83 
 84 	// generate id
 85 	do {
 86 		id = action + Math.floor(Math.random()*4096);
 87 	} while(typeof this.runningCalls[id] != 'undefined');
 88 
 89 	// store information about the calls
 90 	this.runningCalls[id] = { 'action': action };
 91 	if(typeof data.module != 'undefined')
 92 		this.runningCalls[id].module = data.module;
 93 
 94 	fileurl = JXG.serverBase + 'JXGServer.py';
 95     passdata = 'action=' + escape(action) + '&id=' + id + '&dataJSON=' + escape(JXG.Util.Base64.encode(dataJSONStr));
 96 
 97 	this.cbp = function(d) {
 98 		var str, data,
 99 		tmp, inject, paramlist, id,
100 		i, j;
101 
102 		str = (new JXG.Util.Unzip(JXG.Util.Base64.decodeAsArray(d))).unzip();
103 		if(JXG.isArray(str) && str.length > 0)
104 			str = str[0][0];
105 
106         if(typeof str != 'string')
107             return;
108 
109 		data =  eval("(" + str + ")");
110 
111 		if(data.type == 'error') {
112 			this.handleError(data);
113 		} else if (data.type == 'response') {
114 			id = data.id;
115 
116 			// inject fields
117 			for(i=0; i<data.fields.length; i++) {
118 				tmp = data.fields[i];
119 				inject = tmp.namespace + ( typeof eval(tmp.namespace) == 'object' ? '.' : '.prototype.') + tmp.name + ' = ' + tmp.value;
120 				eval(inject);
121 			}
122 
123 			// inject handlers
124 			for(i=0; i<data.handler.length; i++) {
125 				tmp = data.handler[i];
126 				paramlist = [];
127 
128 				for(j=0; j<tmp.parameters.length; j++) {
129 					paramlist[j] = '"' + tmp.parameters[j] + '": ' + tmp.parameters[j];
130 				}
131 				// insert subnamespace named after module.
132 				inject = 'if(typeof JXG.Server.modules.' + this.runningCalls[id].module + ' == "undefined")' +
133 				'JXG.Server.modules.' + this.runningCalls[id].module + ' = {};';
134 
135 				// insert callback method which fetches and uses the server's data for calculation in JavaScript
136 				inject += 'JXG.Server.modules.' + this.runningCalls[id].module + '.' + tmp.name + '_cb = ' + tmp.callback + ';';
137 
138 				// insert handler as JXG.Server.modules.<module name>.<handler name>
139 				inject += 'JXG.Server.modules.' + this.runningCalls[id].module + '.' + tmp.name + ' = function (' + tmp.parameters.join(',') + ', __JXGSERVER_CB__) {' +
140 				'if(typeof __JXGSERVER_CB__ == "undefined") __JXGSERVER_CB__ = JXG.Server.modules.' + this.runningCalls[id].module + '.' + tmp.name + '_cb;' +
141 				'var __JXGSERVER_PAR__ = {' + paramlist.join(',') + ', "module": "' + this.runningCalls[id].module + '", "handler": "' + tmp.name + '" };' +
142 				'JXG.Server.callServer("exec", __JXGSERVER_CB__, __JXGSERVER_PAR__);' +
143 				'};';
144 				eval(inject);
145 			}
146 
147 			delete this.runningCalls[id];
148 
149 			// handle data
150 			callback(data.data);
151 		}
152 	};
153 
154 	// bind cbp callback method to JXG.Server to get access to JXG.Server fields from within cpb
155 	this.cb = JXG.bind(this.cbp, this);
156 
157     // we're using our own XMLHttpRequest object in here because of a/sync and POST
158     if (window.XMLHttpRequest) {
159         AJAX = new XMLHttpRequest();
160         AJAX.overrideMimeType('text/plain; charset=iso-8859-1');
161     } else {                                  
162         AJAX = new ActiveXObject("Microsoft.XMLHTTP");
163     }
164     if (AJAX) {
165         // POST is required if data sent to server is too long for a url.
166         // some browsers/http servers don't accept long urls.
167         AJAX.open("POST", fileurl, !sync);
168         AJAX.setRequestHeader("Content-type", "application/x-www-form-urlencoded");
169 
170         if(!sync) {
171             // Define function to fetch data received from server
172             // that function returning a function is required to make this.cb known to the function.
173             AJAX.onreadystatechange = (function(cb){ return function () {
174                 switch(AJAX.readyState) {
175                     // server is ready for take-off
176                     case 4:
177                         if(AJAX.status != 200)
178                             alert("Fehler:" + AJAX.status);
179                         else  // grab it and call the server callback to debase64, unzip, and parse the data
180                             cb(AJAX.responseText);
181                     break;
182                     default:
183                         return false;
184                     break;
185                 }
186             }})(this.cb);
187         }
188 
189         // send the data
190         AJAX.send(passdata);
191         if(sync)
192             this.cb(AJAX.responseText);
193     } else {
194         return false;
195     }
196 
197 //	JXG.FileReader.parseFileContent(fileurl, this.cb, 'raw', !sync);
198 };

200 /**
201  * Callback for the default action 'load'.
202  */
203 JXG.Server.loadModule_cb = function(data) {
204 	var i;
205 	for(i=0; i<data.length; i++)
206 		alert(data[i].name + ': ' + data[i].value);
207 };
208 
209 /**
210  * Loads a module from the server.
211  * @param {string} module A string containing the module. Has to match the filename of the Python module on the server exactly including
212  * lower and upper case letters without the file ending .py.
213  */
214 JXG.Server.loadModule = function(module) {
215 	return JXG.Server.callServer('load', JXG.Server.loadModule_cb, {'module': module}, true);
216 };
