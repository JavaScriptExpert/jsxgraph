 49 /**
 50  * A point is the basic geometric element. Based on points lines and circles can be constructed which can be intersected
 51  * which in turn are points again which can be used to construct new lines, circles, polygons, etc. This class holds methods for
 52  * all kind of points like free points, gliders, and intersection points.
 53  * @class Creates a new point object. Do not use this constructor to create a point. Use {@link JXG.Board#create} with
 54  * type {@link Point}, {@link Glider}, or {@link Intersection} instead.  
 55  * @augments JXG.GeometryElement
 56  * @param {string,JXG.Board} board The board the new point is drawn on.
 57  * @param {Array} coordinates An array with the affine user coordinates of the point.
 58  * @param {String} id Unique identifier for the point. If null or an empty string is given,
 59  *  an unique id will be generated by Board
 60  * @param {String} name Not necessarily unique name for the point. If null or an
 61  *  empty string is given, an unique name will be generated
 62  * @param {boolean} show False if the point is invisible, True otherwise
 63  * @see JXG.Board#generateName
 64  * @see JXG.Board#addPoint
 65  */

260 /**
261  * Checks whether (x,y) is near the point.
262  * @param {int} x Coordinate in x direction, screen coordinates.
263  * @param {int} y Coordinate in y direction, screen coordinates.
264  * @type boolean
265  * @return True if (x,y) is near the point, False otherwise.
266  * @private
267  */
268 JXG.Point.prototype.hasPoint = function (x,y) {
269     var coordsScr = this.coords.scrCoords, r;
270     r = this.visProp['size'];
271     if(r < this.board.options.precision.hasPoint) {
272         r = this.board.options.precision.hasPoint;
273     }
274     return ((Math.abs(coordsScr[1]-x) < r+2) && (Math.abs(coordsScr[2]-y)) < r+2);
275 };

283 /**
284  * Updates the position of the point.
285  */
286 JXG.Point.prototype.update = function (fromParent) {
287     if (!this.needsUpdate) { return; }
288 
289     if(typeof fromParent == 'undefined') {
290         fromParent = false;
291     }
292   
293     if(this.traced) {
294         this.cloneToBackground(true);
295     }
296  /*
297      * We need to calculate the new coordinates no matter of the points visibility because
298      * a child could be visible and depend on the coordinates of the point (e.g. perpendicular).
299      * 
300      * Check if point is a glider and calculate new coords in dependency of this.slideObject.
301      * This function is called with fromParent==true for example if
302      * the defining elements of the line or circle have been changed.
303      */
304     if(this.type == JXG.OBJECT_TYPE_GLIDER) {
305         if(this.slideObject.type == JXG.OBJECT_TYPE_CIRCLE) {
306 //fromParent = false;        
307             if (fromParent) {
308                 this.coords.setCoordinates(JXG.COORDS_BY_USER, [this.slideObject.midpoint.X()+Math.cos(this.position),this.slideObject.midpoint.Y()+Math.sin(this.position)]);
309                 this.coords  = this.board.algebra.projectPointToCircle(this, this.slideObject);
310             } else {
311                 this.coords  = this.board.algebra.projectPointToCircle(this, this.slideObject);
312                 this.position = this.board.algebra.rad([this.slideObject.midpoint.X()+1.0,this.slideObject.midpoint.Y()],this.slideObject.midpoint,this);
313             }
314         } else if(this.slideObject.type == JXG.OBJECT_TYPE_LINE) {
315             this.coords  = this.board.algebra.projectPointToLine(this, this.slideObject);
316             
317             var p1coords = this.slideObject.point1.coords;
318             var p2coords = this.slideObject.point2.coords;
319             if (fromParent) {
320                 if (Math.abs(p1coords.usrCoords[0])>=JXG.Math.eps && Math.abs(p2coords.usrCoords[0])>=JXG.Math.eps) {
321                     this.coords.setCoordinates(JXG.COORDS_BY_USER, 
322                                            [p1coords.usrCoords[1] + this.position*(p2coords.usrCoords[1] - p1coords.usrCoords[1]),
323                                             p1coords.usrCoords[2] + this.position*(p2coords.usrCoords[2] - p1coords.usrCoords[2])]);
324                 }
325             } else {
326                 var factor = 1;
327                 var distP1S = p1coords.distance(JXG.COORDS_BY_USER, this.coords);
328                 var distP1P2 = p1coords.distance(JXG.COORDS_BY_USER, p2coords);
329                 var distP2S = p2coords.distance(JXG.COORDS_BY_USER, this.coords);
330                 
331                 if( ((distP1S > distP1P2) || (distP2S > distP1P2)) && (distP1S < distP2S)) { // Glider not between P1 & P2 and beyond P1
332                     factor = -1;
333                 }
334                 this.position = factor*distP1S/distP1P2;
335 
336                 // Snap the glider point of the slider into its appropiate position
337                 // First, recalculate the new value of this.position
338                 // Second, call update(fromParent==true) to make the positioning snappier.
339                 if (this.snapWidth!=null && Math.abs(this._smax-this._smin)>=JXG.Math.eps) {
340                     if (this.position<0.0) this.position = 0.0;
341                     if (this.position>1.0) this.position = 1.0;
342                     
343                     var v = this.position*(this._smax-this._smin)+this._smin;
344                         v = Math.round(v/this.snapWidth)*this.snapWidth;
345                     this.position = (v-this._smin)/(this._smax-this._smin);
346                     this.update(true);
347                 }
348             }
349             var p1Scr = this.slideObject.point1.coords.scrCoords;
350             var p2Scr = this.slideObject.point2.coords.scrCoords;
351 
352             var i;
353             if(this.slideObject.getSlope() == 0) {
354                 i = 1;
355             } else {
356                 i = 2;
357             }
358 
359             var y = this.coords.scrCoords[i];
360             if(!this.slideObject.visProp['straightFirst']) {
361                 if(p1Scr[i] < p2Scr[i]) {
362                     if(y < p1Scr[i]) {
363                        this.coords = this.slideObject.point1.coords;
364                        this.position = 0;
365                     }
366                 }
367                 else if(p1Scr[i] > p2Scr[i]) {
368                     if(y > p1Scr[i]) {
369                        this.coords = this.slideObject.point1.coords;
370                        this.position = 0;
371                     }
372                 }
373             }
374             if(!this.slideObject.visProp['straightLast']) {
375                 if(p1Scr[i] < p2Scr[i]) {
376                     if(y > p2Scr[i]) {
377                        this.coords = this.slideObject.point2.coords;
378                        this.position = 1;
379                     }
380                 }
381                 else if(p1Scr[i] > p2Scr[i]) {
382                     if(y < p2Scr[i]) {
383                        this.coords = this.slideObject.point2.coords;
384                        this.position = 1;
385                     }
386                 }
387             }  
388 
389             if(this.onPolygon) {
390                 var p1 = this.slideObject.point1.coords;
391                 var p2 = this.slideObject.point2.coords;
392                 if(Math.abs(this.coords.scrCoords[1]-p1.scrCoords[1])<this.board.options.precision.hasPoint && Math.abs(this.coords.scrCoords[2]-p1.scrCoords[2])<this.board.options.precision.hasPoint) {
393                     var poly = this.slideObject.parentPolygon;
394                     for(var i=0; i<poly.borders.length; i++) {
395                         if(this.slideObject == poly.borders[i]) {
396                             this.slideObject = poly.borders[(i - 1 + poly.borders.length) % poly.borders.length];
397                             break;
398                         }
399                     }
400                 }
401                 else if(Math.abs(this.coords.scrCoords[1]-p2.scrCoords[1])<this.board.options.precision.hasPoint && Math.abs(this.coords.scrCoords[2]-p2.scrCoords[2])<this.board.options.precision.hasPoint) {
402                     var poly = this.slideObject.parentPolygon;
403                     for(var i=0; i<poly.borders.length; i++) {
404                         if(this.slideObject == poly.borders[i]) {
405                             this.slideObject = poly.borders[(i + 1 + poly.borders.length) % poly.borders.length];
406                             break;                        
407                         }
408                     }
409                 }
410             }
411         } else if(this.slideObject.type == JXG.OBJECT_TYPE_TURTLE) {
412             this.updateConstraint(); // In case, the point is a constrained glider.
413             this.coords  = this.board.algebra.projectPointToTurtle(this, this.slideObject);
414         } else if(this.slideObject.elementClass == JXG.OBJECT_CLASS_CURVE) {
415             this.updateConstraint(); // In case, the point is a constrained glider.
416             this.coords  = this.board.algebra.projectPointToCurve(this, this.slideObject);
417         }
418     }
419     
420     /* If point is a calculated point, call updateConstraint() to calculate new coords. */
421     if (this.type == JXG.OBJECT_TYPE_CAS) {
422         this.updateConstraint();
423     }
424 
425     this.updateTransform();
426     
427     //this.updateRenderer();
428     this.needsUpdate = false;
429     return this;
430 };

432 /**
433  * Calls the renderer to update the drawing.
434  * @private
435  */
436 JXG.Point.prototype.updateRenderer = function () {
437     /* Call the renderer only if point is visible. */
438     if(this.visProp['visible']) {
439         var wasReal = this.isReal;
440         this.isReal = (isNaN(this.coords.usrCoords[1]+this.coords.usrCoords[2]))?false:true;
441         this.isReal = (Math.abs(this.coords.usrCoords[0])>this.board.algebra.eps)?this.isReal:false;  //Homogeneous coords: ideal point
442         if (this.isReal) {
443             if (wasReal!=this.isReal) { 
444                 this.board.renderer.show(this); 
445                 if(this.hasLabel && this.label.content.visProp['visible']) this.board.renderer.show(this.label.content); 
446             }
447             this.board.renderer.updatePoint(this);
448         } else {
449             if (wasReal!=this.isReal) { 
450                 this.board.renderer.hide(this); 
451                 if(this.hasLabel && this.label.content.visProp['visible']) this.board.renderer.hide(this.label.content); 
452             }
453         }
454     } 
455 
456     /* Update the label if visible. */
457     if(this.hasLabel && this.label.content.visProp['visible'] && this.isReal) {
458         //this.label.setCoordinates(this.coords);
459         this.label.content.update();
460         //this.board.renderer.updateLabel(this.label);
461         this.board.renderer.updateText(this.label.content);
462     }
463     return this;
464 };

526 /**
527  * Getter method for the distance to a second point, this is required for CAS-elements.
528  * Here, function inlining seems to be worthwile  (for plotting).
529  * @param {JXG.Point} point2 The point to which the distance shall be calculated.
530  * @return Distance in user coordinate to the given point
531  * @type number
532  */
533 JXG.Point.prototype.Dist = function(point2) {
534     var sum,
535         c = point2.coords.usrCoords,
536         ucr = this.coords.usrCoords,
537         f;
538         
539     f = ucr[0]-c[0];
540     sum = f*f;
541     f = ucr[1]-c[1];
542     sum += f*f;
543     f = ucr[2]-c[2];
544     sum += f*f;
545     return Math.sqrt(sum);
546     //return this.coords.distance(JXG.COORDS_BY_USER, point2.coords);
547 };

865 /**
866  * Animates a glider. Is called by the browser after startAnimation is called.
867  * @param {number} direction The direction the glider is animated.
868  * @param {number} stepCount The number of steps.
869  * @see #startAnimation
870  * @see #stopAnimation
871  * @private
872  */
873 JXG.Point.prototype._anim = function(direction, stepCount) {
874     var distance, slope, dX, dY, alpha, startPoint,
875         factor = 1, newX, radius;
876     
877     this.intervalCount++;
878     if(this.intervalCount > stepCount)
879         this.intervalCount = 0;
880     
881     if(this.slideObject.elementClass == JXG.OBJECT_CLASS_LINE) {
882         distance = this.slideObject.point1.coords.distance(JXG.COORDS_BY_SCREEN, this.slideObject.point2.coords);
883         slope = this.slideObject.getSlope();
884         if(slope != 'INF') {
885             alpha = Math.atan(slope);
886             dX = Math.round((this.intervalCount/stepCount) * distance*Math.cos(alpha));
887             dY = Math.round((this.intervalCount/stepCount) * distance*Math.sin(alpha));
888         } else {
889             dX = 0;
890             dY = Math.round((this.intervalCount/stepCount) * distance);
891         }
892         
893         if(direction < 0) {
894             startPoint = this.slideObject.point2;
895             if(this.slideObject.point2.coords.scrCoords[1] - this.slideObject.point1.coords.scrCoords[1] > 0)
896                 factor = -1;
897             else if(this.slideObject.point2.coords.scrCoords[1] - this.slideObject.point1.coords.scrCoords[1] == 0) {
898                 if(this.slideObject.point2.coords.scrCoords[2] - this.slideObject.point1.coords.scrCoords[2] > 0)
899                     factor = -1;
900             }
901         } else {
902             startPoint = this.slideObject.point1;
903             if(this.slideObject.point1.coords.scrCoords[1] - this.slideObject.point2.coords.scrCoords[1] > 0)
904                 factor = -1;
905             else if(this.slideObject.point1.coords.scrCoords[1] - this.slideObject.point2.coords.scrCoords[1] == 0) {
906                 if(this.slideObject.point1.coords.scrCoords[2] - this.slideObject.point2.coords.scrCoords[2] > 0)
907                     factor = -1;
908             }
909         }
910         
911         this.coords.setCoordinates(JXG.COORDS_BY_SCREEN, [startPoint.coords.scrCoords[1] + factor*dX, startPoint.coords.scrCoords[2] + factor*dY]);
912     } else if(this.slideObject.elementClass == JXG.OBJECT_CLASS_CURVE) {
913         if(direction > 0) {
914             newX = Math.round(this.intervalCount/stepCount * this.board.canvasWidth);
915         } else {
916             newX = Math.round((stepCount - this.intervalCount)/stepCount * this.board.canvasWidth);
917         }
918   
919         this.coords.setCoordinates(JXG.COORDS_BY_SCREEN, [newX, 0]);
920         this.coords = this.board.algebra.projectPointToCurve(this, this.slideObject);
921     } else if(this.slideObject.elementClass == JXG.OBJECT_CLASS_CIRCLE) {
922         if(direction < 0) {
923             alpha = this.intervalCount/stepCount * 2*Math.PI;
924         } else {
925             alpha = (stepCount - this.intervalCount)/stepCount * 2*Math.PI;
926         }
927 
928         radius = this.slideObject.Radius();
929 
930         this.coords.setCoordinates(JXG.COORDS_BY_USER, [this.slideObject.midpoint.coords.usrCoords[1] + radius*Math.cos(alpha), this.slideObject.midpoint.coords.usrCoords[2] + radius*Math.sin(alpha)]);
931     }
932     
933     this.board.update(this);
934     return this;
935 };

1154 JXG.createPoint = function(/** JXG.Board */ board, /** array */ parents, /** object */ atts) {
1155     var el, isConstrained = false, i, show;
1156     atts = JXG.checkAttributes(atts,{withLabel:JXG.readOption(board.options,'point','withLabel'), layer:null});
1157     show = (typeof atts['visible']=='undefined') || JXG.str2Bool(atts['visible']);
1158     
1159     for (i=0;i<parents.length;i++) {
1160         if (typeof parents[i]=='function' || typeof parents[i]=='string') {
1161             isConstrained = true;
1162         }
1163     }
1164     if (!isConstrained) {
1165         if ( (JXG.isNumber(parents[0])) && (JXG.isNumber(parents[1])) ) {
1166             el = new JXG.Point(board, parents, atts['id'], atts['name'], show, atts['withLabel'], atts['layer']);
1167             if ( atts["slideObject"] != null ) {
1168                 el.makeGlider(atts["slideObject"]);
1169             } else {
1170                 el.baseElement = el; // Free point
1171             }
1172         } else if ( (typeof parents[0]=='object') && (typeof parents[1]=='object') ) { // Transformation
1173             el = new JXG.Point(board, [0,0], atts['id'], atts['name'], show, atts['withLabel'], atts['layer']);   
1174             el.addTransform(parents[0],parents[1]);
1175         }
1176         else {// Failure
1177             throw new Error("JSXGraph: Can't create point with parent types '" + (typeof parents[0]) + "' and '" + (typeof parents[1]) + "'.");
1178         }
1179     } else {
1180         el = new JXG.Point(board, [0,0], atts['id'], atts['name'], show, atts['withLabel'], atts['layer']);
1181         el.addConstraint(parents);
1182     }
1183     return el;
1184 };
