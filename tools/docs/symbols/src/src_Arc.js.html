 31 /**
 32  * @class This element is used to provide a constructor for arc elements.
 33  * @pseudo
 34  * @description An is a segment of the circumference of a circle.
 35  * @name Arc
 36  * @augments Curve
 37  * @constructor
 38  * @type JXG.Curve
 39  * @throws {Exception} If the element cannot be constructed with the given parent objects an exception is thrown.
 40  * @param {JXG.Point_JXG.Point_JXG.Point} p1,p2,p3 The result will be an arc of a circle around p1 through p2. The arc is drawn
 41  * counter-clockwise from p2 to p3.
 42  * @example
 43  * // Create an arc out of three free points
 44  * var p1 = board.create('point', [2.0, 2.0]);
 45  * var p2 = board.create('point', [1.0, 0.5]);
 46  * var p3 = board.create('point', [3.5, 1.0]);
 47  *
 48  * var a = board.create('arc', [p1, p2, p3]);
 49  * </pre><div id="114ef584-4a5e-4686-8392-c97501befb5b" style="width: 300px; height: 300px;"></div>
 50  * <script type="text/javascript">
 51  *   var arex1_board = JXG.JSXGraph.initBoard('114ef584-4a5e-4686-8392-c97501befb5b', {boundingbox: [-1, 7, 7, -1], axis: true, showcopyright: false, shownavigation: false});
 52  * var arex1_p1 = arex1_board.create('point', [2.0, 2.0]);
 53  * var arex1_p2 = arex1_board.create('point', [1.0, 0.5]);
 54  * var arex1_p3 = arex1_board.create('point', [3.5, 1.0]);
 55  *
 56  * var arex1_a = arex1_board.create('arc', [arex1_p1, arex1_p2, arex1_p3]);
 57  * </script><pre>
 58  */
 59 JXG.createArc = function(board, parents, attributes) {
 60     var el, defaults, key, options;
 61         
 62     // Alles 3 Punkte?
 63     if ( !(JXG.isPoint(parents[0]) && JXG.isPoint(parents[1]) && JXG.isPoint(parents[2]))) {
 64         throw new Error("JSXGraph: Can't create Arc with parent types '" + 
 65                         (typeof parents[0]) + "' and '" + (typeof parents[1]) + "' and '" + 
 66                         (typeof parents[2]) + "'." +
 67                         "\nPossible parent types: [point,point,point]");
 68     }
 69 
 70     // Read the default values from Options and use them in case they are not set by the user
 71     // in attributes
 72     defaults = {withLabel:JXG.readOption(board.options,'elements','withLabel'), 
 73                 layer:JXG.readOption(board.options,'layer','arc'),
 74                 useDirection:false}; // useDirection is necessary for circumCircleArcs
 75     defaults['strokeWidth'] =  board.options.elements['strokeWidth'];
 76     options = board.options.arc;
 77     for (key in options) {
 78         defaults[key] = options[key];
 79     }
 80     attributes = JXG.checkAttributes(attributes, defaults);
 81         
 82     el = board.create('curve',[[0],[0]],attributes);
 83     el.type = JXG.OBJECT_TYPE_ARC;
 84     /**
 85      * Midpoint of the arc.
 86      * @type JXG.Point
 87      */
 88     el.midpoint = JXG.getReference(board, parents[0]);
 89     /**
 90      * Point defining the arcs circle.
 91      * @type JXG.Point
 92      */
 93     el.point2 = JXG.getReference(board, parents[1]);
 94     /**
 95      * The point defining the angle of the arc.
 96      * @type JXG.Point
 97      */
 98     el.point3 = JXG.getReference(board, parents[2]);
 99     /* Add arc as child to defining points */
100     el.midpoint.addChild(el);
101     el.point2.addChild(el);
102     el.point3.addChild(el);
103     
104     el.useDirection = attributes['useDirection'];      // useDirection is necessary for circumCircleArcs
105 
106     el.updateDataArray = function() {
107         var A = this.point2,
108             B = this.midpoint,
109             C = this.point3,
110             beta, co, si, matrix,
111             phi = JXG.Math.Geometry.rad(A,B,C),
112             i,
113             //n = 100, 
114             n = Math.ceil(phi/Math.PI*90)+1,
115             delta = phi/n, //Math.PI/90.0,
116             x = B.X(),
117             y = B.Y(),
118             v,
119             det, p0c, p1c, p2c;
120 
121         if (this.useDirection) {  // This is true for circumCircleArcs. In that case there is
122                                   // a fourth parent element: [midpoint, point1, point3, point2]
123             p0c = parents[1].coords.usrCoords;
124             p1c = parents[3].coords.usrCoords;
125             p2c = parents[2].coords.usrCoords;
126             det = (p0c[1]-p2c[1])*(p0c[2]-p1c[2]) - (p0c[2]-p2c[2])*(p0c[1]-p1c[1]);
127             if(det < 0) {
128                 this.point2 = parents[1];
129                 this.point3 = parents[2];
130             } else {
131                 this.point2 = parents[2];
132                 this.point3 = parents[1];
133             }
134         }
135         this.dataX = [A.X()];
136         this.dataY = [A.Y()];
137 
138         for (beta=delta,i=1; i<=n; i++, beta+=delta) {
139             co = Math.cos(beta);
140             si = Math.sin(beta);
141             matrix = [[1,            0,   0],
142                       [x*(1-co)+y*si,co,-si],
143                       [y*(1-co)-x*si,si, co]];
144             v = JXG.Math.matVecMult(matrix,A.coords.usrCoords);
145             this.dataX.push(v[1]/v[0]);
146             this.dataY.push(v[2]/v[0]);
147         }
148     };
149 
150     /**
151      * Calculates the arcs radius.
152      * @returns {Number} The arcs radius
153      */
154     el.Radius = function() {
155         return this.point2.Dist(this.midpoint);
156     };
157 
158     /**
159      * @deprecated
160      */
161     el.getRadius = function() {
162         return this.Radius();
163     };
164 
165     /**
166      *Checks whether (x,y) is near the arc.
167      * @param {int} x Coordinate in x direction, screen coordinates.
168      * @param {int} y Coordinate in y direction, screen coordinates.
169      * @returns {Boolean} True if (x,y) is near the arc, False otherwise.
170      */
171     el.hasPoint = function (x, y) { 
172         var prec = this.board.options.precision.hasPoint/(this.board.stretchX),
173             checkPoint = new JXG.Coords(JXG.COORDS_BY_SCREEN, [x,y], this.board),
174             r = this.Radius(),
175             dist = this.midpoint.coords.distance(JXG.COORDS_BY_USER,checkPoint),
176             has = (Math.abs(dist-r) < prec),
177             angle;
178             
179         if(has) {
180             angle = JXG.Math.Geometry.rad(this.point2,this.midpoint,checkPoint.usrCoords.slice(1));
181             if (angle>JXG.Math.Geometry.rad(this.point2,this.midpoint,this.point3)) { has = false; }
182         }
183         return has;    
184     };
185 
186     /**
187      * Checks whether (x,y) is within the sector defined by the arc.
188      * @param {int} x Coordinate in x direction, screen coordinates.
189      * @param {int} y Coordinate in y direction, screen coordinates.
190      * @returns {Boolean} True if (x,y) is within the sector defined by the arc, False otherwise.
191      */
192     el.hasPointSector = function (x, y) { 
193         var checkPoint = new JXG.Coords(JXG.COORDS_BY_SCREEN, [x,y], this.board),
194             r = this.Radius(),
195             dist = this.midpoint.coords.distance(JXG.COORDS_BY_USER,checkPoint),
196             has = (dist<r),
197             angle;
198         
199         if(has) {
200             angle = JXG.Math.Geometry.rad(this.point2,this.midpoint,checkPoint.usrCoords.slice(1));
201             if (angle>JXG.Math.Geometry.rad(this.point2,this.midpoint,this.point3)) { has = false; }
202         }
203         return has;    
204     };
205 
206     /**
207      * @returns {JXG.Coords} Coordinates of the text's anchor.
208      */
209     el.getTextAnchor = function() {
210         return this.midpoint.coords;
211     };
212 
213     /**
214      * @returns {JXG.Coords} Coordinates of the label's anchor
215      */
216     el.getLabelAnchor = function() {
217         var angle = JXG.Math.Geometry.rad(this.point2, this.midpoint, this.point3),
218             dx = 10/(this.board.stretchX),
219             dy = 10/(this.board.stretchY),
220             p2c = this.point2.coords.usrCoords,
221             pmc = this.midpoint.coords.usrCoords,
222             bxminusax = p2c[1] - pmc[1],
223             byminusay = p2c[2] - pmc[2],
224             coords, vecx, vecy, len;
225 
226         if(this.label.content != null) {                          
227             this.label.content.relativeCoords = new JXG.Coords(JXG.COORDS_BY_SCREEN, [0,0],this.board);                      
228         }  
229 
230         coords = new JXG.Coords(JXG.COORDS_BY_USER, 
231                         [pmc[1]+ Math.cos(angle*0.5)*bxminusax - Math.sin(angle*0.5)*byminusay, 
232                         pmc[2]+ Math.sin(angle*0.5)*bxminusax + Math.cos(angle*0.5)*byminusay], 
233                         this.board);
234 
235         vecx = coords.usrCoords[1] - pmc[1];
236         vecy = coords.usrCoords[2] - pmc[2];
237     
238         len = Math.sqrt(vecx*vecx+vecy*vecy);
239         vecx = vecx*(len+dx)/len;
240         vecy = vecy*(len+dy)/len;
241 
242         return new JXG.Coords(JXG.COORDS_BY_USER, [pmc[1]+vecx,pmc[2]+vecy],this.board);
243     };
244 
245     el.prepareUpdate().update();
246     return el;
247 };

249 JXG.JSXGraph.registerElement('arc', JXG.createArc);

251 /** TODO: Documentation of those two elements below are to be written like the one above for ARC.
252 /**
253  * Creates a new semicircle. The semicircle is drawn clock-wise between the first and the second defining point.
254  * @param {JXG.Board} board The board the semicircle is put on.
255  * @param {Array} parents Array of two opposite points defining the semicircle.
256  * @param {Object} attributes Object containing properties for the element such as stroke-color and visibility. See @see JXG.GeometryElement#setProperty
257  * @type JXG.Arc
258  * @return Reference to the created arc object.
259  */
260 JXG.createSemicircle = function(board, parents, attributes) {
261     var el, mp, idmp = '';
262     
263     attributes = JXG.checkAttributes(attributes,{});
264     if(attributes['id'] != null) {
265         idmp = attributes['id']+'_mp';
266     }
267     // Alles 2 Punkte?
268     if ( (JXG.isPoint(parents[0])) && (JXG.isPoint(parents[1])) ) {
269         mp = board.create('midpoint', [parents[0], parents[1]], {id:idmp, withLabel:false, visible:false});
270         el = board.create('arc',[mp, parents[1], parents[0]],attributes);
271     } // Ansonsten eine fette Exception um die Ohren hauen
272     else
273         throw new Error("JSXGraph: Can't create Semicircle with parent types '" + 
274                         (typeof parents[0]) + "' and '" + (typeof parents[1]) + "'." +
275                         "\nPossible parent types: [point,point]");
276 
277     return el;
278 };
279 
280 JXG.JSXGraph.registerElement('semicircle', JXG.createSemicircle);

282 /**
283  * Creates a new circumcircle arc through three defining points.
284  * @param {JXG.Board} board The board the arc is put on.
285  * @param {Array} parents Array of three points defining the circumcircle arc.
286  * @param {Object} attributes Object containing properties for the element such as stroke-color and visibility. See @see JXG.GeometryElement#setProperty
287  * @type JXG.Arc
288  * @return Reference to the created arc object.
289  */
290 JXG.createCircumcircleArc = function(board, parents, attributes) {
291     var el, mp, idmp;
292     
293     attributes = JXG.checkAttributes(attributes,{withLabel:JXG.readOption(board.options,'arc','withLabel'), layer:null});
294     if(attributes['id'] != null) {
295         idmp = attributes['id']+'_mp';
296     }
297     
298     // Alles 3 Punkte?
299     if ( (JXG.isPoint(parents[0])) && (JXG.isPoint(parents[1])) && (JXG.isPoint(parents[2]))) {
300         mp = board.create('circumcirclemidpoint',[parents[0], parents[1], parents[2]], {id:idmp, withLabel:false, visible:false});
301         attributes.useDirection = true;
302         el = board.create('arc', [mp,parents[0],parents[2],parents[1]], attributes);
303     } // Ansonsten eine fette Exception um die Ohren hauen
304     else
305         throw new Error("JSXGraph: create Circumcircle Arc with parent types '" + 
306                         (typeof parents[0]) + "' and '" + (typeof parents[1]) + "' and '" + (typeof parents[2]) + "'." +
307                         "\nPossible parent types: [point,point,point]");
308 
309 
310     return el;
311 };

313 JXG.JSXGraph.registerElement('circumcirclearc', JXG.createCircumcircleArc);
