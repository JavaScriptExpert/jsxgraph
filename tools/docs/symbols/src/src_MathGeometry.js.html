 41 /**
 42  * Calculates the angle defined by the points A, B, C.
 43  * @param {JXG.Point,array} A A point  or [x,y] array.
 44  * @param {JXG.Point,array} B Another point or [x,y] array.
 45  * @param {JXG.Point,array} C A circle - no, of course the third point or [x,y] array.
 46  * @type number
 47  * @return The angle in radian measure.
 48  * @deprecated Use {@link JXG.Math.Geometry#rad} instead.
 49  * @see #rad
 50  * @see #trueAngle
 51  */
 52 JXG.Math.Geometry.angle = function(A, B, C) {   
 53     var a = [],
 54         b = [],
 55         c = [],
 56         u, v, s, t;
 57         
 58     if (A.coords == null) {
 59         a[0] = A[0];
 60         a[1] = A[1];
 61     } else {
 62         a[0] = A.coords.usrCoords[1];
 63         a[1] = A.coords.usrCoords[2];
 64     }
 65     if (B.coords == null) {
 66         b[0] = B[0];
 67         b[1] = B[1];
 68     } else {
 69         b[0] = B.coords.usrCoords[1];
 70         b[1] = B.coords.usrCoords[2];
 71     }
 72     if (C.coords == null) {
 73         c[0] = C[0];
 74         c[1] = C[1];
 75     } else {
 76         c[0] = C.coords.usrCoords[1];
 77         c[1] = C.coords.usrCoords[2];
 78     }
 79     u = a[0] - b[0];
 80     v = a[1] - b[1];
 81     s = c[0] - b[0];
 82     t = c[1] - b[1];
 83     return Math.atan2(u*t-v*s,u*s+v*t);    
 84 };

 86 /**
 87  * Calculates the angle defined by the three points A, B, C if you're going from A to C around B counterclockwise.
 88  * @param A Point or [x,y] array
 89  * @param B Point or [x,y] array
 90  * @param C Point or [x,y] array
 91  * @return The angle in degrees.
 92  * @see #rad
 93  */
 94 JXG.Math.Geometry.trueAngle = function(/** JXG.Point */ A, /** JXG.Point */ B, /** JXG.Point */ C) /** number */ {
 95     return this.rad(A,B,C)*57.295779513082323; // *180.0/Math.PI;
 96 };

 98 /**
 99  * Calculates the internal angle defined by the three points A, B, C if you're going from A to C around B counterclockwise.
100  * @param {JXG.Point} A Point or [x,y] array
101  * @param {JXG.Point} B Point or [x,y] array
102  * @param {JXG.Point} C Point or [x,y] array
103  * @type number
104  * @see #trueAngle
105  * @return Angle in radians.
106  */
107 JXG.Math.Geometry.rad = function(A,B,C) {
108     var ax, ay, bx, by, cx, cy,
109         abx, aby, cbx, cby,
110         cp, l1, l2, phiacos, phicos, sp, 
111         phi = 0;
112         
113     if (A.coords == null) {
114         ax = A[0];
115         ay = A[1];
116     } else {
117         ax = A.coords.usrCoords[1];
118         ay = A.coords.usrCoords[2];
119     }
120     if (B.coords == null) {
121         bx = B[0];
122         by = B[1];
123     } else {
124         bx = B.coords.usrCoords[1];
125         by = B.coords.usrCoords[2];
126     }
127     if (C.coords == null) {
128         cx = C[0];
129         cy = C[1];
130     } else {
131         cx = C.coords.usrCoords[1];
132         cy = C.coords.usrCoords[2];
133     }
134     cbx = cx - bx;
135     cby = cy - by;
136     abx = ax - bx;
137     aby = ay - by;
138     
139     //return Math.atan2(cby-aby,cbx-abx);
140 
141     sp = cbx*abx + cby*aby;               // scalar product of c-b and a-b
142     cp = abx*cby - aby*cbx;               // cross product of a-b c-b
143     l1 = Math.sqrt(abx*abx + aby*aby);    // length of a-b
144     l2 = Math.sqrt(cbx*cbx + cby*cby);    // length of c-b
145     phiacos = sp / (l1 * l2);             // calculate the angle as cosine from scalar product
146     if (phiacos > 1) { // these things should not happen, but can happen because of numerical inaccurracy
147         phiacos = 1;
148     } else if (phiacos < -1) {
149         phiacos = -1;
150     }
151     phicos = Math.acos(phiacos); // calculate the angle
152     
153         /*
154          * The calculated angle may not be the right angle because of the calculation of acos 
155         real     | quadrant  | quadrant | algebraic sign 
156         quadrant | by cosine | by sine  | of sine 
157            1.    |   1.      |   1.     |   +
158            2.    |   2.      |   1.     |   +
159            3.    |   2.      |   3.     |   -
160            4.    |   1.      |   3.     |   - 
161          So only for the first quadrant the calculated angle is ok. 
162          But we can use the sine, which is connected with the cross product to select the right angle. 
163          Calculate the sine of the calculated angle and multiply it with the cross product's value.
164          
165         real     | quadrant  | algebraic sign | algebraic sign of 
166         quadrant | by cosine | of sin(phicos) | cross product
167            1.    |   1.      |   +            |   +
168            2.    |   2.      |   +            |   +
169            3.    |   2.      |   +            |   -
170            4.    |   1.      |   +            |   - 
171          So always the negative angle of phicos has to be taken if the product is negative.
172          */
173     if ((Math.sin(phicos) * cp) < 0) {
174         phi = 6.2831853071795862 - phicos; // 2 * Math.PI - phicos;
175     } else {
176         phi = phicos;
177     }
178     return phi;
179 };

181 /**
182  * Calculates the bisection between the three points A, B, C. The bisection is defined by two points:
183  * Parameter B and a point with the coordinates calculated in this function.
184  * @param A Point
185  * @param B Point
186  * @param C Point
187  * @param [board=A.board] Reference to the board
188  * @return Coordinates of the second point defining the bisection.
189  */
190 JXG.Math.Geometry.angleBisector = function(/** JXG.Point */ A, /** JXG.Point */ B, /** JXG.Point */ C, /** JXG.Board */ board) /** JXG.Coords */ {
191     /* First point */
192     var Ac = A.coords.usrCoords,
193         Bc = B.coords.usrCoords, 
194         Cc = C.coords.usrCoords,
195         x = Ac[1]-Bc[1],
196         y = Ac[2]-Bc[2],
197         d = Math.sqrt(x*x+y*y),
198         phiA, phiC, phi;
199     
200     if(typeof board == 'undefined')
201         board = A.board;
202     
203     x /= d;
204     y /= d;
205     
206     phiA = Math.acos(x);
207     if (y<0) { phiA *= -1; }
208     if (phiA<0) { phiA += 2*Math.PI; } 
209     
210     /* Second point */
211     x = Cc[1]-Bc[1];
212     y = Cc[2]-Bc[2];
213     d = Math.sqrt(x*x+y*y);
214     x /= d;
215     y /= d;
216     
217     phiC = Math.acos(x);
218     if (y<0) { phiC *= -1; }
219     if (phiC<0) { phiC += 2*Math.PI; } 
220  
221     phi=(phiA+phiC)*0.5;
222     if (phiA>phiC) { 
223         phi+=Math.PI;
224     }
225 
226     x = Math.cos(phi)+Bc[1];
227     y = Math.sin(phi)+Bc[2];
228     
229     return new JXG.Coords(JXG.COORDS_BY_USER, [x,y], board);
230 };

232 /**
233  * Reflects the point along the line.
234  * @param line Axis of reflection.
235  * @param point Point to reflect.
236  * @param [board=point.board] Reference to the board
237  * @return Coordinates of the reflected point.
238  */  
239 JXG.Math.Geometry.reflection = function(/** JXG.Line */ line, /** JXG.Point */ point, /** JXG.Board */ board) /** JXG.Coords */ {
240     /* (v,w) defines the slope of the line */    
241     var pc = point.coords.usrCoords,
242         p1c = line.point1.coords.usrCoords,
243         p2c = line.point2.coords.usrCoords,
244         x0, y0, x1, y1, v, w, mu;
245 
246     if(typeof board == 'undefined')
247         board = point.board;
248 
249     v = p2c[1]-p1c[1];
250     w = p2c[2]-p1c[2];
251     
252     x0 = pc[1]-p1c[1];
253     y0 = pc[2]-p1c[2];
254     
255     mu = (v*y0-w*x0)/(v*v+w*w);
256     
257     /* point + mu*(-y,x) waere Lotpunkt */
258     x1 = pc[1] + 2*mu*w;
259     y1 = pc[2] - 2*mu*v;
260     
261     return new JXG.Coords(JXG.COORDS_BY_USER, [x1,y1], board);
262 };

264 /**
265  * Computes the new position of a point which is rotated 
266  * around a second point (called rotpoint) by the angle phi.
267  * @param {JXG.Point} rotpoint Center of the rotation
268  * @param {JXG.Point} point point to be rotated
269  * @param {number} phi rotation angle in arc length
270  * @param {JXG.Board} [board=point.board] Reference to the board
271  * @type JXG.Coords
272  * @return Coordinates of the new position.
273  */
274 JXG.Math.Geometry.rotation = function(rotpoint, point, phi, board) {
275     var pc = point.coords.usrCoords,
276         rotpc = rotpoint.coords.usrCoords,
277         x0, y0, c, s, x1, y1;
278 
279     if(typeof board == 'undefined')
280         board = point.board;
281 
282     x0 = pc[1]-rotpc[1];
283     y0 = pc[2]-rotpc[2];
284     
285     c = Math.cos(phi);
286     s = Math.sin(phi);
287     
288     x1 = x0*c-y0*s + rotpc[1];
289     y1 = x0*s+y0*c + rotpc[2];
290     
291     return new JXG.Coords(JXG.COORDS_BY_USER, [x1,y1], board);
292 };

294 /**
295  * Calculates the coordinates of a point on the perpendicular to the given line through
296  * the given point.
297  * @param {JXG.Line} line A line.
298  * @param {JXG.Point} point Intersection point of line to perpendicular.
299  * @param {JXG.Board} [board=point.board] Reference to the board
300  * @type JXG.Coords
301  * @return Coordinates of a point on the perpendicular to the given line through the given point.
302  */
303 JXG.Math.Geometry.perpendicular = function(line, point, board) {
304     var A = line.point1.coords.usrCoords,
305         B = line.point2.coords.usrCoords,
306         C = point.coords.usrCoords,
307         x, y, change,
308         fmd, emc, d0, d1, den;
309 
310     if(typeof board=='undefined')
311         board = point.board;
312 
313     if(point == line.point1) { // Punkt ist erster Punkt der Linie
314         x = A[1] + B[2] - A[2];
315         y = A[2] - B[1] + A[1];
316         change = true;
317     }
318     else if(point == line.point2) {  // Punkt ist zweiter Punkt der Linie    
319         x = B[1] + A[2] - B[2];
320         y = B[2] - A[1] + B[1];
321         change = false;
322     }
323     else if( ((Math.abs(A[1] - B[1]) > JXG.Math.eps) && 
324              (Math.abs(C[2] - (A[2] - B[2])*(C[1]-A[1])/(A[1] - B[1])-A[2]) < JXG.Math.eps)) ||
325              ((Math.abs(A[1] - B[1]) <= JXG.Math.eps) && (Math.abs(A[1] - C[1]) < JXG.Math.eps)) ) { // Punkt liegt auf der Linie
326         x = C[1] + B[2] - C[2];
327         y = C[2] - B[1] + C[1]; 
328         change = true;
329         if(Math.abs(x - C[1]) < JXG.Math.eps && Math.abs(y - C[2]) < JXG.Math.eps) {
330             x = C[1] + A[2] - C[2];
331             y = C[2] - A[1] + C[1];
332             change = false;
333         }
334     }
335     else { // Punkt liegt nicht auf der Linie -> als zweiter Punkt wird der Lotfusspunkt gewaehlt
336         fmd = A[2] - B[2];
337         emc = A[1] - B[1];
338         d0 = B[1]*fmd - B[2]*emc;
339         d1 = C[1]*emc + C[2]*fmd;
340         den = fmd*fmd + emc*emc;
341         if(Math.abs(den)<JXG.Math.eps) {
342             den = JXG.Math.eps;
343         }
344         x = (d0*fmd + d1*emc) / den;
345         y = (d1*fmd - d0*emc) /den;
346         change = true;
347     }                            
348     return [new JXG.Coords(JXG.COORDS_BY_USER, [x, y], board),change];             
349 };

351 /**
352  * Calculates the midpoint of the circumcircle of the three given points.
353  * @param {JXG.Point} point1 Point
354  * @param {JXG.Point} point2 Point
355  * @param {JXG.Point} point3 Point
356  * @param {JXG.Board} [board=point1.board] Reference to the board
357  * @type JXG.Coords
358  * @return Coordinates of the midpoint of the circumcircle of the given points.
359  */
360 JXG.Math.Geometry.circumcenterMidpoint = function(point1, point2, point3, board) {
361     var A = point1.coords.usrCoords,
362         B = point2.coords.usrCoords,
363         C = point3.coords.usrCoords,
364         u, v, den, x, y;
365 
366     if(typeof board=='undefined')
367         board = point1.board;
368 
369     u = ((A[1]-B[1])*(A[1]+B[1]) + (A[2]-B[2])*(A[2]+B[2])) * 0.5;
370     v = ((B[1]-C[1])*(B[1]+C[1]) + (B[2]-C[2])*(B[2]+C[2])) * 0.5;
371     den = (A[1]-B[1])*(B[2]-C[2]) - (B[1]-C[1])*(A[2]-B[2]);
372               
373     if (Math.abs(den) < JXG.Math.eps) {
374         den = JXG.Math.eps;
375     }
376     
377     x = (u * (B[2]-C[2]) - v*(A[2]-B[2])) / den;
378     y = (v * (A[1]-B[1]) - u*(B[1]-C[1])) / den;
379     
380     return new JXG.Coords(JXG.COORDS_BY_USER, [x, y], board);
381 };

383 /**
384  * Calculates euclidean norm for two given arrays of the same length.
385  * @param {array} array1 Array of float or integer.
386  * @param {array} array2 Array of float or integer.
387  * @type number
388  * @return Euclidean distance of the given vectors.
389  */
390 JXG.Math.Geometry.distance = function(array1, array2) {
391     var sum = 0, 
392         i, len;
393         
394     if(array1.length != array2.length) { return; }
395     len = array1.length;
396     for(i=0; i<len; i++) {
397         sum += (array1[i] - array2[i])*(array1[i] - array2[i]);
398     }
399     return Math.sqrt(sum);
400 };
401 
402 /**
403  * Calculates euclidean distance for two given arrays of the same length.
404  * If one of the arrays contains a zero in the first coordinate, and the euclidean distance
405  * is different from zero it is a point at infinity and we return Infinity.
406  * @param {array} array1 Array containing elements of number.
407  * @param {array} array2 Array containing elements of type number.
408  * @type number
409  * @return Euclidean (affine) distance of the given vectors.
410  */
411 JXG.Math.Geometry.affineDistance = function(array1, array2) {
412     var d;
413     if(array1.length != array2.length) { 
414         return; 
415     }
416     d = this.distance(array1, array2);
417     if (d>JXG.Math.eps && (Math.abs(array1[0])<JXG.Math.eps || Math.abs(array2[0])<JXG.Math.eps)) {
418         return Infinity;
419     } else {
420         return d;
421     }
422 };

429 /**
430  * Calculates the coordinates of the intersection of the given lines.
431  * @param {JXG.Line} line1 Line.
432  * @param {JXG.Line} line2 Line.
433  * @param {JXG.Board} [board=line1.board] Reference to the board
434  * @type JXG.Coords
435  * @return Coordinates of the intersection point of the given lines.
436  */
437 JXG.Math.Geometry.intersectLineLine = function(line1, line2, board) {
438     var A = line1.point1.coords.usrCoords,
439         B = line1.point2.coords.usrCoords,
440         C = line2.point1.coords.usrCoords,
441         D = line2.point2.coords.usrCoords,
442         d0, d1, den, x, y;
443 
444     if(typeof board == 'undefined')
445         board = line1.board;
446 
447     d0 = A[1]*B[2] - A[2]*B[1];
448     d1 = C[1]*D[2] - C[2]*D[1];
449     den = (B[2]-A[2])*(C[1]-D[1]) - (A[1]-B[1])*(D[2]-C[2]);
450                  
451     if(Math.abs(den) < JXG.Math.eps) {
452          den = JXG.Math.eps; 
453     }
454     x = (d0*(C[1]-D[1]) - d1*(A[1]-B[1])) / den;
455     y = (d1*(B[2]-A[2]) - d0*(D[2]-C[2])) / den;
456 
457     return new JXG.Coords(JXG.COORDS_BY_USER, [x, y], board);
458 };

460 /**
461  * Calculates the coordinates of the intersection of the given line and circle.
462  * @param {JXG.Circle} circle Circle.
463  * @param {JXG.Line} line Line.
464  * @param {JXG.Board} [board=line.board] Reference to the board
465  * @type array
466  * @return Array of the Coordinates of the intersection points of the given circle with the given line and
467  * the amount of intersection points in the first component of the array.
468  */
469 JXG.Math.Geometry.intersectCircleLine = function(circle, line, board) {
470     var eA = line.point1.coords.usrCoords,
471         eB = line.point2.coords.usrCoords,
472         fM = circle.midpoint.coords.usrCoords,
473         s, d0, d1, b, w, h, r, n1, dx, dy, firstPointX, firstPointY, l, x, y, n1s, firstPoint, secondPoint, d;
474 
475     if(typeof board == 'undefined')
476         board = line.board;
477 
478     s = line.point1.Dist(line.point2);
479     if (s > 0) {
480         d0 = circle.midpoint.Dist(line.point1);
481         d1 = circle.midpoint.Dist(line.point2);
482         b = ((d0 * d0) + (s * s) - (d1 * d1)) / (2 * s);
483         w = (d0 * d0) - (b * b);
484         w = (w < 0) ? 0 : w;
485         h = Math.sqrt(w);
486         
487         r = circle.Radius();
488         n1 = Math.sqrt((r * r) - h*h);
489         dx = eB[1] - eA[1];
490         dy = eB[2] - eA[2];
491         firstPointX = fM[1] + (h / s) * dy;
492         firstPointY = fM[2] - (h / s) * dx;
493         d0 = (eB[1] * dy) - (eB[2] * dx);
494         d1 = (firstPointX * dx) + (firstPointY * dy);
495         l = (dy * dy) + (dx * dx);
496         if (Math.abs(l) < JXG.Math.eps) { l = JXG.Math.eps; }
497         x = ((d0 * dy) + (d1 * dx)) / l;
498         y = ((d1 * dy) - (d0 * dx)) / l;
499         n1s = n1/s;
500         firstPoint =  new JXG.Coords(JXG.COORDS_BY_USER, [x + n1s * dx, y + n1s * dy], board);
501         secondPoint = new JXG.Coords(JXG.COORDS_BY_USER, [x - n1s * dx, y - n1s * dy], board);
502         d = circle.midpoint.coords.distance(JXG.COORDS_BY_USER, firstPoint);
503       
504         if ((r < (d - 1)) || isNaN(d)) {
505             return [0];
506         } else {
507             return [2,firstPoint,secondPoint];
508         }
509     }
510     return [0];
511 };

513 /**
514  * Calculates the coordinates of the intersection of the given circles.
515  * @param {JXG.Circle} circle1 Circle.
516  * @param {JXG.Circle} circle2 Circle.
517  * @param {JXG.Board} [board=circle1.board] Reference to the board
518  * @type array
519  * @return Array of the Coordinates of the intersection points of the given circles and the
520  * amount of intersection points in the first component of the array.
521  */
522 JXG.Math.Geometry.intersectCircleCircle = function(circle1, circle2, board) { 
523     var intersection = {},
524         r1 = circle1.Radius(),
525         r2 = circle2.Radius(),
526         M1 = circle1.midpoint.coords.usrCoords,
527         M2 = circle2.midpoint.coords.usrCoords,
528         rSum, rDiff, s, 
529         dx, dy, a, h;
530 
531     if(typeof board == 'undefined')
532         board = circle1.board;
533 
534     rSum = r1 + r2;
535     rDiff = Math.abs(r1 - r2);    
536     // Abstand der Mittelpunkte der beiden Kreise
537     s = circle1.midpoint.coords.distance(JXG.COORDS_BY_USER, circle2.midpoint.coords);
538     if (s > rSum) {
539         return [0]; // Kreise schneiden sich nicht, liegen nebeneinander
540     } 
541     else if (s < rDiff) {
542         return [0]; // Kreise schneiden sich nicht, liegen ineinander
543     } 
544     else {
545         if (s != 0) {
546             intersection[0] = 1; // es gibt einen Schnitt        
547             dx = M2[1] - M1[1];
548             dy = M2[2] - M1[2];
549             a = (s * s - r2 * r2 + r1 * r1) / (2 * s);
550             h = Math.sqrt(r1 * r1 - a * a);
551             intersection[1] = new JXG.Coords(JXG.COORDS_BY_USER, 
552                                              [M1[1] + (a / s) * dx + (h / s) * dy, 
553                                               M1[2] + (a / s) * dy - (h / s) * dx], 
554                                              board);
555             intersection[2] = new JXG.Coords(JXG.COORDS_BY_USER, 
556                                              [M1[1] + (a / s) * dx - (h / s) * dy, 
557                                               M1[2] + (a / s) * dy + (h / s) * dx], 
558                                              board);    
559         }
560         else {
561             return [0]; // vorsichtshalber... 
562         }                                     
563         return intersection;
564     }
565 };

567 /**
568  * @private
569  * Computes the intersection of a pair of lines, circles or both.
570  * It uses the internal data array stdform of these elements.
571  * @param {Array} el1 stdform of the first element (line or circle)
572  * @param {Array} el2 stdform of the second element (line or circle)
573  * @param {number} i Index of the intersection point that should be returned.
574  * @param board Reference to the board.
575  * @type JXG.Coords
576  * @return Coordinates of one of the possible two or more intersection points. 
577  * Which point will be returned is determined by i.
578  */
579 JXG.Math.Geometry.meet = function(el1, el2, /** number */ i, /** JXG.Board */ board) /** JXG.Coords */ {
580     var eps = JXG.Math.eps; //    var eps = 0.000001;
581 
582     if (Math.abs(el1[3])<eps && Math.abs(el2[3])<eps) { // line line
583         return this.meetLineLine(el1,el2,i,board);
584     } else if (Math.abs(el1[3])>=eps && Math.abs(el2[3])<eps) { // circle line
585         return this.meetLineCircle(el2,el1,i,board);
586     } else if (Math.abs(el1[3])<eps && Math.abs(el2[3])>=eps) { // line circle
587         return this.meetLineCircle(el1,el2,i,board);
588     } else {  // circle circle
589         return this.meetCircleCircle(el1,el2,i,board);
590     }
591 };

593 /**
594   * @private
595   * Intersection of two lines using the stdform.
596   * @param {Array} l1 stdform of the first line
597   * @param {Array} l2 stdform of the second line
598   * @param {number} i unused
599   * @param {JXG.Board} board Reference to the board.
600   * @type JXG.Coords
601   * @return Coordinates of the intersection point.
602   */
603 JXG.Math.Geometry.meetLineLine = function(l1,l2,i,board) {
604     var s = JXG.Math.crossProduct(l1,l2);
605     if (Math.abs(s[0])>JXG.Math.eps) {
606         s[1] /= s[0];
607         s[2] /= s[0];
608         s[0] = 1.0;
609     }
610     return new JXG.Coords(JXG.COORDS_BY_USER, s, board);
611 };

613 /**
614   * @private
615   * 
616   * Intersection of line and circle using the stdform.
617   * 
618   * @param {Array} lin stdform of the line
619   * @param {Array} circ stdform of the circle
620   * @param {number} i number of the returned intersection point. 
621   *   i==0: use the positive square root, 
622   *   i==1: use the negative square root.
623   * @param {JXG.Board} board Reference to a board.
624   * @type JXG.Coords
625   * @return Coordinates of the intersection point
626   */
627  JXG.Math.Geometry.meetLineCircle = function(lin,circ,i,board) {
628     var a,b,c,d,n, A,B,C, k,t;
629 
630     if (circ[4]<JXG.Math.eps) { // Radius is zero, return center of circle
631         return new JXG.Coords(JXG.COORDS_BY_USER, circ.slice(1,3), board);
632     }
633     c = circ[0];
634     b = circ.slice(1,3);
635     a = circ[3];
636     d = lin[0];
637     n = lin.slice(1,3);
638 
639     // Line is normalized, therefore nn==1 and we can skip some operations:
640     /*
641     var nn = n[0]*n[0]+n[1]*n[1];
642     A = a*nn;
643     B = (b[0]*n[1]-b[1]*n[0])*nn;
644     C = a*d*d - (b[0]*n[0]+b[1]*n[1])*d + c*nn;
645     */
646     A = a;
647     B = (b[0]*n[1]-b[1]*n[0]);
648     C = a*d*d - (b[0]*n[0]+b[1]*n[1])*d + c;
649 
650     k = B*B-4*A*C;
651     if (k>=0) {
652         k = Math.sqrt(k);
653         t = [(-B+k)/(2*A),(-B-k)/(2*A)];
654         return ((i==0)
655             ? new JXG.Coords(JXG.COORDS_BY_USER, [-t[0]*(-n[1])-d*n[0],-t[0]*n[0]-d*n[1]], board)
656             : new JXG.Coords(JXG.COORDS_BY_USER, [-t[1]*(-n[1])-d*n[0],-t[1]*n[0]-d*n[1]], board)
657             );
658 /*
659             new JXG.Coords(JXG.COORDS_BY_USER, [-t[0]*(-n[1])-d*n[0]/nn,-t[0]*n[0]-d*n[1]/nn], this.board),
660             new JXG.Coords(JXG.COORDS_BY_USER, [-t[1]*(-n[1])-d*n[0]/nn,-t[1]*n[0]-d*n[1]/nn], this.board)
661 */
662     } else {
663         return new JXG.Coords(JXG.COORDS_BY_USER, [NaN,NaN], board);
664     }
665     // Returns do not work with homogeneous coordinates, yet
666 };

668 /**
669   * @private
670   * 
671   * Intersection of two circles using the stdform.
672   * 
673   * @param {Array} circ1 stdform of the first circle
674   * @param {Array} circ2 stdform of the second circle
675   * @param {number} i number of the returned intersection point. 
676   *   i==0: use the positive square root, 
677   *   i==1: use the negative square root.
678   * @param {JXG.Board} board Reference to the board.
679   * @type JXG.Coords
680   * @return Coordinates of the intersection point
681   */
682 JXG.Math.Geometry.meetCircleCircle = function(circ1,circ2,i,board) {
683     var radicalAxis;
684     if (circ1[4]<JXG.Math.eps) { // Radius are zero, return center of circle, if on other circle
685         if (this.distance(circ1.slice(1,3),circ2.slice(1,3))==circ2[4]) {
686             return new JXG.Coords(JXG.COORDS_BY_USER, circ1.slice(1,3), board);
687         } else {
688             return new JXG.Coords(JXG.COORDS_BY_USER, [NaN,NaN], board);
689         }
690     }
691     if (circ2[4]<JXG.Math.eps) { // Radius are zero, return center of circle, if on other circle
692         if (this.distance(circ2.slice(1,3),circ1.slice(1,3))==circ1[4]) {
693             return new JXG.Coords(JXG.COORDS_BY_USER, circ2.slice(1,3), board);
694         } else {
695             return new JXG.Coords(JXG.COORDS_BY_USER, [NaN,NaN], board);
696         }
697     }
698     radicalAxis = [circ2[3]*circ1[0]-circ1[3]*circ2[0],
699                    circ2[3]*circ1[1]-circ1[3]*circ2[1],
700                    circ2[3]*circ1[2]-circ1[3]*circ2[2],
701                    0,1,Infinity, Infinity, Infinity];
702     radicalAxis = JXG.Math.normalize(radicalAxis);
703     return this.meetLineCircle(radicalAxis,circ1,i,board);
704     // Returns do not work with homogeneous coordinates, yet
705 };

707 /**
708  * Compute an intersection of the curves c1 and c2
709  * with a generalized Newton method.
710  * We want to find values t1, t2 such that
711  * c1(t1) = c2(t2), i.e.
712  * (c1_x(t1)-c2_x(t2),c1_y(t1)-c2_y(t2)) = (0,0).
713  * We set
714  * (e,f) := (c1_x(t1)-c2_x(t2),c1_y(t1)-c2_y(t2))
715  *
716  * The Jacobian J is defined by
717  * J = (a, b)
718  *     (c, d)
719  * where
720  * a = c1_x'(t1)
721  * b = -c2_x'(t2)
722  * c = c1_y'(t1)
723  * d = -c2_y'(t2)
724  *
725  * The inverse J^(-1) of J is equal to
726  *  (d, -b)/
727  *  (-c, a) / (ad-bc)
728  *
729  * Then, (t1new, t2new) := (t1,t2) - J^(-1)*(e,f).
730  * If the function meetCurveCurve possesses the properties
731  * t1memo and t2memo then these are taken as start values
732  * for the Newton algorithm.
733  * After stopping of the Newton algorithm the values of t1 and t2 are stored in
734  * t1memo and t2memo.
735  * 
736  * @param {JXG.Curve} c1 Curve, Line or Circle
737  * @param {JXG.Curve} c2 Curve, Line or Circle
738  * @param {float} t1ini start value for t1
739  * @param {float} t2ini start value for t2
740  * @param {JXG.Board} [board=c1.board] Reference to a board object.
741  * @type {JXG.Coords}
742  * @return coordinate object for the intersection point
743  **/
744 JXG.Math.Geometry.meetCurveCurve = function(c1,c2,t1ini,t2ini,board) {
745     var count = 0,
746         t1, t2,
747         a, b, c, d, disc,
748         e, f, F, 
749         D00, D01, 
750         D10, D11;
751     
752     if(typeof board == 'undefined')
753         board = c1.board;
754         
755     if (arguments.callee.t1memo) {
756         t1 = arguments.callee.t1memo;
757         t2 = arguments.callee.t2memo;
758     } else {
759         t1 = t1ini;
760         t2 = t2ini;
761     }
762 /*
763     if (t1>c1.maxX()) { t1 = c1.maxX(); }
764     if (t1<c1.minX()) { t1 = c1.minX(); }
765     if (t2>c2.maxX()) { t2 = c2.maxX(); }
766     if (t2<c2.minX()) { t2 = c2.minX(); }
767 */
768     e = c1.X(t1)-c2.X(t2);
769     f = c1.Y(t1)-c2.Y(t2);
770     F = e*e+f*f;
771     
772     D00 = c1.board.D(c1.X,c1);
773     D01 = c2.board.D(c2.X,c2);
774     D10 = c1.board.D(c1.Y,c1);
775     D11 = c2.board.D(c2.Y,c2);
776     
777     while (F>JXG.Math.eps && count<10) {
778         a =  D00(t1);
779         b = -D01(t2);
780         c =  D10(t1);
781         d = -D11(t2);
782         disc = a*d-b*c;
783         t1 -= (d*e-b*f)/disc;
784         t2 -= (a*f-c*e)/disc;
785         e = c1.X(t1)-c2.X(t2);
786         f = c1.Y(t1)-c2.Y(t2);
787         F = e*e+f*f;
788         count++;
789     }
790 //console.log(t1+' '+t2);
791 
792     arguments.callee.t1memo = t1;
793     arguments.callee.t2memo = t2;
794 
795     //return (new JXG.Coords(JXG.COORDS_BY_USER, [2,2], this.board));
796     if (Math.abs(t1)<Math.abs(t2)) {
797         return (new JXG.Coords(JXG.COORDS_BY_USER, [c1.X(t1),c1.Y(t1)], board));
798     } else {
799         return (new JXG.Coords(JXG.COORDS_BY_USER, [c2.X(t2),c2.Y(t2)], board));
800     }
801 };

803 /**
804  * order of input does not matter for el1 and el2.
805  * @param {JXG.Curve,JXG.Line} el1 Curve or Line
806  * @param {JXG.Curve,JXG.Line} el2 Curve or Line
807  * @param {?} nr
808  * @param {JXG.Board} [board=el1.board] Reference to a board object.
809  * @type {JXG.Coords}
810  * @return coordinate object for the intersection point
811  */
812 JXG.Math.Geometry.meetCurveLine = function(el1,el2,nr,board) {
813     var t, t2, i, cu, li, func, z,
814         tnew, steps, delta, tstart, cux, cuy;
815     
816     if(typeof board == 'undefined')
817         board = el1.board;
818     
819     
820     for (i=0;i<arguments.length-1;i++) {
821         if (arguments[i].elementClass==JXG.OBJECT_CLASS_CURVE) { cu = arguments[i]; }
822         else if (arguments[i].elementClass==JXG.OBJECT_CLASS_LINE) { li = arguments[i]; }
823         else 
824             throw new Error("JSXGraph: Can't call meetCurveLine with parent class '" + (arguments[i].elementClass) + ".");
825     }
826     
827     func = function(t) {
828         return li.stdform[0]*1.0 + li.stdform[1]*cu.X(t) + li.stdform[2]*cu.Y(t);
829     };
830     
831     if (arguments.callee.t1memo) {
832         tstart = arguments.callee.t1memo;
833     } else {
834         tstart = cu.minX();
835     }
836     t = JXG.Math.Numerics.root(func, tstart);
837     arguments.callee.t1memo = t;
838     cux = cu.X(t);
839     cuy = cu.Y(t);
840     
841     if (nr==1) {  
842         if (arguments.callee.t2memo) {
843             tstart = arguments.callee.t2memo;
844             t2 = JXG.Math.Numerics.root(func, tstart);
845         } 
846         if (!(Math.abs(t2-t)>0.1 && Math.abs(cux-cu.X(t2))>0.1 && Math.abs(cuy-cu.Y(t2))>0.1)) {
847             steps = 20;
848             delta = (cu.maxX()-cu.minX())/steps;
849             tnew = cu.minX();
850             for (i=0;i<steps;i++) {
851                 t2 = JXG.Math.Numerics.root(func, tnew);
852                 if (Math.abs(t2-t)>0.1 && Math.abs(cux-cu.X(t2))>0.1 && Math.abs(cuy-cu.Y(t2))>0.1) {
853                     break;
854                 }
855                 tnew += delta;
856             }
857         }
858         t = t2;
859         arguments.callee.t2memo = t;
860     }
861 
862     if (Math.abs(func(t))>JXG.Math.eps)
863         z = 0.0;
864     else
865         z = 1.0;
866     
867     return (new JXG.Coords(JXG.COORDS_BY_USER, [z, cu.X(t),cu.Y(t)], board));
868 };

964 /**
965  * Calculates the coordinates of the projection of a coordinates pair on a given curve. In case of
966  * function graphs this is the
967  * intersection point of the curve and the parallel to y-axis through the given point.
968  * @param {float} x coordinate to project.
969  * @param {float} y coordinate to project.
970  * @param {float} start value for newtons method
971  * @param {JXG.Curve} graph Curve on that the point is projected.
972  * @param {JXG.Board} [board=curve.board] Reference to a board.
973  * @type JXG.Coords
974  * @see #projectPointToCurve
975  * @return Array containing the coordinates of the projection of the given point on the given graph and 
976  * the position on the curve.
977  */
978 JXG.Math.Geometry.projectCoordsToCurve = function(x,y,t,curve,board) {
979     var newCoords, x0, y0, x1, y1, den, i, mindist, dist, lbda, j,
980         infty = 1000000.0, minfunc, tnew, fnew, fold, delta, steps;
981         
982     if(typeof board=='undefined')
983         board = curve.board;
984         
985     if (curve.curveType=='parameter' || curve.curveType=='polar') { 
986         // Function to minimize
987         minfunc = function(t){ 
988                     var dx = x-curve.X(t),
989                         dy = y-curve.Y(t);
990                     return dx*dx+dy*dy;
991                 };
992         //t = JXG.Math.Numerics.root(JXG.Math.Numerics.D(minfunc),t);
993         fold = minfunc(t);
994         steps = 20;
995         delta = (curve.maxX()-curve.minX())/steps;
996         tnew = curve.minX();
997         for (j=0;j<steps;j++) {
998             fnew = minfunc(tnew);
999             if (fnew<fold) {
1000                 t = tnew;
1001                 fold = fnew;
1002             }
1003             tnew += delta;
1004         }
1005         t = JXG.Math.Numerics.root(JXG.Math.Numerics.D(minfunc),t);
1006 
1007         if (t<curve.minX()) { t = curve.maxX()+t-curve.minX(); } // Cyclically
1008         if (t>curve.maxX()) { t = curve.minX()+t-curve.maxX(); }
1009         newCoords = new JXG.Coords(JXG.COORDS_BY_USER, [curve.X(t),curve.Y(t)], board);
1010     } else if (curve.curveType == 'plot') {
1011         mindist = infty;
1012         for (i=0;i<curve.numberPoints;i++) {
1013             x0 = x-curve.X(i);
1014             y0 = y-curve.Y(i);
1015             dist = Math.sqrt(x0*x0+y0*y0);
1016             if (dist<mindist) {
1017                 mindist = dist;
1018                 t = i;
1019             }
1020             if (i==curve.numberPoints-1) { continue; }
1021 
1022             x1 = curve.X(i+1)-curve.X(i);
1023             y1 = curve.Y(i+1)-curve.Y(i);
1024             den = x1*x1+y1*y1;
1025             if (den>=JXG.Math.eps) {
1026                 lbda = (x0*x1+y0*y1)/den;
1027                 dist = Math.sqrt( x0*x0+y0*y0 - lbda*(x0*x1+y0*y1) );
1028             } else {
1029                 lbda = 0.0;
1030                 dist = Math.sqrt(x0*x0+y0*y0);
1031             }
1032             if (lbda>=0.0 && lbda<=1.0 && dist<mindist) { 
1033                 t = i+lbda;
1034                 mindist = dist;
1035             } 
1036         }
1037         i = Math.floor(t);
1038         lbda = t-i;
1039         if (i<curve.numberPoints-1) {
1040             x = lbda*curve.X(i+1)+(1.0-lbda)*curve.X(i);
1041             y = lbda*curve.Y(i+1)+(1.0-lbda)*curve.Y(i);
1042         } else {
1043             x = curve.X(i);
1044             y = curve.Y(i);
1045         }
1046         newCoords = new JXG.Coords(JXG.COORDS_BY_USER, [x,y], board); 
1047     } else {             // functiongraph
1048         t = x;
1049         x = t; //curve.X(t);
1050         y = curve.Y(t);
1051         newCoords = new JXG.Coords(JXG.COORDS_BY_USER, [x,y], board); 
1052     }
1053     return [curve.updateTransform(newCoords),t];
1054 };

1056 /**
1057  * Calculates the coordinates of the projection of a given point on a given turtle. A turtle consists of
1058  * one or more curves of curveType 'plot'. Uses {@link #projectPointToCurve}.
1059  * @param {JXG.Point} point Point to project.
1060  * @param {JXG.Turtle} turtle on that the point is projected.
1061  * @param {JXG.Board} [board=point.board] Reference to a board.
1062  * @type JXG.Coords
1063  * @return The coordinates of the projection of the given point on the given turtle.
1064  */
1065 JXG.Math.Geometry.projectPointToTurtle = function(point,turtle,board) {
1066     var newCoords, t, x, y, i,
1067         np = 0, 
1068         npmin = 0,
1069         mindist = 1000000.0, 
1070         dist, el, minEl, 
1071         len = turtle.objects.length;
1072         
1073     if(typeof board == 'undefined')
1074         board = point.board;
1075     
1076     for(i=0;i<len;i++) {  // run through all curves of this turtle
1077         el = turtle.objects[i];
1078         if (el.elementClass==JXG.OBJECT_CLASS_CURVE) {
1079             newCoords = this.projectPointToCurve(point,el);
1080             dist = this.distance(newCoords.usrCoords,point.coords.usrCoords);
1081             if (dist<mindist) {
1082                 x = newCoords.usrCoords[1];
1083                 y = newCoords.usrCoords[2];
1084                 t = point.position;
1085                 mindist = dist;
1086                 minEl = el;
1087                 npmin = np;
1088             }
1089             np += el.numberPoints;
1090         }
1091     }
1092     newCoords = new JXG.Coords(JXG.COORDS_BY_USER, [x,y], board);
1093     point.position = t+npmin;
1094     return minEl.updateTransform(newCoords);
1095 };
