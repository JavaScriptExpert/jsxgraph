 28  * @fileoverview AbstractRenderer is the base class for all renderers. This class is subject
 29  * to change the next time that only a few methods are strictly required for a special
 30  * renderer derived from this class to have a working special renderer. Of course other

 34 /**
 35  * Constructs a new AbstractRenderer object.
 36  * @class The AbstractRenderer is a base class that should be considered what in other languages
 37  * is called an abstract class, even though no such thing really exists in JavaScript. All members
 38  * essential for a renderer are defined here.
 39  * @constructor
 40  * @see JXG.SVGRenderer
 41  * @see JXG.VMLRenderer
 42  */
 43 JXG.AbstractRenderer = function() {
 44 	/**
 45 	 * The vertical offset for {@link Text} elements. Every {@link Text} element will
 46 	 * be placed this amount of pixels below the user given coordinates.
 47 	 * @type number
 48 	 * @default 8
 49 	 */
 50     this.vOffsetText = 8;
 51     
 52     /**
 53      * If this property is set to <tt>true</tt> the visual properties of the elements are updated
 54      * on every update. Visual properties means: All the stuff stored in the
 55      * {@link GeometryElement#visProp} property won't be set if enhancedRendering is <tt>false</tt>
 56      * @type boolean
 57      * @default true 
 58      */
 59     this.enhancedRendering = true;
 60 };

 67 /**
 68  * Draws a point on the {@link JXG.Board}.
 69  * @param el Reference to a {@link Point} object, that has to be drawn.
 70  * @see Point
 71  * @see JXG.Point
 72  * @see #updatePoint
 73  * @see #changePointStyle
 74  */
 75 JXG.AbstractRenderer.prototype.drawPoint = function(/** Point */ el) {
 76     var node,
 77         f = el.visProp['face'];
 78 
 79     // determine how the point looks like
 80     if(f == 'cross' || f == 'x') { // x
 81         node = this.createPrim('path',el.id);
 82         this.appendChildPrim(node,el.layer);
 83         this.appendNodesToElement(el, 'path');
 84     }
 85     else if(f == 'circle' || f == 'o') { // circle
 86         node = this.createPrim('circle',el.id);
 87         this.appendChildPrim(node,el.layer);
 88         this.appendNodesToElement(el, 'circle');
 89     }
 90     else if(f == 'square' || f == '[]') { // rectangle
 91         node = this.createPrim('rect',el.id);
 92         this.appendChildPrim(node,el.layer);
 93         this.appendNodesToElement(el, 'rect');
 94     }
 95     else if(f == 'plus' || f == '+') { // +
 96         node = this.createPrim('path',el.id);
 97         this.appendChildPrim(node,el.layer);  
 98         this.appendNodesToElement(el, 'path');
 99     }
100     else if(f == 'diamond' || f == '<>') {
101         node = this.createPrim('path',el.id);
102         this.appendChildPrim(node,el.layer);  
103         this.appendNodesToElement(el, 'path');
104     }
105     else if(f == 'triangleup' || f == 'a' || f == '^') {
106         node = this.createPrim('path',el.id);
107         this.appendChildPrim(node,el.layer);  
108         this.appendNodesToElement(el, 'path');    
109     }
110     else if(f == 'triangledown' || f == 'v') {
111         node = this.createPrim('path',el.id);
112         this.appendChildPrim(node,el.layer);  
113         this.appendNodesToElement(el, 'path');
114     }        
115     else if(f == 'triangleleft' || f == '<') {
116         node = this.createPrim('path',el.id);
117         this.appendChildPrim(node,el.layer);  
118         this.appendNodesToElement(el, 'path');    
119     }
120     else if(f == 'triangleright' || f == '>') {
121         node = this.createPrim('path',el.id);
122         this.appendChildPrim(node,el.layer);  
123         this.appendNodesToElement(el, 'path');    
124     }
125     
126     // adjust visual propertys
127     this.setObjectStrokeWidth(el,el.visProp['strokeWidth']);
128     this.setObjectStrokeColor(el,el.visProp['strokeColor'],el.visProp['strokeOpacity']);
129     this.setObjectFillColor(el,el.visProp['fillColor'],el.visProp['fillOpacity']);
130 
131     // By now we only created the xml nodes and set some styles, in updatePoint
132     // the attributes are filled with data.
133     this.updatePoint(el);
134 };

136 /**
137  * Updates visual appearance of the renderer element assigned to the given {@link Point}.
138  * @param el Reference to a {@link Point} object, that has to be updated.
139  * @see Point
140  * @see JXG.Point
141  * @see #drawPoint
142  * @see #changePointStyle
143  */
144 JXG.AbstractRenderer.prototype.updatePoint = function(/** Point */ el) {
145     var size = el.visProp['size'],
146     	f = el.visProp['face'];
147     if (isNaN(el.coords.scrCoords[2]) || isNaN(el.coords.scrCoords[1])) return;
148     
149     if (this.enhancedRendering) {
150         if (!el.visProp['draft']) {
151             this.setObjectStrokeWidth(el,el.visProp['strokeWidth']);
152             this.setObjectStrokeColor(el,el.visProp['strokeColor'],el.visProp['strokeOpacity']);
153             this.setObjectFillColor(el,el.visProp['fillColor'],el.visProp['fillOpacity']);
154         } else {
155             this.setDraft(el);
156         }
157     }
158     // Zoom does not work for traces.
159     size *= ((!el.board || !el.board.options.point.zoom)?1.0:Math.sqrt(el.board.zoomX*el.board.zoomY));
160     
161     if(f == 'cross' || f == 'x') { // x
162         this.updatePathPrim(el.rendNode, this.updatePathStringPoint(el, size,'x'), el.board); 
163     }
164     else if(f == 'circle' || f == 'o') { // circle
165         this.updateCirclePrim(el.rendNode,el.coords.scrCoords[1], el.coords.scrCoords[2], size+1);            
166     }
167     else if(f == 'square' || f == '[]') { // rectangle
168         this.updateRectPrim(el.rendNode,
169                 el.coords.scrCoords[1]-size, el.coords.scrCoords[2]-size, size*2, size*2);
170     }
171     else if(f == 'plus' || f == '+') { // +
172         this.updatePathPrim(el.rendNode, this.updatePathStringPoint(el,size,'+'), el.board); 
173     }
174     else if(f == 'diamond' || f == '<>') { // diamond
175         this.updatePathPrim(el.rendNode, this.updatePathStringPoint(el,size,'diamond'), el.board); 
176     }  
177     else if(f == 'triangleup' || f == 'a') { // triangleUp
178         this.updatePathPrim(el.rendNode, this.updatePathStringPoint(el,size,'A'), el.board); 
179     } 
180     else if(f == 'triangledown' || f == 'v') { // triangleDown
181         this.updatePathPrim(el.rendNode, this.updatePathStringPoint(el,size,'v'), el.board); 
182     } 
183     else if(f == 'triangleleft' || f == '<') { // triangleLeft
184         this.updatePathPrim(el.rendNode, this.updatePathStringPoint(el,size,'<'), el.board); 
185     }  
186     else if(f == 'triangleright' || f == '>') { // triangleRight
187         this.updatePathPrim(el.rendNode, this.updatePathStringPoint(el,size,'>'), el.board); 
188     }        
189     this.setShadow(el);
190 };

192 /**
193  * Changes the style of a {@link Point} that already exists on the canvas. This is required because
194  * the point styles differ in what elements have to be drawn, e.g. if the point is marked by
195  * a "x" or a "+" two lines are drawn, if it's marked by spot a circle is drawn. This method removes
196  * the old renderer element(s) and creates the new one(s).
197  * @param el Reference to a {@link Point} object, that's style is changed.
198  * @see Point
199  * @see JXG.Point
200  * @see #updatePoint
201  * @see #drawPoint
202  */
203 JXG.AbstractRenderer.prototype.changePointStyle = function(/** Point */el) {
204     var node = this.getElementById(el.id);
205     if(node != null) {
206         this.remove(node);
207     }
208     this.drawPoint(el);
209     JXG.clearVisPropOld(el);
210 
211     if(!el.visProp['visible']) {
212         this.hide(el);
213     }
214     if(el.visProp['draft']) {
215         this.setDraft(el);
216     }
217 };

241 /**
242  * Updates visual appearance of the renderer element assigned to the given {@link Line}.
243  * @param el Reference to the {@link Line} object that has to be updated.
244  * @see Line
245  * @see JXG.Line
246  * @see #drawLine
247  * @see #calcStraight
248  */
249 JXG.AbstractRenderer.prototype.updateLine = function(/** Line */ el) {
250     var screenCoords1 = new JXG.Coords(JXG.COORDS_BY_USER, el.point1.coords.usrCoords, el.board),
251         screenCoords2 = new JXG.Coords(JXG.COORDS_BY_USER, el.point2.coords.usrCoords, el.board),
252         ax, ay, bx, by, beta, sgn, x, y, m;
253         
254     this.calcStraight(el,screenCoords1,screenCoords2); 
255     this.updateLinePrim(el.rendNode,screenCoords1.scrCoords[1],screenCoords1.scrCoords[2],
256                         screenCoords2.scrCoords[1],screenCoords2.scrCoords[2],el.board);
257 
258     // Update the image which is connected to the line:
259     if (el.image!=null) {
260         ax = screenCoords1.scrCoords[1];
261         ay = screenCoords1.scrCoords[2];
262         bx = screenCoords2.scrCoords[1];
263         by = screenCoords2.scrCoords[2];
264 
265         beta = Math.atan2(by-ay,bx-ax);
266         x = 250; //ax;
267         y = 256; //ay;//+el.image.size[1]*0.5;
268         m = [
269                  [1,                                    0,             0],
270                  [x*(1-Math.cos(beta))+y*Math.sin(beta),Math.cos(beta),-Math.sin(beta)],
271                  [y*(1-Math.cos(beta))-x*Math.sin(beta),Math.sin(beta), Math.cos(beta)]
272             ];
273         el.imageTransformMatrix = m;
274     }
275     
276     // if this line has arrows attached, update them, too.
277     this.makeArrows(el);
278     
279     if (this.enhancedRendering) {
280         if (!el.visProp['draft']) {
281             this.setObjectStrokeWidth(el,el.visProp['strokeWidth']);
282             this.setObjectStrokeColor(el,el.visProp['strokeColor'],el.visProp['strokeOpacity']);
283             this.setDashStyle(el,el.visProp);    
284             this.setShadow(el);
285         } else {
286             this.setDraft(el);
287         }
288     }     
289 };

291 /**
292  * Calculates drawing start and end point for a line. A segment is only drawn from start to end point, a straight line
293  * is drawn until it meets the boards boundaries.
294  * @param el Reference to a line object, that needs calculation of start and end point.
295  * @param point1 Coordinates of the point where line drawing begins. This value is calculated and set by this method.
296  * @param point2 Coordinates of the point where line drawing ends. This value is calculated and set by this method.
297  * @see Line
298  * @see JXG.Line
299  * @see #drawLine
300  * @see #updateLine
301  */
302 JXG.AbstractRenderer.prototype.calcStraight = function(/** Line */ el, /** JXG.Coords */ point1, /** JXG.Coords */ point2) {
303     var takePoint1, takePoint2, intersect1, intersect2, straightFirst, straightLast, 
304         //b, 
305         c, s, i, j, 
306         p1, p2;
307     
308     //b = el.board.algebra;
309     straightFirst = el.visProp['straightFirst'];
310     straightLast  = el.visProp['straightLast'];
311 
312 /*
313     if (Math.abs(point1.scrCoords[0])<b.eps||Math.abs(point2.scrCoords[0])<b.eps) {
314         straightFirst = true;
315         straightLast  = true;
316     }
317 */    
318     // If one of the point is an ideal point in homogeneous coordinates
319     // drawing of line segments or rays are not possible. 
320     if (Math.abs(point1.scrCoords[0])<JXG.Math.eps) {
321         straightFirst = true;
322     }
323     if (Math.abs(point2.scrCoords[0])<JXG.Math.eps) {
324         straightLast  = true;
325     }
326 
327     if ( !straightFirst && !straightLast ) {  // Do nothing in case of line segments (inside or outside of the board)
328         return;
329     }
330     
331     // Compute the stdform of the line in screen coordinates.
332     c = [];
333     c[0] = el.stdform[0] - 
334            el.stdform[1]*el.board.origin.scrCoords[1]/el.board.stretchX+
335            el.stdform[2]*el.board.origin.scrCoords[2]/el.board.stretchY;
336     c[1] = el.stdform[1]/el.board.stretchX;
337     c[2] = el.stdform[2]/(-el.board.stretchY);
338 
339     if (isNaN(c[0]+c[1]+c[2])) return; // p1=p2
340     
341     // Intersect the line with the four borders of the board.
342     s = [];
343     s[0] = JXG.Math.crossProduct(c,[0,0,1]);  // top
344     s[1] = JXG.Math.crossProduct(c,[0,1,0]);  // left
345     s[2] = JXG.Math.crossProduct(c,[-el.board.canvasHeight,0,1]);  // bottom
346     s[3] = JXG.Math.crossProduct(c,[-el.board.canvasWidth,1,0]);   // right
347 
348     // Normalize the intersections 
349     for (i=0;i<4;i++) {
350         if (Math.abs(s[i][0])>JXG.Math.eps) {
351             for (j=2;j>0;j--) {
352                 s[i][j] /= s[i][0];
353             }
354             s[i][0] = 1.0;
355         }
356     }
357     
358     takePoint1 = false;
359     takePoint2 = false;
360     if (!straightFirst &&    // Line starts at point1 and point2 is inside the board
361             point1.scrCoords[1]>=0.0 && point1.scrCoords[1]<=el.board.canvasWidth &&
362             point1.scrCoords[2]>=0.0 && point1.scrCoords[2]<=el.board.canvasHeight) {
363         takePoint1 = true;
364     }
365     if (!straightLast &&    // Line ends at point2 and point2 is inside the board
366             point2.scrCoords[1]>=0.0 && point2.scrCoords[1]<=el.board.canvasWidth &&
367             point2.scrCoords[2]>=0.0 && point2.scrCoords[2]<=el.board.canvasHeight) {
368         takePoint2 = true;
369     }
370 
371     if (Math.abs(s[1][0])<JXG.Math.eps) {           // line is parallel to "left", take "top" and "bottom"
372         intersect1 = s[0];                          // top
373         intersect2 = s[2];                          // bottom
374     } else if (Math.abs(s[0][0])<JXG.Math.eps) {           // line is parallel to "top", take "left" and "right"
375         intersect1 = s[1];                          // left
376         intersect2 = s[3];                          // right
377     } else if (s[1][2]<0) {                         // left intersection out of board (above)
378         intersect1 = s[0];                          // top
379         if (s[3][2]>el.board.canvasHeight) {        // right intersection out of board (below)
380             intersect2 = s[2];                      // bottom
381         } else {
382             intersect2 = s[3];                      // right
383         }
384     } else if (s[1][2]>el.board.canvasHeight) {     // left intersection out of board (below)
385         intersect1 = s[2];                          // bottom
386         if (s[3][2]<0) {                            // right intersection out of board (above)
387             intersect2 = s[0];                      // top
388         } else {
389             intersect2 = s[3];                      // right
390         }
391     } else {
392         intersect1 = s[1];                          // left
393         if (s[3][2]<0) {                            // right intersection out of board (above)
394             intersect2 = s[0];                      // top
395         } else if (s[3][2]>el.board.canvasHeight) { // right intersection out of board (below)
396             intersect2 = s[2];                      // bottom
397         } else {
398             intersect2 = s[3];                      // right
399         }
400     }
401     
402     intersect1 = new JXG.Coords(JXG.COORDS_BY_SCREEN, intersect1.slice(1), el.board);
403     intersect2 = new JXG.Coords(JXG.COORDS_BY_SCREEN, intersect2.slice(1), el.board);
404  
405     if (!takePoint1 && !takePoint2) {              // If both points are outside and the complete ray is outside we do nothing
406         if (!straightFirst && straightLast &&      // Ray starting at point 1
407             !this.isSameDirection(point1, point2, intersect1) && !this.isSameDirection(point1, point2, intersect2)) {
408             return;
409         } else if (straightFirst && !straightLast &&  // Ray starting at point 2
410             !this.isSameDirection(point2, point1, intersect1) && !this.isSameDirection(point2, point1, intersect2)) {
411             return;
412         }
413     }
414     
415     if (!takePoint1) {
416         if (!takePoint2) {                // Two border intersection points are used
417             if (this.isSameDirection(point1, point2, intersect1)) {
418                 if (!this.isSameDirection(point1, point2, intersect2)) {
419                     p2 = intersect1;
420                     p1 = intersect2;
421                 } else {
422                     if (el.board.algebra.affineDistance(point2.usrCoords,intersect1.usrCoords)<el.board.algebra.affineDistance(point2.usrCoords,intersect2.usrCoords)) {
423                         p1 = intersect1;
424                         p2 = intersect2;
425                     } else {
426                         p2 = intersect1;
427                         p1 = intersect2;
428                     }
429                 }
430             } else {
431                 if (this.isSameDirection(point1, point2, intersect2)) {
432                     p1 = intersect1;
433                     p2 = intersect2;
434                 } else {
435                     if (el.board.algebra.affineDistance(point2.usrCoords,intersect1.usrCoords)<el.board.algebra.affineDistance(point2.usrCoords,intersect2.usrCoords)) {
436                         p2 = intersect1;
437                         p1 = intersect2;
438                     } else {
439                         p1 = intersect1;
440                         p2 = intersect2;
441                     }
442                 }
443             }
444         } else {                          // Instead of point1 the border intersection is taken
445             if (this.isSameDirection(point2, point1, intersect1)) {
446                 p1 = intersect1;
447             } else {
448                 p1 = intersect2;
449             }
450         }
451     } else {
452         if (!takePoint2) {                // Instead of point2 the border intersection is taken
453             if (this.isSameDirection(point1, point2, intersect1)) {
454                 p2 = intersect1;
455             } else {
456                 p2 = intersect2;
457             }
458         }
459     }
460 
461     if (p1) point1.setCoordinates(JXG.COORDS_BY_USER, p1.usrCoords.slice(1));
462     if (p2) point2.setCoordinates(JXG.COORDS_BY_USER, p2.usrCoords.slice(1));
463 };

465 /**
466  * If you're looking from point "start" towards point "s" and can see the point "p", true is returned. Otherwise false.
467  * @param start The point you're standing on.
468  * @param p The point in which direction you're looking.
469  * @param s The point that should be visible.
470  * @return True, if from start the point p is in the same direction as s is, that means s-start = k*(p-start) with k>=0.
471  */
472 JXG.AbstractRenderer.prototype.isSameDirection = function(/** JXG.Coords */ start, /** JXG.Coords */ p, /** JXG.Coords */ s) /** boolean */ {
473     var dx, dy, sx, sy;
474     
475     dx = p.usrCoords[1]-start.usrCoords[1];
476     dy = p.usrCoords[2]-start.usrCoords[2];
477 
478     sx = s.usrCoords[1]-start.usrCoords[1];
479     sy = s.usrCoords[2]-start.usrCoords[2];
480 
481     if (Math.abs(dx)<JXG.Math.eps) dx=0;
482     if (Math.abs(dy)<JXG.Math.eps) dy=0;
483     if (Math.abs(sx)<JXG.Math.eps) sx=0;
484     if (Math.abs(sy)<JXG.Math.eps) sy=0;
485 
486     if (dx>=0&&sx>=0) {
487         if ((dy>=0&&sy>=0) || (dy<=0&&sy<=0)) { return true; }
488     } else if (dx<=0&&sx<=0){
489         if ((dy>=0&&sy>=0) || (dy<=0&&sy<=0)) { return true; }        
490     }
491 
492     return false;
493 };

551 /**
552  * Updates properties of an {@link Arrow} that already exists on the {@link JXG.Board}.
553  * @param el Reference to an {@link Line} object, that has to be updated.
554  * @see Arrow
555  * @see Line
556  * @see JXG.Line
557  * @see #drawArrow
558  */
559 JXG.AbstractRenderer.prototype.updateArrow = function(/** Line */ el) {

907 /**
908  * Creates a grid on the board, i.e. light helper lines to support the user on creating and manipulating a construction.
909  * @param board Board on which the grid is drawn.
910  * @see #removeGrid
911  */
912 JXG.AbstractRenderer.prototype.drawGrid = function(/** JXG.Board */ board) { 
913     var gridX = board.gridX,
914         gridY = board.gridY,
915         k = new JXG.Coords(JXG.COORDS_BY_SCREEN, [0,0], board),
916         k2 = new JXG.Coords(JXG.COORDS_BY_SCREEN, [board.canvasWidth, board.canvasHeight], board),
917         tmp = Math.ceil(k.usrCoords[1]),
918         j = 0,
919         i, j2, l, l2,
920         gx, gy, topLeft, bottomRight, node2,
921         el, eltmp;
922         
923     board.hasGrid = true;
924 
925     for(i = 0; i <= gridX+1; i++) {
926         if(tmp-i/gridX < k.usrCoords[1]) {
927             j = i-1;
928             break;
929         }
930     }
931 
932     tmp = Math.floor(k2.usrCoords[1]);
933     j2 = 0;
934     for(i = 0; i <= gridX+1; i++) {
935         if(tmp+i/gridX > k2.usrCoords[1]) {
936             j2 = i-1;
937             break;
938         }
939     } 
940 
941     tmp = Math.ceil(k2.usrCoords[2]);
942     l2 = 0;
943     for(i = 0; i <= gridY+1; i++) {
944         if(tmp-i/gridY < k2.usrCoords[2]) {
945             l2 = i-1;
946             break;
947         }
948     }
949 
950     tmp = Math.floor(k.usrCoords[2]);
951     l = 0;
952     for(i = 0; i <= gridY+1; i++) {
953         if(tmp+i/gridY > k.usrCoords[2]) {
954             l = i-1;
955             break;
956         }
957     }
958 
959     gx = Math.round((1.0/gridX)*board.stretchX);
960     gy = Math.round((1.0/gridY)*board.stretchY);
961 
962     topLeft = new JXG.Coords(JXG.COORDS_BY_USER, 
963                              [Math.ceil(k.usrCoords[1])-j/gridX, Math.floor(k.usrCoords[2])+l/gridY],
964                              board);
965     bottomRight = new JXG.Coords(JXG.COORDS_BY_USER,
966                                  [Math.floor(k2.usrCoords[1])+j2/gridX, Math.ceil(k2.usrCoords[2])-l2/gridY],
967                                  board);
968                                      
969     node2 = this.drawVerticalGrid(topLeft, bottomRight, gx, board);
970     this.appendChildPrim(node2, board.options.layer['grid']);
971     if(!board.snapToGrid) {
972         el = new Object();
973         el.visProp = {};
974         el.rendNode = node2;
975         el.elementClass = JXG.OBJECT_CLASS_LINE;
976         el.id = "gridx";
977         JXG.clearVisPropOld(el);
978         this.setObjectStrokeColor(el, board.gridColor, board.gridOpacity);
979     }
980     else {
981         el = new Object();
982         el.visProp = {};
983         el.rendNode = node2;
984         el.elementClass = JXG.OBJECT_CLASS_LINE;
985         el.id = "gridx";        
986         JXG.clearVisPropOld(el);
987         this.setObjectStrokeColor(el, '#FF8080', 0.5); //board.gridOpacity);    
988     }
989     this.setPropertyPrim(node2,'stroke-width', '0.4px');  
990     if(board.gridDash) {
991         this.setGridDash("gridx"); 
992     }
993 
994     node2 = this.drawHorizontalGrid(topLeft, bottomRight, gy, board);
995     this.appendChildPrim(node2, board.options.layer['grid']); // Attention layer=1
996     if(!board.snapToGrid) {
997         el = new Object();
998         el.visProp = {};
999         el.rendNode = node2;
1000         el.elementClass = JXG.OBJECT_CLASS_LINE;
1001         el.id = "gridy";   
1002         JXG.clearVisPropOld(el);
1003         this.setObjectStrokeColor(el, board.gridColor, board.gridOpacity);
1004     }
1005     else {
1006         el = new Object();
1007         el.visProp = {};

1010         el.id = "gridy";

1276 /**
1277  * findSplit() is a subroutine for {@link #RamenDouglasPeuker}.
1278  * It searches for the point between index i and j which 
1279  * has the largest distance from the line petween the points i and j.
1280  * @param pts Array of {@link Point}s
1281  * @param i Index of a point in pts
1282  * @param j Index of a point in pts
1283  **/
1284 JXG.AbstractRenderer.prototype.findSplit = function(/** array */ pts, /** number */ i, /** number */ j) {
1285     var dist = 0, 
1286         f = i, 
1287         d, k, ci, cj, ck,
1288         x0, y0, x1, y1,
1289         den, lbda;
1290         
1291     if (j-i<2) return [-1.0,0];
1292     
1293     ci = pts[i].scrCoords;
1294     cj = pts[j].scrCoords;
1295     if (isNaN(ci[1]+ci[2]+cj[1]+cj[2])) return [NaN,j];
1296     
1297     for (k=i+1; k<j; k++) {
1298         ck = pts[k].scrCoords;
1299         x0 = ck[1]-ci[1];
1300         y0 = ck[2]-ci[2];
1301         x1 = cj[1]-ci[1];
1302         y1 = cj[2]-ci[2];
1303         den = x1*x1+y1*y1;
1304         if (den>=JXG.Math.eps) {
1305             lbda = (x0*x1+y0*y1)/den;
1306             d = x0*x0+y0*y0 - lbda*(x0*x1+y0*y1);
1307         } else {
1308             lbda = 0.0;
1309             d = x0*x0+y0*y0;
1310         }
1311         if (lbda<0.0) {
1312             d = x0*x0+y0*y0;
1313         } else if (lbda>1.0) {
1314             x0 = ck[1]-cj[1];
1315             y0 = ck[2]-cj[2];
1316             d = x0*x0+y0*y0;
1317         }
1318         if (d>dist) {
1319             dist = d;
1320             f = k;
1321         }
1322     }
1323     return [Math.sqrt(dist),f];
1324 };

1326 /**
1327  * RDB() is a subroutine for {@link #RamenDouglasPeuker}.
1328  * It runs recursively through the point set and searches the
1329  * point which has the largest distance from the line between the first point and
1330  * the last point. If the distance from the line is greater than eps, this point is 
1331  * included in our new point set otherwise it is discarded.
1332  * If it is taken, we recursively apply the subroutine to the point set before
1333  * and after the chosen point.
1334  * @param pts Array of {@link Point}s
1335  * @param i Index of an element of pts
1336  * @param j Index of an element of pts
1337  * @param eps If the absolute value of a given number <tt>x</tt> is smaller than <tt>eps</tt> it is considered to be equal <tt>0</tt>.
1338  * @param newPts Array of {@link Point}s
1339  */
1340 JXG.AbstractRenderer.prototype.RDP = function(/** array */ pts, /** number */ i, /** number */ j, /** number */ eps, /** array */ newPts) {
1341     var result = this.findSplit(pts,i,j);
1342     if (result[0]>eps) {
1343         this.RDP(pts, i,result[1], eps,newPts);
1344         this.RDP(pts, result[1],j, eps,newPts);
1345     } else {
1346         newPts.push(pts[j]);
1347     }
1348 };

1350 /**
1351  * Ramen-Douglas-Peuker algorithm.
1352  * It discards points which are not necessary from the polygonal line defined by the point array
1353  * pts. The computation is done in screen coordinates.
1354  * Average runtime is O(nlog(n)), worst case runtime is O(n^2), where n is the number of points.
1355  * @param pts Array of {@link Point}s
1356  * @param eps If the absolute value of a given number <tt>x</tt> is smaller than <tt>eps</tt> it is considered to be equal <tt>0</tt>.
1357  * @return An array containing points which represent an apparently identical curve as the points of pts do, but contains fewer points.
1358  */
1359 JXG.AbstractRenderer.prototype.RamenDouglasPeuker = function(/** array */ pts, /** number */ eps) /** array */ {
1360     var newPts = [], i, k, len;
1361 
1362     len = pts.length;
1363     
1364     // Search for the left most point woithout NaN coordinates
1365     i = 0;
1366     while (i<len && isNaN(pts[i].scrCoords[1]+pts[i].scrCoords[2])) {i++;}
1367     // Search for the right most point woithout NaN coordinates
1368     k = len - 1;
1369     while (k>i && isNaN(pts[k].scrCoords[1]+pts[k].scrCoords[2])) {k--;}
1370 
1371     // Exit if nothing is left
1372     if (i>k || i==len) { return []; }
1373     
1374     newPts[0] = pts[i];
1375     this.RDP(pts,i,k,eps,newPts);
1376     //this.RDP(pts,0,pts.length-1,eps,newPts);
1377     return newPts;
1378 };
