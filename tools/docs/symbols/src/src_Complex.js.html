 27 /**
 28  * @fileoverview A class for complex arithmetics JXG.Complex is defined in this
 29  * file. Also a namespace JXG.C is included to provide instance-independent
 30  * arithmetic functions.
 31  * @author graphjs
 32  */
 33 
 34 /**
 35  * Creates a new complex number.
 36  * @class This class is for calculating with complex numbers.
 37  * @param [x=0] Real part of the resulting complex number.
 38  * @param [y=0] Imaginary part of the resulting complex number.
 39  * @returns An object representing the complex number <tt>x + iy</tt>.
 40  */
 41 JXG.Complex = function(/** number */ x, /** number */ y) {
 42 	/**
 43 	 * This property is only to signalize that this object is of type JXG.Complex. Only
 44 	 * used internally to distinguish between normal JavaScript numbers and JXG.Complex numbers.
 45 	 * @type boolean
 46 	 * @default true
 47 	 * @private
 48 	 */
 49     this.isComplex = true;
 50     
 51     if (typeof x == 'undefined') {
 52         x = 0;
 53     }
 54     if (typeof y == 'undefined') {
 55         y = 0;
 56     }
 57 
 58     /* is the first argument a complex number? if it is,
 59      * extract real and imaginary part. */
 60     if (x.isComplex) {
 61         y = x.imaginary;
 62         x = x.real;
 63     }
 64 
 65     /**
 66      * Real part of the complex number.
 67      * @type number
 68      * @default 0
 69      */
 70     this.real = x;
 71 
 72     /**
 73      * Imaginary part of the complex number.
 74      * @type number
 75      * @default 0
 76      */
 77     this.imaginary = y;
 78 
 79     /**
 80      * Absolute value in the polar form of the complex number. Currently unused.
 81      * @type number
 82      */
 83     this.absval = 0;
 84 
 85     /**
 86      * Angle value in the polar form of the complex number. Currently unused.
 87      * @type number
 88      */
 89     this.angle = 0;
 90 };
 91 
 92 /**
 93  * Converts a complex number into a string.
 94  * @return Formatted string containing the complex number in human readable form (algebraic form).
 95  */
 96 JXG.Complex.prototype.toString = function() /** string */{
 97     return '' + this.real + ' + ' + this.imaginary + 'i';
 98 };
 99 
100 /**
101  * Add another complex number to this complex number.
102  * @param c A JavaScript number or a JXG.Complex object to be added to the current object.
103  */
104 JXG.Complex.prototype.add = function(/** JXG.Complex,number */ c) /** undefined */ {
105     if(typeof c == 'number') {
106         this.real += c;
107     } else {
108         this.real += c.real;
109         this.imaginary += c.imaginary;
110     }
111 };
112 
113 /**
114  * Subtract another complex number from this complex number.
115  * @param c A JavaScript number or a JXG.Complex object to subtract from the current object.
116  */
117 JXG.Complex.prototype.sub = function(/** JXG.Complex,number */ c) /** undefined */{
118     if(typeof c == 'number') {
119         this.real -= c;
120     } else {
121         this.real -= c.real;
122         this.imaginary -= c.imaginary;
123     }
124 };
125 
126 /**
127  * Multiply another complex number to this complex number.
128  * @param c A JavaScript number or a JXG.Complex object to
129  * multiply with the current object.
130  */
131 JXG.Complex.prototype.mult = function(/** JXG.Complex,number */ c) /** undefined */{
132     if(typeof c == 'number') {
133         this.real *= c;
134         this.imaginary *= c;
135     } else {
136         //  (a+ib)(x+iy) = ax-by + i(xb+ay)
137         this.real = this.real*c.real - this.imaginary*c.imaginary;
138         this.imaginary = this.real*c.imaginary + this.imaginary*c.real;
139     }
140 };
141 
142 
143 /**
144  * Divide this complex number by the given complex number.
145  * @param c A JavaScript number or a JXG.Complex object to
146  * divide the current object by.
147  */
148 JXG.Complex.prototype.div = function(/** JXG.Complex,number */ c) /** undefined */{
149     var denom;
150 
151     if(typeof c == 'number') {
152         if(Math.abs(c) < Math.eps) {
153             this.real = Infinity;
154             this.imaginary = Infinity;
155             
156             return;
157         }
158         this.real /= c;
159         this.imaginary /= c;
160     } else {
161         //  (a+ib)(x+iy) = ax-by + i(xb+ay)
162         if( (Math.abs(c.real) < Math.eps) && (Math.abs(c.imaginary) < Math.eps) ){
163             this.real = Infinity;
164             this.imaginary = Infinity;
165 
166             return;
167         }
168 
169         denom = c.real*c.real + c.imaginary*c.imaginary;
170 
171         this.real = (this.real*c.real + this.imaginary*c.imaginary)/denom;
172         this.imaginary = (this.imaginary*c.real - this.real*c.imaginary)/denom;
173     }
174 };

177 /**
178  * @description
179  * JXG.C is the complex number (name)space. It provides functions to calculate with
180  * complex numbers (defined in {@link JXG.Complex}). With this namespace you don't have to modify
181  * your existing complex numbers, e.g. to add two complex numbers:
182  * <pre class="code">   var z1 = new JXG.Complex(1, 0);
183  *    var z2 = new JXG.Complex(0, 1);
184  *    z = JXG.C.add(z1, z1);</pre>
185  * z1 and z2 here remain unmodified. With the object oriented approach above this
186  * section the code would look like:
187  * <pre class="code">   var z1 = new JXG.Complex(1, 0);
188  *    var z2 = new JXG.Complex(0, 1);
189  *    var z = new JXG.Complex(z1);
190  *    z.add(z2);</pre>
191  * @namespace Namespace for the complex number arithmetic functions.
192  */
193 JXG.C = {};
194 
195 /**
196  * Add two (complex) numbers z1 and z2 and return the result as a (complex) number.
197  * @param z1 Summand
198  * @param z2 Summand
199  * @return A complex number equal to the sum of the given parameters.
200  */
201 JXG.C.add = function(/** JXG.Complex,number */ z1, /** JXG.Complex,number */ z2) /** JXG.Complex */{
202     var z = new JXG.Complex(z1);
203     z.add(z2);
204     return z;
205 };
206 
207 /**
208  * Subtract two (complex) numbers z1 and z2 and return the result as a (complex) number.
209  * @param z1 Minuend
210  * @param z2 Subtrahend
211  * @return A complex number equal to the difference of the given parameters.
212  */
213 JXG.C.sub = function(/** JXG.Complex,number */ z1, /** JXG.Complex,number */ z2) /** JXG.Complex */{
214     var z = new JXG.Complex(z1);
215     z.sub(z2);
216     return z;
217 };
218 
219 /**
220  * Multiply two (complex) numbers z1 and z2 and return the result as a (complex) number.
221  * @param z1 Factor
222  * @param z2 Factor
223  * @return A complex number equal to the product of the given parameters.
224  */
225 JXG.C.mult = function(/** JXG.Complex,number */ z1, /** JXG.Complex,number */ z2) /** JXG.Complex */{
226     var z = new JXG.Complex(z1);
227     z.mult(z2);
228     return z;
229 };
230 
231 /**
232  * Divide two (complex) numbers z1 and z2 and return the result as a (complex) number.
233  * @param z1 Dividend
234  * @param z2 Divisor
235  * @return A complex number equal to the quotient of the given parameters.
236  */
237 JXG.C.div = function(/** JXG.Complex,number */ z1, /** JXG.Complex,number */ z2) /** JXG.Complex */{
238     var z = new JXG.Complex(z1);
239     z.div(z2);
240     return z;
241 };
