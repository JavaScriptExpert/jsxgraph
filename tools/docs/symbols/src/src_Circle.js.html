 35 /**
 36  * A circle consists of all points with a given distance from one point. This point is called midpoint, the distance is called radius.
 37  * A circle can be constructed by providing a midpoint and a point on the circle or a midpoint and a radius (given as a number, function,
 38  * line, or circle). 
 39  * @class Creates a new circle object. Do not use this constructor to create a circle. Use {@link JXG.Board#create} with
 40  * type {@link Circle} instead.  
 41  * @constructor
 42  * @augments JXG.GeometryElement
 43  * @param {String,JXG.Board} board The board the new circle is drawn on.
 44  * @param {String} method Can be 
 45  * <ul><li> <b>'twoPoints'</b> which means the circle is defined by its midpoint and a point on the circle.</li>
 46  * <li><b>'pointRadius'</b> which means the circle is defined by its midpoint and its radius in user units</li>
 47  * <li><b>'pointLine'</b> which means the circle is defined by its midpoint and its radius given by the distance from the startpoint and the endpoint of the line</li>
 48  * <li><b>'pointCircle'</b> which means the circle is defined by its midpoint and its radius given by the radius of another circle</li></ul>
 49  * The parameters p1, p2 and radius must be set according to this method parameter.
 50  * @param {JXG.Point} p1 Midpoint of the circle.
 51  * @param {JXG.Point,JXG.Line,JXG.Circle} p2 Can be
 52  *<ul><li>a point on the circle if method is 'twoPoints'</li>
 53  <li>a line if the method is 'pointLine'</li>
 54  <li>a circle if the method is 'pointCircle'</li></ul>
 55  * @param {float} radius Only used when method is set to 'pointRadius'. Must be a given radius in user units.
 56  * @param {String} id Unique identifier for this object. If null or an empty string is given,
 57  * an unique id will be generated by Board
 58  * @param {String} name Not necessarily unique name. If null or an
 59  * empty string is given, an unique name will be generated.
 60  * @see JXG.Board#generateName
 61  */

 63 JXG.Circle = function (board, method, par1, par2, id, name, withLabel, layer) {
 64     /* Call the constructor of GeometryElement */
 65     this.constructor();
 66 
 67     /**
 68      * Type of element; Value is {@link JXG.OBJECT_TYPE_CIRCLE}.
 69      * @default {@link JXG.OBJECT_TYPE_CIRCLE}
 70      * @constant
 71      * @type number
 72      * @private
 73      */
 74     this.type = JXG.OBJECT_TYPE_CIRCLE;
 75     /**
 76      * Class of this element; Values is OBJECT_CLASS_CIRCLE.
 77      * @constant
 78      * @type number
 79      * @private
 80      */
 81     this.elementClass = JXG.OBJECT_CLASS_CIRCLE; 
 82 
 83     this.init(board, id, name);
 84     
 85     /**
 86      * Set the display layer.
 87      */
 88     if (layer == null) layer = board.options.layer['circle'];
 89     this.layer = layer;
 90 
 91     /**
 92      * Stores the given method.
 93      * Can be 
 94      * <ul><li><b>'twoPoints'</b> which means the circle is defined by its midpoint and a point on the circle.</li>
 95      * <li><b>'pointRadius'</b> which means the circle is defined by its midpoint and its radius given in user units or as term.</li>
 96      * <li><b>'pointLine'</b> which means the circle is defined by its midpoint and its radius given by the distance from the startpoint and the endpoint of the line.</li>
 97      * <li><b>'pointCircle'</b> which means the circle is defined by its midpoint and its radius given by the radius of another circle.</li></ul>
 98      * @type string
 99      * @see #midpoint
100      * @see #point2
101      * @see #radius
102      * @see #line
103      * @see #circle
104      */
105     this.method = method;
106     
107     /**
108      * The circles midpoint. Do not set this parameter directly as it will break JSXGraph's update system.
109      * @type JXG.Point
110      */    
111     this.midpoint = JXG.getReference(this.board, par1); 
112     this.midpoint.addChild(this);
113     
114     /* documented in GeometryElement */
115     this.visProp['visible'] = true;
116     this.visProp['fillColor'] = this.board.options.circle.fillColor;
117     this.visProp['highlightFillColor'] = this.board.options.circle.highlightFillColor;
118     this.visProp['strokeColor'] = this.board.options.circle.strokeColor;
119     this.visProp['highlightStrokeColor'] = this.board.options.circle.highlightStrokeColor;       
120     
121     /** Point on the circle only set if method equals 'twoPoints'. Do not set this parameter directly as it will break JSXGraph's update system.
122      * @type JXG.Point
123      * @see #method
124      */
125     this.point2 = null;
126     
127     /** Radius of the circle
128      * only set if method equals 'pointRadius'
129      * @type JXG.Point
130      * @default null
131      * @see #method     
132      */    
133     this.radius = 0;
134     
135     /** Line defining the radius of the circle given by the distance from the startpoint and the endpoint of the line
136      * only set if method equals 'pointLine'. Do not set this parameter directly as it will break JSXGraph's update system.
137      * @type JXG.Line
138      * @default null
139      * @see #method     
140      */    
141     this.line = null;
142     
143     /** Circle defining the radius of the circle given by the radius of the other circle
144      * only set if method equals 'pointLine'. Do not set this parameter directly as it will break JSXGraph's update system.
145      * @type JXG.Circle
146      * @default null
147      * @see #method     
148      */     
149     this.circle = null;
150 
151     if(method == 'twoPoints') {
152         this.point2 = JXG.getReference(board,par2);
153         this.point2.addChild(this);
154         this.radius = this.Radius(); 
155     }
156     else if(method == 'pointRadius') {
157         this.generateTerm(par2);  // Converts GEONExT syntax into JavaScript syntax
158         this.updateRadius();                        // First evaluation of the graph  
159     }
160     else if(method == 'pointLine') {
161         // dann ist p2 die Id eines Objekts vom Typ Line!
162         this.line = JXG.getReference(board,par2);
163         this.radius = this.line.point1.coords.distance(JXG.COORDS_BY_USER, this.line.point2.coords);    
164     }
165     else if(method == 'pointCircle') {
166         // dann ist p2 die Id eines Objekts vom Typ Circle!
167         this.circle = JXG.getReference(board,par2);
168         this.radius = this.circle.Radius();     
169     } 
170     
171     // create Label
172     if (withLabel!=null) 
173         this.createLabel(withLabel);
174     
175     if(method == 'twoPoints') {
176         //this.point2 = JXG.getReference(board,par2);
177         //this.point2.addChild(this);
178         //this.radius = this.Radius(); 
179         this.id = this.board.addCircle(this);           
180     }
181     else if(method == 'pointRadius') {
182         //this.generateTerm(par2);  // Converts GEONExT syntax into JavaScript syntax
183         //this.updateRadius();                        // First evaluation of the graph
184         this.id = this.board.addCircle(this);
185         this.notifyParents(par2);      
186     }
187     else if(method == 'pointLine') {
188         // dann ist p2 die Id eines Objekts vom Typ Line!
189         //this.line = JXG.getReference(board,par2);
190         //this.radius = this.line.point1.coords.distance(JXG.COORDS_BY_USER, this.line.point2.coords);
191         this.line.addChild(this);
192         this.id = this.board.addCircle(this);        
193     }
194     else if(method == 'pointCircle') {
195         // dann ist p2 die Id eines Objekts vom Typ Circle!
196         //this.circle = JXG.getReference(board,par2);
197         //this.radius = this.circle.Radius();
198         this.circle.addChild(this);
199         this.id = this.board.addCircle(this);        
200     }

212 /*
213     var genauigkeit = this.board.options.precision.hasPoint;
214     genauigkeit = genauigkeit/(this.board.stretchX); 
215     
216     var checkPoint = new JXG.Coords(JXG.COORDS_BY_SCREEN, [x,y], this.board);
217     var r = this.Radius();

226         p = new JXG.Coords(JXG.COORDS_BY_SCREEN, [x,y], this.board),

233 /**
234  * Used to generate a polynomial for a point p that lies on this circle.
235  * @param p The point for that the polynomial is generated.
236  * @return An array containing the generated polynomial.
237  * @private
238  */
239 JXG.Circle.prototype.generatePolynomial = function (p) {
240     /*
241      * We have four methods to construct a circle:
242      *   (a) Two points
243      *   (b) Midpoint and radius
244      *   (c) Midpoint and radius given by length of a segment
245      *   (d) Midpoint and radius given by another circle
246      *
247      * In case (b) we have to distinguish two cases:
248      *  (i)  radius is given as a number
249      *  (ii) radius is given as a function
250      * In the latter case there's no guarantee the radius depends on other geometry elements
251      * in a polynomial way so this case has to be omitted.
252      *
253      * Another tricky case is case (d):
254      * The radius depends on another circle so we have to cycle through the ancestors of each circle
255      * until we reach one that's radius does not depend on another circles radius.
256      *
257      *
258      * All cases (a) to (d) vary only in calculation of the radius. So the basic formulae for
259      * a glider G (g1,g2) on a circle with midpoint M (m1,m2) and radius r is just:
260      *
261      *     (g1-m1)^2 + (g2-m2)^2 - r^2 = 0
262      *
263      * So the easiest case is (b) with a fixed radius given as a number. The other two cases (a)
264      * and (c) are quite the same: Euclidean distance between two points A (a1,a2) and B (b1,b2),
265      * squared:
266      *
267      *     r^2 = (a1-b1)^2 + (a2-b2)^2
268      *
269      * For case (d) we have to cycle recursively through all defining circles and finally return the
270      * formulae for calculating r^2. For that we use JXG.Circle.symbolic.generateRadiusSquared().
271      */
272 
273     var m1 = this.midpoint.symbolic.x;
274     var m2 = this.midpoint.symbolic.y;
275     var g1 = p.symbolic.x;
276     var g2 = p.symbolic.y;
277 
278     var rsq = this.generateRadiusSquared();
279 
280     /* No radius can be calculated (Case b.ii) */
281     if (rsq == '')
282         return [];
283 
284     var poly = '((' + g1 + ')-(' + m1 + '))^2 + ((' + g2 + ')-(' + m2 + '))^2 - (' + rsq + ')';
285     return [poly];
286 };

288 /**
289  * Generate symbolic radius calculation for loci determination with Groebner-Basis algorithm.
290  * @type String
291  * @return String containing symbolic calculation of the circle's radius or an empty string
292  * if the radius can't be expressed in a polynomial equation.
293  * @private
294  */
295 JXG.Circle.prototype.generateRadiusSquared = function () {
296     /*
297      * Four cases:
298      *
299      *   (a) Two points
300      *   (b) Midpoint and radius
301      *   (c) Midpoint and radius given by length of a segment
302      *   (d) Midpoint and radius given by another circle
303      */
304 
305     var rsq = '';
306 
307     if (this.method == "twoPoints") {
308         var m1 = this.midpoint.symbolic.x;
309         var m2 = this.midpoint.symbolic.y;
310         var p1 = this.point2.symbolic.x;
311         var p2 = this.point2.symbolic.y;
312 
313         rsq = '(' + p1 + '-' + m1 + ')^2 + (' + p2 + '-' + m2 + ')^2';
314     } else if (this.method == "pointRadius") {
315         if (typeof(this.radius) == 'number')
316             rsq = '' + this.radius*this.radius;
317     } else if (this.method == "pointLine") {
318         var p1 = this.line.point1.symbolic.x;
319         var p2 = this.line.point1.symbolic.y;
320 
321         var q1 = this.line.point2.symbolic.x;
322         var q2 = this.line.point2.symbolic.y;
323 
324         rsq = '(' + p1 + '-' + q1 + ')^2 + (' + p2 + '-' + q2 + ')^2';
325     } else if (this.method == "pointCircle") {
326         rsq = this.circle.Radius();
327     }
328 
329     return rsq;
330 };

332 /**
333  * Uses the boards renderer to update the circle.
334  */
335 JXG.Circle.prototype.update = function () {
336     if(this.traced) {
337         this.cloneToBackground(true);
338     }
339     
340     if (this.needsUpdate) {
341         if(this.method == 'pointLine') {
342             this.radius = this.line.point1.coords.distance(JXG.COORDS_BY_USER, this.line.point2.coords); 
343         }
344         else if(this.method == 'pointCircle') {
345             this.radius = this.circle.Radius();
346         }
347         else if(this.method == 'pointRadius') {
348             this.radius = this.updateRadius();
349         }
350         if (!this.board.geonextCompatibilityMode) {
351             this.updateStdform();
352             this.updateQuadraticform();
353         }
354     }
355 };

451 /**
452  * Calculates the radius of the circle.
453  * @type float
454  * @return The radius of the circle
455  */
456 JXG.Circle.prototype.Radius = function() {
457     if(this.method == 'twoPoints') {
458         return(Math.sqrt(Math.pow(this.midpoint.coords.usrCoords[1]-this.point2.coords.usrCoords[1],2) + Math.pow(this.midpoint.coords.usrCoords[2]-this.point2.coords.usrCoords[2],2)));
459     }
460     else if(this.method == 'pointLine' || this.method == 'pointCircle') {
461         return this.radius;
462     }
463     else if(this.method == 'pointRadius') {
464         return this.updateRadius();
465     }
466 };

499 /**
500  * Clone the circle to the background.
501  * @param addToTrace Not used yet. Always true.
502  */
503 JXG.Circle.prototype.cloneToBackground = function(/** boolean */ addToTrace) {
504     var copy = {};
505     copy.id = this.id + 'T' + this.numTraces;
506     this.numTraces++;
507     copy.midpoint = {};
508     copy.midpoint.coords = this.midpoint.coords;
509     var r = this.Radius();
510     copy.Radius = function() { return r; };
511     copy.getRadius = function() { return r; }; // deprecated
512     
513     copy.board = {};
514     copy.board.unitX = this.board.unitX;
515     copy.board.unitY = this.board.unitY;
516     copy.board.zoomX = this.board.zoomX;
517     copy.board.zoomY = this.board.zoomY;
518     copy.board.stretchX = this.board.stretchX;
519     copy.board.stretchY = this.board.stretchY;
520 
521     copy.visProp = this.visProp;
522     JXG.clearVisPropOld(copy);
523     
524     this.board.renderer.drawCircle(copy);
525     this.traces[copy.id] = this.board.renderer.getElementById(copy.id);
526 
527     delete copy;
528 };

616  * @class This element is used to provide a constructor for a circle. 
617  * @pseudo
618  * @description  A circle consists of all points with a given distance from one point. This point is called midpoint, the distance is called radius.
619  * A circle can be constructed by providing a midpoint and a point on the circle or a midpoint and a radius (given as a number, function,
620  * line, or circle). 
621  * @name Circle
622  * @augments JXG.Circle
623  * @constructor
624  * @type JXG.Circle
625  * @throws {Exception} If the element cannot be constructed with the given parent objects an exception is thrown.
626  * @param {JXG.Point_number,JXG.Point,JXG.Line,JXG.Circle} midpoint,radius The midpoint must be given as a {@link JXG.Point}, but the radius can be given
627  * as a number (which will create a circle with a fixed radius), another {@link JXG.Point}, a {@link JXG.Line} (the distance of start and end point of the
628  * line will determine the radius), or another {@link JXG.Circle}.
629  * @example
630  * // Create a circle providing two points
631  * var p1 = board.create('point', [2.0, 2.0]);
632  * var p2 = board.create('point', [2.0, 0.0]);
633  * var c1 = board.create('circle', [p1, p2]);
634  * 
635  * // Create another circle using the above circle
636  * var p3 = board.create('point', [3.0, 2.0]);
637  * var c2 = board.create('circle', [p3, c1]);
638  * </pre><div id="5f304d31-ef20-4a8e-9c0e-ea1a2b6c79e0" style="width: 400px; height: 400px;"></div>
639  * <script type="text/javascript">
640  *   var cex1_board = JXG.JSXGraph.initBoard('5f304d31-ef20-4a8e-9c0e-ea1a2b6c79e0', {boundingbox: [-1, 9, 9, -1], axis: true, showcopyright: false, shownavigation: false});
641  *   var cex1_p1 = cex1_board.create('point', [2.0, 2.0]);
642  *   var cex1_p2 = cex1_board.create('point', [2.0, 0.0]);
643  *   var cex1_c1 = cex1_board.create('circle', [cex1_p1, cex1_p2]);
644  *   var cex1_p3 = cex1_board.create('point', [3.0, 2.0]);
645  *   var cex1_c2 = cex1_board.create('circle', [cex1_p3, cex1_c1]);
646  * </script><pre>
647  */
648 JXG.createCircle = function(board, parentArr, atts) {
649     var el, p, i;
650     atts = JXG.checkAttributes(atts,{withLabel:JXG.readOption(board.options,'circle','withLabel'), layer:null});
651     
652     p = [];
653     for (i=0;i<parentArr.length;i++) {
654         if (JXG.isPoint(parentArr[i])) {
655             p[i] = parentArr[i];              // Point
656         } else if (parentArr[i].length>1) {
657             p[i] = board.create('point', parentArr[i], {visible:false,fixed:true});  // Coordinates
658         } else {
659             p[i] = parentArr[i];              // Something else (number, function, string)
660         }
661     }
662     if( parentArr.length==2 && JXG.isPoint(p[0]) && JXG.isPoint(p[1]) ) {
663         // Point/Point
664         el = new JXG.Circle(board, 'twoPoints', p[0], p[1], atts['id'], atts['name'],atts['withLabel'],atts['layer']);
665     } else if( ( JXG.isNumber(p[0]) || JXG.isFunction(p[0]) || JXG.isString(p[0])) && JXG.isPoint(p[1]) ) {
666         // Number/Point
667         el = new JXG.Circle(board, 'pointRadius', p[1], p[0], atts['id'], atts['name'],atts['withLabel'],atts['layer']);
668     } else if( ( JXG.isNumber(p[1]) || JXG.isFunction(p[1]) || JXG.isString(p[1])) && JXG.isPoint(p[0]) ) {
669         // Point/Number
670         el = new JXG.Circle(board, 'pointRadius', p[0], p[1], atts['id'], atts['name'],atts['withLabel'],atts['layer']);
671     } else if( (p[0].type == JXG.OBJECT_TYPE_CIRCLE) && JXG.isPoint(p[1]) ) {
672         // Circle/Point
673         el = new JXG.Circle(board, 'pointCircle', p[1], p[0], atts['id'], atts['name'],atts['withLabel'],atts['layer']);
674     } else if( (p[1].type == JXG.OBJECT_TYPE_CIRCLE) && JXG.isPoint(p[0])) {
675         // Point/Circle
676         el = new JXG.Circle(board, 'pointCircle', p[0], p[1], atts['id'], atts['name'],atts['withLabel'],atts['layer']);
677     } else if( (p[0].type == JXG.OBJECT_TYPE_LINE) && JXG.isPoint(p[1])) {
678         // Circle/Point
679         el = new JXG.Circle(board, 'pointLine', p[1], p[0], atts['id'], atts['name'],atts['withLabel'],atts['layer']);
680     } else if( (p[1].type == JXG.OBJECT_TYPE_LINE) && JXG.isPoint(p[0])) {
681         // Point/Circle
682         el = new JXG.Circle(board, 'pointLine', p[0], p[1], atts['id'], atts['name'],atts['withLabel'],atts['layer']);
683     } else if( parentArr.length==3 && JXG.isPoint(p[0]) && JXG.isPoint(p[1]) && JXG.isPoint(p[2])) {
684         // Circle through three points
685         var arr = JXG.createCircumcircle(board, p, atts); // returns [center, circle]
686         arr[0].setProperty({visible:false});
687         return arr[1];
688     } else
689         throw new Error("JSXGraph: Can't create circle with parent types '" + (typeof parentArr[0]) + "' and '" + (typeof parentArr[1]) + "'.");
690     
691     return el;
692 };
693 
694 JXG.JSXGraph.registerElement('circle', JXG.createCircle);
