 53 /**
 54  * Calculates the angle defined by the points A, B, C.
 55  * @param {JXG.Point,array} A A point  or [x,y] array.
 56  * @param {JXG.Point,array} B Another point or [x,y] array.
 57  * @param {JXG.Point,array} C A circle - no, of course the third point or [x,y] array.
 58  * @type number
 59  * @return The angle in radian measure.
 60  * @deprecated Use {@link JXG.Algebra#rad} instead.
 61  * @see #rad
 62  * @see #trueAngle
 63  */
 64 JXG.Algebra.prototype.angle = function(A, B, C) {   
 65     var a = [],
 66         b = [],
 67         c = [],
 68         u, v, s, t;
 69         
 70     if (A.coords == null) {
 71         a[0] = A[0];
 72         a[1] = A[1];
 73     } else {
 74         a[0] = A.coords.usrCoords[1];
 75         a[1] = A.coords.usrCoords[2];
 76     }
 77     if (B.coords == null) {
 78         b[0] = B[0];
 79         b[1] = B[1];
 80     } else {
 81         b[0] = B.coords.usrCoords[1];
 82         b[1] = B.coords.usrCoords[2];
 83     }
 84     if (C.coords == null) {
 85         c[0] = C[0];
 86         c[1] = C[1];
 87     } else {
 88         c[0] = C.coords.usrCoords[1];
 89         c[1] = C.coords.usrCoords[2];
 90     }
 91     u = a[0] - b[0];
 92     v = a[1] - b[1];
 93     s = c[0] - b[0];
 94     t = c[1] - b[1];
 95     return Math.atan2(u*t-v*s,u*s+v*t);    
 96 };

 98 /**
 99  * Calculates the angle defined by the three points A, B, C if you're going from A to C around B counterclockwise.
100  * @param A Point or [x,y] array
101  * @param B Point or [x,y] array
102  * @param C Point or [x,y] array
103  * @return The angle in degrees.
104  * @see #rad

106 JXG.Algebra.prototype.trueAngle = function(/** JXG.Point */ A, /** JXG.Point */ B, /** JXG.Point */ C) /** number */ {
107     return this.rad(A,B,C)*57.295779513082323; // *180.0/Math.PI;

110 /**
111  * Calculates the internal angle defined by the three points A, B, C if you're going from A to C around B counterclockwise.
112  * @param {JXG.Point} A Point or [x,y] array
113  * @param {JXG.Point} B Point or [x,y] array
114  * @param {JXG.Point} C Point or [x,y] array
115  * @type number
116  * @see #trueAngle
117  * @return Angle in radians.
118  */
119 JXG.Algebra.prototype.rad = function(A,B,C) {
120     var ax, ay, bx, by, cx, cy,
121         abx, aby, cbx, cby,
122         cp, l1, l2, phiacos, phicos, sp, 
123         phi = 0;
124         
125     if (A.coords == null) {
126         ax = A[0];
127         ay = A[1];
128     } else {
129         ax = A.coords.usrCoords[1];
130         ay = A.coords.usrCoords[2];
131     }
132     if (B.coords == null) {
133         bx = B[0];
134         by = B[1];
135     } else {
136         bx = B.coords.usrCoords[1];
137         by = B.coords.usrCoords[2];
138     }
139     if (C.coords == null) {
140         cx = C[0];
141         cy = C[1];
142     } else {
143         cx = C.coords.usrCoords[1];
144         cy = C.coords.usrCoords[2];
145     }
146     cbx = cx - bx;
147     cby = cy - by;
148     abx = ax - bx;
149     aby = ay - by;
150     
151     //return Math.atan2(cby-aby,cbx-abx);
152 
153     sp = cbx*abx + cby*aby;               // scalar product of c-b and a-b
154     cp = abx*cby - aby*cbx;               // cross product of a-b c-b
155     l1 = Math.sqrt(abx*abx + aby*aby);    // length of a-b
156     l2 = Math.sqrt(cbx*cbx + cby*cby);    // length of c-b
157     phiacos = sp / (l1 * l2);             // calculate the angle as cosine from scalar product
158     if (phiacos > 1) { // these things should not happen, but can happen because of numerical inaccurracy
159         phiacos = 1;
160     } else if (phiacos < -1) {
161         phiacos = -1;
162     }
163     phicos = Math.acos(phiacos); // calculate the angle
164     
165         /*
166          * The calculated angle may not be the right angle because of the calculation of acos 
167         real     | quadrant  | quadrant | algebraic sign 
168         quadrant | by cosine | by sine  | of sine 
169            1.    |   1.      |   1.     |   +
170            2.    |   2.      |   1.     |   +
171            3.    |   2.      |   3.     |   -
172            4.    |   1.      |   3.     |   - 
173          So only for the first quadrant the calculated angle is ok. 
174          But we can use the sine, which is connected with the cross product to select the right angle. 
175          Calculate the sine of the calculated angle and multiply it with the cross product's value.
176          
177         real     | quadrant  | algebraic sign | algebraic sign of 
178         quadrant | by cosine | of sin(phicos) | cross product
179            1.    |   1.      |   +            |   +
180            2.    |   2.      |   +            |   +
181            3.    |   2.      |   +            |   -
182            4.    |   1.      |   +            |   - 
183          So always the negative angle of phicos has to be taken if the product is negative.
184          */
185     if ((Math.sin(phicos) * cp) < 0) {
186         phi = 6.2831853071795862 - phicos; // 2 * Math.PI - phicos;
187     } else {
188         phi = phicos;
189     }
190     return phi;
191 };

193 /**
194  * Calculates the bisection between the three points A, B, C. The bisection is defined by two points:
195  * Parameter B and a point with the coordinates calculated in this function.
196  * @param A Point
197  * @param B Point
198  * @param C Point
199  * @return Coordinates of the second point defining the bisection.
200  */
201 JXG.Algebra.prototype.angleBisector = function(/** JXG.Point */ A, /** JXG.Point */ B, /** JXG.Point */ C) /** JXG.Coords */ {
202     /* First point */
203     var Ac = A.coords.usrCoords,
204         Bc = B.coords.usrCoords, 
205         Cc = C.coords.usrCoords,
206         x = Ac[1]-Bc[1],
207         y = Ac[2]-Bc[2],
208         d = Math.sqrt(x*x+y*y),
209         phiA, phiC, phi;
210     x /= d;
211     y /= d;
212     
213     phiA = Math.acos(x);
214     if (y<0) { phiA *= -1; }
215     if (phiA<0) { phiA += 2*Math.PI; } 
216     
217     /* Second point */
218     x = Cc[1]-Bc[1];
219     y = Cc[2]-Bc[2];
220     d = Math.sqrt(x*x+y*y);
221     x /= d;
222     y /= d;
223     
224     phiC = Math.acos(x);
225     if (y<0) { phiC *= -1; }
226     if (phiC<0) { phiC += 2*Math.PI; } 
227  
228     phi=(phiA+phiC)*0.5;
229     if (phiA>phiC) { 
230         phi+=Math.PI;
231     }
232 
233     x = Math.cos(phi)+Bc[1];
234     y = Math.sin(phi)+Bc[2];
235     
236     return new JXG.Coords(JXG.COORDS_BY_USER, [x,y], this.board);
237 };

257 /**
258  @private
259  @deprecated
260  OBSOLETE
261  * Calculates the coordinates of a point on the parallel through the given point to the given line through point1 and point2.
262  * @param {JXG.Point} point1 First point lying on the given line.
263  * @param {JXG.Point} point2 Second point lying on the given line.
264  * @param {JXG.Point} point Point through which the parallel is drawn.
265  * @type JXG.Coords
266  * @return Coordinates of a point defining the parallel together with the given point.
267  */
268 JXG.Algebra.prototype.parallel = function(point1, point2, point) {
269     var factor = 1,
270         pc = point.coords.usrCoords,
271         p1c = point1.coords.usrCoords,
272         p2c = point2.coords.usrCoords,
273         x = pc[1] + factor*(p2c[1] - p1c[1]),
274         y = pc[2] + factor*(p2c[2] - p1c[2]);
275     
276     return new JXG.Coords(JXG.COORDS_BY_USER, [x,y], this.board);
277 };

279 /**
280  * Reflects the point along the line.
281  * @param {JXG.Line} line Axis of reflection.
282  * @param {JXG.Point} point Point to reflect.
283  * @type JXG.Coords
284  * @return Coordinates of the reflected point.
285  */  
286 JXG.Algebra.prototype.reflection = function(line,point) {
287     /* (v,w) defines the slope of the line */    
288     var pc = point.coords.usrCoords,
289         p1c = line.point1.coords.usrCoords,
290         p2c = line.point2.coords.usrCoords,
291         x0, y0, x1, y1, v, w, mu;
292         
293     v = p2c[1]-p1c[1];
294     w = p2c[2]-p1c[2];
295     
296     x0 = pc[1]-p1c[1];
297     y0 = pc[2]-p1c[2];
298     
299     mu = (v*y0-w*x0)/(v*v+w*w);
300     
301     /* point + mu*(-y,x) waere Lotpunkt */
302     x1 = pc[1] + 2*mu*w;
303     y1 = pc[2] - 2*mu*v;
304     
305     return new JXG.Coords(JXG.COORDS_BY_USER, [x1,y1], this.board);
306 };

337 /**
338  * Calculates the coordinates of a point on the perpendicular to the given line through
339  * the given point.
340  * @param {JXG.Line} line A line.
341  * @param {JXG.Point} point Intersection point of line to perpendicular.
342  * @type JXG.Coords
343  * @return Coordinates of a point on the perpendicular to the given line through the given point.
344  */
345 JXG.Algebra.prototype.perpendicular = function(line, point) {
346     var A = line.point1.coords.usrCoords,
347         B = line.point2.coords.usrCoords,
348         C = point.coords.usrCoords,
349         x, y, change,
350         fmd, emc, d0, d1, den;
351     
352     if(point == line.point1) { // Punkt ist erster Punkt der Linie
353         x = A[1] + B[2] - A[2];
354         y = A[2] - B[1] + A[1];
355         change = true;
356     }
357     else if(point == line.point2) {  // Punkt ist zweiter Punkt der Linie    
358         x = B[1] + A[2] - B[2];
359         y = B[2] - A[1] + B[1];
360         change = false;
361     }
362     else if( ((Math.abs(A[1] - B[1]) > this.eps) && 
363              (Math.abs(C[2] - (A[2] - B[2])*(C[1]-A[1])/(A[1] - B[1])-A[2]) < this.eps)) ||
364              ((Math.abs(A[1] - B[1]) <= this.eps) && (Math.abs(A[1] - C[1]) < this.eps)) ) { // Punkt liegt auf der Linie
365         x = C[1] + B[2] - C[2];
366         y = C[2] - B[1] + C[1]; 
367         change = true;
368         if(Math.abs(x - C[1]) < this.eps && Math.abs(y - C[2]) < this.eps) {
369             x = C[1] + A[2] - C[2];
370             y = C[2] - A[1] + C[1];
371             change = false;
372         }
373     }
374     else { // Punkt liegt nicht auf der Linie -> als zweiter Punkt wird der Lotfusspunkt gewaehlt
375         fmd = A[2] - B[2];
376         emc = A[1] - B[1];
377         d0 = B[1]*fmd - B[2]*emc;
378         d1 = C[1]*emc + C[2]*fmd;
379         den = fmd*fmd + emc*emc;
380         if(Math.abs(den)<this.eps) {
381             den = this.eps;
382         }
383         x = (d0*fmd + d1*emc) / den;
384         y = (d1*fmd - d0*emc) /den;
385         change = true;
386     }                            
387     return [new JXG.Coords(JXG.COORDS_BY_USER, [x, y], this.board),change];             
388 };

390 /**
391  * Calculates the midpoint of the circumcircle of the three given points.
392  * @param {JXG.Point} point1 Point
393  * @param {JXG.Point} point2 Point
394  * @param {JXG.Point} point3 Point
395  * @type JXG.Coords
396  * @return Coordinates of the midpoint of the circumcircle of the given points.
397  */
398 JXG.Algebra.prototype.circumcenterMidpoint = function(point1, point2, point3) {
399     var A = point1.coords.usrCoords,
400         B = point2.coords.usrCoords,
401         C = point3.coords.usrCoords,
402         u, v, den, x, y;
403 
404     u = ((A[1]-B[1])*(A[1]+B[1]) + (A[2]-B[2])*(A[2]+B[2])) * 0.5;
405     v = ((B[1]-C[1])*(B[1]+C[1]) + (B[2]-C[2])*(B[2]+C[2])) * 0.5;
406     den = (A[1]-B[1])*(B[2]-C[2]) - (B[1]-C[1])*(A[2]-B[2]);
407               
408     if (Math.abs(den) < this.eps) {
409         den = this.eps;
410     }
411     
412     x = (u * (B[2]-C[2]) - v*(A[2]-B[2])) / den;
413     y = (v * (A[1]-B[1]) - u*(B[1]-C[1])) / den;
414     
415     return new JXG.Coords(JXG.COORDS_BY_USER, [x, y], this.board);
416 };

418 /**
419  * Calculates the coordinates of the intersection of the given lines.
420  * @param {JXG.Line} line1 Line.
421  * @param {JXG.Line} line2 Line.
422  * @type JXG.Coords
423  * @return Coordinates of the intersection point of the given lines.
424  */
425 JXG.Algebra.prototype.intersectLineLine = function(line1, line2) {
426     var A = line1.point1.coords.usrCoords,
427         B = line1.point2.coords.usrCoords,
428         C = line2.point1.coords.usrCoords,
429         D = line2.point2.coords.usrCoords,
430         d0, d1, den, x, y;
431            
432     d0 = A[1]*B[2] - A[2]*B[1];
433     d1 = C[1]*D[2] - C[2]*D[1];
434     den = (B[2]-A[2])*(C[1]-D[1]) - (A[1]-B[1])*(D[2]-C[2]);
435                  
436     if(Math.abs(den) < this.eps) {
437          den = this.eps; 
438     }
439     x = (d0*(C[1]-D[1]) - d1*(A[1]-B[1])) / den;
440     y = (d1*(B[2]-A[2]) - d0*(D[2]-C[2])) / den;
441 
442     return new JXG.Coords(JXG.COORDS_BY_USER, [x, y], this.board);
443 };

445 /**
446  * Calculates the coordinates of the intersection of the given line and circle.
447  * @param {JXG.Circle} circle Circle.
448  * @param {JXG.Line} line Line.
449  * @type array
450  * @return Array of the Coordinates of the intersection points of the given circle with the given line and
451  * the amount of intersection points in the first component of the array.
452  */
453 JXG.Algebra.prototype.intersectCircleLine = function(circle, line) {
454     var eA = line.point1.coords.usrCoords,
455         eB = line.point2.coords.usrCoords,
456         fM = circle.midpoint.coords.usrCoords,
457         s, d0, d1, b, w, h, r, n1, dx, dy, firstPointX, firstPointY, l, x, y, n1s, firstPoint, secondPoint, d;
458 
459     s = line.point1.Dist(line.point2);
460     if (s > 0) {
461         d0 = circle.midpoint.Dist(line.point1);
462         d1 = circle.midpoint.Dist(line.point2);
463         b = ((d0 * d0) + (s * s) - (d1 * d1)) / (2 * s);
464         w = (d0 * d0) - (b * b);
465         w = (w < 0) ? 0 : w;
466         h = Math.sqrt(w);
467         
468         r = circle.Radius();
469         n1 = Math.sqrt((r * r) - h*h);
470         dx = eB[1] - eA[1];
471         dy = eB[2] - eA[2];
472         firstPointX = fM[1] + (h / s) * dy;
473         firstPointY = fM[2] - (h / s) * dx;
474         d0 = (eB[1] * dy) - (eB[2] * dx);
475         d1 = (firstPointX * dx) + (firstPointY * dy);
476         l = (dy * dy) + (dx * dx);
477         if (Math.abs(l) < this.eps) { l = this.eps; }
478         x = ((d0 * dy) + (d1 * dx)) / l;
479         y = ((d1 * dy) - (d0 * dx)) / l;
480         n1s = n1/s;
481         firstPoint =  new JXG.Coords(JXG.COORDS_BY_USER, [x + n1s * dx, y + n1s * dy], this.board);
482         secondPoint = new JXG.Coords(JXG.COORDS_BY_USER, [x - n1s * dx, y - n1s * dy], this.board);
483         d = circle.midpoint.coords.distance(JXG.COORDS_BY_USER, firstPoint);
484       
485         if ((r < (d - 1)) || isNaN(d)) {
486             return [0];
487         } else {
488             return [2,firstPoint,secondPoint];       
489         }
490     }
491     return [0];
492 };

494 /**
495  * Calculates the coordinates of the intersection of the given circles.
496  * @param {JXG.Circle} circle1 Circle.
497  * @param {JXG.Circle} circle2 Circle.
498  * @type array
499  * @return Array of the Coordinates of the intersection points of the given circles and the
500  * amount of intersection points in the first component of the array.
501  */
502 JXG.Algebra.prototype.intersectCircleCircle = function(circle1, circle2) { 
503     var intersection = {},
504         r1 = circle1.Radius(),
505         r2 = circle2.Radius(),
506         M1 = circle1.midpoint.coords.usrCoords,
507         M2 = circle2.midpoint.coords.usrCoords,
508         rSum, rDiff, s, 
509         dx, dy, a, h;
510         
511     rSum = r1 + r2;
512     rDiff = Math.abs(r1 - r2);    
513     // Abstand der Mittelpunkte der beiden Kreise
514     s = circle1.midpoint.coords.distance(JXG.COORDS_BY_USER, circle2.midpoint.coords);
515     if (s > rSum) {
516         return [0]; // Kreise schneiden sich nicht, liegen nebeneinander
517     } 
518     else if (s < rDiff) {
519         return [0]; // Kreise schneiden sich nicht, liegen ineinander
520     } 
521     else {
522         if (s != 0) {
523             intersection[0] = 1; // es gibt einen Schnitt        
524             dx = M2[1] - M1[1];
525             dy = M2[2] - M1[2];
526             a = (s * s - r2 * r2 + r1 * r1) / (2 * s);
527             h = Math.sqrt(r1 * r1 - a * a);
528             intersection[1] = new JXG.Coords(JXG.COORDS_BY_USER, 
529                                              [M1[1] + (a / s) * dx + (h / s) * dy, 
530                                               M1[2] + (a / s) * dy - (h / s) * dx], 
531                                              this.board);
532             intersection[2] = new JXG.Coords(JXG.COORDS_BY_USER, 
533                                              [M1[1] + (a / s) * dx - (h / s) * dy, 
534                                               M1[2] + (a / s) * dy + (h / s) * dx],

536         }

618 /**
619  * Calculates the coordinates of the projection of a coordinates pair on a given curve. In case of
620  * function graphs this is the
621  * intersection point of the curve and the parallel to y-axis through the given point.
622  * @param {float} x coordinate to project.
623  * @param {float} y coordinate to project.
624  * @param {float} start value for newtons method
625  * @param {JXG.Curve} graph Curve on that the point is projected.
626  * @type JXG.Coords
627  * @see #projectPointToCurve
628  * @return Array containing the coordinates of the projection of the given point on the given graph and 
629  * the position on the curve.
630  */
631 JXG.Algebra.prototype.projectCoordsToCurve = function(x,y,t,curve) {
632     var newCoords, x0, y0, x1, y1, den, i, mindist, dist, lbda, j,
633         infty = 1000000.0, minfunc, tnew, fnew, fold, delta, steps;
634         
635     if (curve.curveType=='parameter' || curve.curveType=='polar') { 
636         // Function to minimize
637         minfunc = function(t){ 
638                     var dx = x-curve.X(t),
639                         dy = y-curve.Y(t);
640                     return dx*dx+dy*dy;
641                 };
642         //t = JXG.Math.Numerics.root(JXG.Math.Numerics.D(minfunc),t);
643         fold = minfunc(t);
644         steps = 20;
645         delta = (curve.maxX()-curve.minX())/steps;
646         tnew = curve.minX();
647         for (j=0;j<steps;j++) {
648             fnew = minfunc(tnew);
649             if (fnew<fold) {
650                 t = tnew;
651                 fold = fnew;
652             }
653             tnew += delta;
654         }
655         t = JXG.Math.Numerics.root(JXG.Math.Numerics.D(minfunc),t);
656 
657         if (t<curve.minX()) { t = curve.maxX()+t-curve.minX(); } // Cyclically
658         if (t>curve.maxX()) { t = curve.minX()+t-curve.maxX(); }
659         newCoords = new JXG.Coords(JXG.COORDS_BY_USER, [curve.X(t),curve.Y(t)], this.board);
660     } else if (curve.curveType == 'plot') {
661         mindist = infty;
662         for (i=0;i<curve.numberPoints;i++) {
663             x0 = x-curve.X(i);
664             y0 = y-curve.Y(i);
665             dist = Math.sqrt(x0*x0+y0*y0);
666             if (dist<mindist) {
667                 mindist = dist;
668                 t = i;
669             }
670             if (i==curve.numberPoints-1) { continue; }
671 
672             x1 = curve.X(i+1)-curve.X(i);
673             y1 = curve.Y(i+1)-curve.Y(i);
674             den = x1*x1+y1*y1;
675             if (den>=JXG.Math.eps) {
676                 lbda = (x0*x1+y0*y1)/den;
677                 dist = Math.sqrt( x0*x0+y0*y0 - lbda*(x0*x1+y0*y1) );
678             } else {
679                 lbda = 0.0;
680                 dist = Math.sqrt(x0*x0+y0*y0);
681             }
682             if (lbda>=0.0 && lbda<=1.0 && dist<mindist) { 
683                 t = i+lbda;
684                 mindist = dist;
685             } 
686         }
687         i = Math.floor(t);
688         lbda = t-i;
689         if (i<curve.numberPoints-1) {
690             x = lbda*curve.X(i+1)+(1.0-lbda)*curve.X(i);
691             y = lbda*curve.Y(i+1)+(1.0-lbda)*curve.Y(i);
692         } else {
693             x = curve.X(i);
694             y = curve.Y(i);
695         }
696         newCoords = new JXG.Coords(JXG.COORDS_BY_USER, [x,y], this.board); 
697     } else {             // functiongraph
698         t = x;
699         x = t; //curve.X(t);
700         y = curve.Y(t);
701         newCoords = new JXG.Coords(JXG.COORDS_BY_USER, [x,y], this.board); 
702     }
703     return [curve.updateTransform(newCoords),t];
704 };

743 /**
744  * Converts expression of the form <i>leftop^rightop</i> into <i>Math.pow(leftop,rightop)</i>.
745  * @param {String} te Expression of the form <i>leftop^rightop</i>
746  * @type String
747  * @return Converted expression.
748  */
749 JXG.Algebra.prototype.replacePow = function(te) {
750     var count, pos, c,
751         leftop, rightop, pre, p, left, i, right, expr;
752     //te = te.replace(/\s+/g,''); // Loesche allen whitespace
753                                 // Achtung: koennte bei Variablennamen mit Leerzeichen
754                                 // zu Problemen fuehren.
755     i = te.indexOf('^');
756     while (i>=0) {
757         left = te.slice(0,i);
758         if (left.charAt(left.length-1)==')') {
759             count = 1;
760             pos = left.length-2;
761             while (pos>=0 && count>0) {
762                 c = left.charAt(pos);
763                 if (c==')') { count++; }
764                 else if (c=='(') { count--; }
765                 pos--;
766             }   
767             if (count==0) {
768                 leftop = '';
769                 pre = left.substring(0,pos+1);   // finde evtl. F vor (...)^
770                 p = pos;
771                 while (p>=0 && pre.substr(p,1).match(/(\w+)/)) {
772                     leftop = RegExp.$1+leftop;
773                     p--;
774                 }
775                 leftop += left.substring(pos+1,left.length);
776                 leftop = leftop.replace(/([\(\)\+\*\%\^\-\/\]\[])/g,"\\$1");
777             }
778         } else {
779             leftop = '[\\w\\.]+';
780         }
781         right = te.slice(i+1);
782         if (right.match(/^([\w\.]*\()/)) {
783             count = 1;
784             pos = RegExp.$1.length;
785             while (pos<right.length && count>0) {
786                 c = right.charAt(pos);
787                 if (c==')') { count--; }
788                 else if (c=='(') { count++; }
789                 pos++;
790             }
791             if (count==0) {
792                 rightop = right.substring(0,pos);
793                 rightop = rightop.replace(/([\(\)\+\*\%\^\-\/\[\]])/g,"\\$1");
794             }
795         } else {
796             rightop = '[\\w\\.]+';  // ^b 
797         }
798         expr = new RegExp('(' + leftop + ')\\^(' + rightop + ')');
799         te = te.replace(expr,"this.board.algebra.pow($1,$2)");

801     }
802     return te;

805 /**
806  * Converts expression of the form <i>If(a,b,c)</i> into <i>(a)?(b):(c)/i>.
807  * @param {String} te Expression of the form <i>If(a,b,c)</i>
808  * @type String
809  * @return Converted expression.
810  */
811 JXG.Algebra.prototype.replaceIf = function(te) {
812     var s = '',
813         left, right,
814         first = null,
815         second = null,
816         third = null,
817         i, pos, count, k1, k2, c, meat;
818     
819     i = te.indexOf('If(');
820     if (i<0) { return te; }
821 
822     te = te.replace(/""/g,'0'); // "" means not defined. Here, we replace it by 0
823     while (i>=0) {
824         left = te.slice(0,i);
825         right = te.slice(i+3); 
826         
827         // Search the end of the If() command and take out the meat
828         count = 1;
829         pos = 0;
830         k1 = -1;
831         k2 = -1;
832         while (pos<right.length && count>0) {
833             c = right.charAt(pos);
834             if (c==')') { 
835                 count--;
836             } else if (c=='(') {
837                 count++;
838             } else if (c==',' && count==1) {
839                 if (k1<0) { 
840                     k1 = pos; // first komma
841                 } else {
842                     k2 = pos; // second komma
843                 }
844             }
845             pos++;
846         } 
847         meat = right.slice(0,pos-1);
848         right = right.slice(pos);
849         
850         // Test the two kommas
851         if (k1<0) { return ''; } // , missing
852         if (k2<0) { return ''; } // , missing
853         
854         first = meat.slice(0,k1);
855         second = meat.slice(k1+1,k2);
856         third = meat.slice(k2+1);
857         first = this.replaceIf(first);    // Recurse
858         second = this.replaceIf(second);  // Recurse
859         third = this.replaceIf(third);    // Recurse
860 
861         s += left + '((' + first + ')?' + '('+second+'):('+third+'))';  
862         te = right;
863         first = null;
864         second = null;
865         i = te.indexOf('If(');
866     }
867     s += right;
868     return s;
869 };

1237     if (circ[4]<this.eps) { // Radius is zero, return center of circle

1313 /**
1314   * @private
1315   *
1316   * Normalize the stdform [c,b0,b1,a,k,r,q0,q1].
1317   * @param {Array} stdform to be normalized.
1318   * @type {Array}
1319   * @return The normalized stdform.
1320   */
1321 JXG.Algebra.prototype.normalize = function(stdform) {
1322     var a2 = 2*stdform[3],
1323         r = stdform[4]/(a2),  // k/(2a)
1324         n, signr; 
1325     stdform[5] = r;
1326     stdform[6] = -stdform[1]/a2;
1327     stdform[7] = -stdform[2]/a2;
1328     if (r==Infinity || isNaN(r)) {
1329         n = Math.sqrt(stdform[1]*stdform[1]+stdform[2]*stdform[2]);
1330         stdform[0] /= n;
1331         stdform[1] /= n;
1332         stdform[2] /= n;
1333         stdform[3] = 0;
1334         stdform[4] = 1;
1335     } else if (Math.abs(r)>=1) {
1336         stdform[0] = (stdform[6]*stdform[6]+stdform[7]*stdform[7]-r*r)/(2*r);
1337         stdform[1] = -stdform[6]/r;
1338         stdform[2] = -stdform[7]/r;
1339         stdform[3] = 1/(2*r);
1340         stdform[4] = 1;
1341     } else {
1342         signr = (r<=0)?(-1):(1/*(r==0)?0:1*/);
1343         stdform[0] = signr*(stdform[6]*stdform[6]+stdform[7]*stdform[7]-r*r)*0.5;
1344         stdform[1] = -signr*stdform[6];
1345         stdform[2] = -signr*stdform[7];
1346         stdform[3] = signr/2;
1347         stdform[4] = signr*r;
1348     }
1349     return stdform;

1352 /**
1353  * Compute an intersection of the curves c1 and c2
1354  * with a generalized Newton method.
1355  * We want to find values t1, t2 such that
1356  * c1(t1) = c2(t2), i.e.
1357  * (c1_x(t1)-c2_x(t2),c1_y(t1)-c2_y(t2)) = (0,0).
1358  * We set
1359  * (e,f) := (c1_x(t1)-c2_x(t2),c1_y(t1)-c2_y(t2))
1360  *
1361  * The Jacobian J is defined by
1362  * J = (a, b)
1363  *     (c, d)
1364  * where
1365  * a = c1_x'(t1)
1366  * b = -c2_x'(t2)
1367  * c = c1_y'(t1)
1368  * d = -c2_y'(t2)
1369  *
1370  * The inverse J^(-1) of J is equal to
1371  *  (d, -b)/
1372  *  (-c, a) / (ad-bc)
1373  *
1374  * Then, (t1new, t2new) := (t1,t2) - J^(-1)*(e,f).
1375  * If the function meetCurveCurve possesses the properties
1376  * t1memo and t2memo then these are taken as start values
1377  * for the Newton algorithm.
1378  * After stopping of the Newton algorithm the values of t1 and t2 are stored in
1379  * t1memo and t2memo.
1380  * 
1381  * @param {JXG.Curve} c1: Curve, Line or Circle
1382  * @param {JXG.Curve} c2: Curve, Line or Circle
1383  * @param {float} t1ini: start value for t1
1384  * @param {float} t2ini: start value for t2
1385  * @type {JXG.Coords}
1386  * @return coordinate object for the intersection point
1387  **/
1388 JXG.Algebra.prototype.meetCurveCurve = function(c1,c2,t1ini,t2ini) {
1389     var count = 0,
1390         t1, t2,
1391         a, b, c, d, disc,
1392         e, f, F, 
1393         D00, D01, 
1394         D10, D11;
1395         
1396     if (arguments.callee.t1memo) {
1397         t1 = arguments.callee.t1memo;
1398         t2 = arguments.callee.t2memo;
1399     } else {
1400         t1 = t1ini;
1401         t2 = t2ini;
1402     }
1403 /*
1404     if (t1>c1.maxX()) { t1 = c1.maxX(); }
1405     if (t1<c1.minX()) { t1 = c1.minX(); }
1406     if (t2>c2.maxX()) { t2 = c2.maxX(); }
1407     if (t2<c2.minX()) { t2 = c2.minX(); }
1408 */
1409     e = c1.X(t1)-c2.X(t2);
1410     f = c1.Y(t1)-c2.Y(t2);
1411     F = e*e+f*f;
1412     
1413     D00 = c1.board.D(c1.X,c1);
1414     D01 = c2.board.D(c2.X,c2);
1415     D10 = c1.board.D(c1.Y,c1);
1416     D11 = c2.board.D(c2.Y,c2);
1417     
1418     while (F>JXG.Math.eps && count<10) {
1419         a =  D00(t1);
1420         b = -D01(t2);
1421         c =  D10(t1);
1422         d = -D11(t2);
1423         disc = a*d-b*c;
1424         t1 -= (d*e-b*f)/disc;
1425         t2 -= (a*f-c*e)/disc;
1426         e = c1.X(t1)-c2.X(t2);
1427         f = c1.Y(t1)-c2.Y(t2);
1428         F = e*e+f*f;
1429         count++;
1430     }
1431 //console.log(t1+' '+t2);
1432 
1433     arguments.callee.t1memo = t1;
1434     arguments.callee.t2memo = t2;
1435 
1436     //return (new JXG.Coords(JXG.COORDS_BY_USER, [2,2], this.board));
1437     if (Math.abs(t1)<Math.abs(t2)) {
1438         return (new JXG.Coords(JXG.COORDS_BY_USER, [c1.X(t1),c1.Y(t1)], this.board));
1439     } else {
1440         return (new JXG.Coords(JXG.COORDS_BY_USER, [c2.X(t2),c2.Y(t2)], this.board));
1441     }
1442 };
