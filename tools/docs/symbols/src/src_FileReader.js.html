 28 this.parseFileContent = function(url, board, format) {
 29     this.request = false;
 30     var e;
 31     try {
 32         //this.request = new ActiveXObject("Msxml2.XMLHTTP");
 33         this.request = new XMLHttpRequest();
 34         if(format.toLowerCase()=='raw')
 35             this.request.overrideMimeType('text/plain; charset=iso-8859-1');
 36         else
 37             this.request.overrideMimeType('text/xml; charset=iso-8859-1');
 38     } catch (e) {
 39         try {
 40             this.request = new ActiveXObject("Msxml2.XMLHTTP");
 41         } catch (e) {
 42             try {
 43                 this.request = new ActiveXObject("Microsoft.XMLHTTP");
 44             } catch (e) {
 45                 this.request = false;
 46             }
 47         }
 48     }
 49     if (!this.request) {
 50         alert("AJAX not activated!");
 51         return;
 52     }
 53     this.request.open("GET", url, true);
 54     if(format.toLowerCase()=='raw') {
 55         this.cbp = function() {
 56             var request = this.request;
 57             if (request.readyState == 4) {
 58                 board(request.responseText);
 59             }
 60         }; //).bind(this);
 61     } else {
 62         this.cbp = function() {
 63             var request = this.request;
 64             if (request.readyState == 4) {
 65                 var text = '';
 66 
 67                 if (typeof request.responseStream!='undefined' && request.responseText.slice(0,2) == "PK") {
 68                     text = (new JXG.Util.Unzip(JXG.Util.Base64.decodeAsArray(BinFileReader(this.request)))).unzip();
 69                     text = text[0][0];
 70                 } else {
 71                     text = request.responseText;
 72                 }
 73                 this.parseString(text, board, format, false);
 74             }
 75         }; //).bind(this);
 76     }
 77     this.cb = JXG.bind(this.cbp,this);
 78     this.request.onreadystatechange = this.cb;
 79 
 80     try {
 81         this.request.send(null);
 82     } catch (e) {
 83         throw new Error("JSXGraph: problems opening " + url + " !");
 84     }
 85 }; // end: this.parseFileContent

 87 this.cleanWhitespace = function(el) {
 88     var cur = el.firstChild;
 89     while ( cur != null ) {
 90         if ( cur.nodeType == 3 && ! /\S/.test(cur.nodeValue) ) {
 91             el.removeChild( cur );
 92         } else if ( cur.nodeType == 1 ) {
 93             this.cleanWhitespace( cur );
 94         }
 95         cur = cur.nextSibling;
 96     }
 97 };

 99 this.stringToXMLTree = function(fileStr) {
100     // The string "fileStr" is converted into a XML tree.
101     if(typeof DOMParser == "undefined") {
102        // IE workaround, since there is no DOMParser
103        DOMParser = function () {};
104        DOMParser.prototype.parseFromString = function (str, contentType) {
105           if (typeof ActiveXObject != "undefined") {
106              var d = new ActiveXObject("MSXML.DomDocument");
107              d.loadXML(str);
108              return d;
109           }
110        };
111     }
112     var parser=new DOMParser();
113 
114     var tree = parser.parseFromString(fileStr,"text/xml");
115     this.cleanWhitespace(tree);
116     return tree;
117 };
118 
119 this.parseString = function(fileStr, board, format, isString) {
120     var tree;
121     
122     if (format.toLowerCase()=='cdy') {
123     	// if isString is true, fileStr is the base64 encoded zip file, otherwise it's just the zip file
124     	if(isString)
125     		fileStr = JXG.Util.Base64.decode(fileStr);
126         fileStr = JXG.CinderellaReader.readCinderella(fileStr, board);
127         board.xmlString = fileStr;
128         board.afterLoad();
129         return;
130     }
131     
132     // fileStr is a string containing the XML code of the construction
133     if (format.toLowerCase()=='geonext') {
134         fileStr = JXG.GeonextReader.prepareString(fileStr);
135     }
136     if (format.toLowerCase()=='geogebra') {
137         // if isString is true, fileStr is a base64 encoded string, otherwise it's the zipped file
138     	fileStr = JXG.GeogebraReader.prepareString(fileStr, isString);
139     }
140     if (format.toLowerCase()=='intergeo') {
141         if(isString)
142             fileStr = JXG.Util.Base64.decode(fileStr);
143     	fileStr = JXG.IntergeoReader.prepareString(fileStr);
144     }
145     board.xmlString = fileStr;
146     var tree = this.stringToXMLTree(fileStr);
147     // Now, we can walk through the tree
148     this.readElements(tree, board, format);
149 }; // end this.parse
150 
151 /**
152  * Reading the elements of a geonext or geogebra file
153  * @param {} tree expects the content of the parsed geonext file returned by function parseFromString
154  * @param {Object} board board object
155  */
156 this.readElements = function(tree, board, format) {
157     if (format.toLowerCase()=='geonext') {
158         board.suspendUpdate();
159         if(tree.getElementsByTagName('GEONEXT').length != 0) {
160             JXG.GeonextReader.readGeonext(tree, board);
161         }
162         board.unsuspendUpdate();
163     }
164     else if(tree.getElementsByTagName('geogebra').length != 0) {
165         JXG.GeogebraReader.readGeogebra(tree, board);
166     }
167     else if(format.toLowerCase()=='intergeo') {
168         JXG.IntergeoReader.readIntergeo(tree, board);
169     }
170     // cdy is already parsed in parseString()
171     board.afterLoad();
172 }; // end: this.readElements()
173 
174 }; // end: FileReader()
