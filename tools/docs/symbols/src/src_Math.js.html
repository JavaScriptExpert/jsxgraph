 40 /**
 41  * Represents a vector.
 42  * @constructor
 43  * @param {Array} elements An array of numerical values containing the coefficients to be put into the vector.
 44  * @return JXG.Math.Vector
 45  */
 46 JXG.Math.Vector = function(elements) {
 47     var i;
 48     this.length = 0;
 49     
 50     if((typeof elements != undefined) && (elements != null)) {
 51         for(i=0; i<elements.length; i++) {
 52             this.push(elements[i]);
 53         }
 54     }
 55 };

 70 /**
 71  * Exchanges two elements of the vector.
 72  * @param {int} i The first element that is to be exchanged.
 73  * @param {int} j The second element that is to be exchanged.
 74  */
 75 JXG.Math.Vector.prototype.exchange = function(i, j) {
 76     var temp = this[i];
 77     
 78     this[i] = this[j];
 79     this[j] = temp; 
 80 };

 82 /**
 83  * Represents a matrix.
 84  * @constructor
 85  * @param {Array} elements An 2-dimensional array of numerical values containing the coefficients to be put into the vector.
 86  * @throws {JXG.DimensionMismatchException} If the rows of the matrix don't have all the same length.
 87  * @return JXG.Math.Vector
 88  */
 89 JXG.Math.Matrix = function(elements) {
 90     var oldLength = 0,
 91         testLength = false,
 92         i, j, len, leni;
 93     
 94     this.length = 0;
 95     
 96     if ((typeof elements != undefined) && (elements != null)) {
 97         len = elements.length;
 98         for (i=0; i<len; i++) {
 99             leni = elements[i].length;
100             this.push(new Array());
101           
102             if (testLength) {
103                 if (oldLength != leni) {
104                     this.length = 0;
105                     throw new JXG.DimensionMismatchException("Your array contains arrays with different lengths.");
106                 }
107             }
108                 
109             for (j=0; j<leni; j++) {
110                 this[i].push(elements[i][j]);
111             }
112           
113             oldLength = leni;
114             testLength = true;
115         }
116     }
117 };

178         return default_msg + ".";

208 /**
209  * Matrix-vector multiplication.
210  * @param {Array} mat1 Two dimensional array of numbers
211  * @param {Array} vec Array of numbers
212  * @return {Array} Array of numbers containing result
213  */
214 JXG.Math.matVecMult = function(/** array */ mat1, /** array */ vec) /** array */ {
215     var m = mat1.length,
216         n = vec.length,
217         res = [],
218         i, s, k;
219     if (n==3) {
220         for (i=0;i<m;i++) {
221             res[i] = mat1[i][0]*vec[0] + mat1[i][1]*vec[1] + mat1[i][2]*vec[2];
222         }
223     } else {
224         for (i=0;i<m;i++) {
225             s = 0;
226             for (k=0;k<n;k++) { s += mat1[i][k]*vec[k]; }
227             res[i] = s;
228         }
229     }
230     return res;
231 };

233 /**
234  * Matrix-matrix multiplication.
235  * @param {Array} mat1 Two dimensional array of numbers
236  * @param {Array} mat2 Two dimensional array of numbers
237  * @return {Array} Two dimensional Array of numbers containing result
238  */
239 JXG.Math.matMatMult = function(/** array */ mat1, /** array */ mat2) /** array */ {
240     var m = mat1.length,
241         n = mat2[0].length,
242         m2 = mat2.length,
243         res = [], 
244         i, j, s, k;
245         
246     for (i=0;i<m;i++) {
247         res[i] = [];
248     }
249 
250     for (i=0;i<m;i++) {
251         for (j=0;j<n;j++) {
252             s = 0;
253             for (k=0;k<m2;k++) {
254                 s += mat1[i][k]*mat2[k][j];
255             }
256             res[i][j] = s;
257         }
258     }
259     return res;
260 };

262 /**
263  * Transpose a matrix which is of type array of arrays.
264  * @param {Array} M 
265  * @return {Array} transpose of M
266  */
267 JXG.Math.Matrix.transpose = function(/** Array */ M) /** Array*/  {
268     var MT = [], i, j, 
269         m, n;
270     
271     m = M.length;                   // number of rows of M
272     n = (M.length>0)?M[0].length:0; // number of columns of M
273 
274     for (i=0;i<n;i++) {
275         MT.push([]);
276         for (j=0;j<m;j++) {
277             MT[i].push(M[j][i]);
278         }
279     }
280     return MT;
281 }

283 /**
284   * Calculates the crossproducts of two vectors
285   * of length three.
286   * In case of homogeneous coordinates this is either
287   * - the intersection of two lines
288   * - the line through two points.
289   * @param {Array} c1 homogeneous coordinates of line (point) 1
290   * @param {Array} c2 homogeneous coordinates of line (point) 2
291   * @type Array
292   * @return vector of length 3:  homogeneous coordinates
293   *   of the resulting line / point.
294   */
295 JXG.Math.crossProduct = function(c1,c2) {
296     return [c1[1]*c2[2]-c1[2]*c2[1],
297             c1[2]*c2[0]-c1[0]*c2[2],
298             c1[0]*c2[1]-c1[1]*c2[0]];
299 };

301 /**
302  * Inner product of two vectors a, b. n is the length of the vectors.
303  * @param a Vector
304  * @param b Vector
305  * @param [n] Length of the Vectors. If not given the length of the first vector is taken.
306  * @return The inner product of a and b.
307  */
308 JXG.Math.innerProduct = function(a, b, n) {    
309     var i, s = 0;
310     
311     if(typeof n == 'undefined')
312         n = a.length;
313     
314     for (i=0;i<n;i++) {
315         s += a[i]*b[i];
316     }
317     return s;
318 };

322 /**
323 * Dynamic programming approach for recursive functions.
324 * From "Speed up your JavaScript, Part 3" by Nicholas C. Zakas.
325 * @see JXG.Math.factorial
326 * http://blog.thejit.org/2008/09/05/memoization-in-javascript/
327 */
328 JXG.memoizer = function (f) {
329     var cache, join;
330     
331     if (f.memo) {
332         return f.memo;
333     }
334     cache = {};
335     join = Array.prototype.join;
336 
337     return (f.memo = function() {
338         var key = join.call(arguments);
339         //return (key in cache)
340         return (typeof cache[key]!='undefined') // Seems to be a bit faster than "if (a in b)"
341             ? cache[key]
342             : cache[key] = f.apply(this, arguments);
343     });
344 };

346 /**
347 * Compute the factorial of a positive integer.
348 * @param {integer n}
349 * @return {return n*(n-1)...2*1}
350 */
351 JXG.Math.factorial = JXG.memoizer(function (n) {
352         if (n<0) return NaN; 
353         if (n==0 || n==1) return 1;
354         return n*arguments.callee(n-1);
355 });

357 /**
358 * Comupte the binomial coefficient.
359 * @param {integer n}
360 * @param {integer k}
361 * 
362 * @return {n\choose k}
363 */
364 JXG.Math.binomial = JXG.memoizer(function(n,k) {
365     var b, i;
366     
367     if (k>n || k<0) return 0;
368     if (k==0 || k==n) return 1;
369     
370     b = 1;
371     for (i=0;i<k;i++) {
372         b *= (n-i);
373         b /= (i+1);
374     }
375     return b;
376     //return arguments.callee(n-1,k-1)+arguments.callee(n-1,k);
377 });

387 /**
388 * Round a decimal number to n decimal places
389 * @deprecated Use (number).toFixed(n) instead.
390 * @param {float num} Number to round
391 * @param {integer n} number of digits after the point to leave
392 * 
393 * @return {rounded num}
394 */
395 JXG.Math.round = function(num, n) {
396     var z, s;
397     //return Math.round(num*Math.pow(10,n))/Math.pow(10,n);
398     //var z = num.toFixed(n);
399     
400     z = num - Math.ceil(num);
401     s = z.toString();
402     if (z < 0) {
403         s = s.substr(0,n+3);
404     }
405     else {
406         s = s.substr(0,n+2);
407     }
408     z = parseFloat(s);
409     t = parseInt(num.toString());
410     return t+z;
411 };

414  * Cosine hyperbolicus of x.
415  * @param {float} x The number the cosine hyperbolicus will be calculated of.
416  * @return {float} Cosine hyperbolicus of the given value.
417  */
418 JXG.Math.cosh = function(/** number */ x) /** number */ {
