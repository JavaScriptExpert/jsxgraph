224 /**
225  * Apply the options stored in this object to all objects on the given board.
226  * @param {JXG.Board} board The board to which objects the options will be applied.
227  */
228 JXG.useStandardOptions = function(board) {
229     var o = JXG.Options,
230         boardHadGrid = board.hasGrid,
231         el, t;
232 
233     board.hasGrid = o.grid.hasGrid;
234     board.gridX = o.grid.gridX;
235     board.gridY = o.grid.gridY;
236     board.gridColor = o.grid.gridColor;
237     board.gridOpacity = o.grid.gridOpacity;
238     board.gridDash = o.grid.gridDash;
239     board.snapToGrid = o.grid.snapToGrid;
240     board.snapSizeX = o.grid.SnapSizeX;
241     board.snapSizeY = o.grid.SnapSizeY;
242     board.takeSizeFromFile = o.takeSizeFromFile;
243 
244     for(el in board.objects) {
245         p = board.objects[el];
246         if(p.elementClass == JXG.OBJECT_CLASS_POINT) {
247             p.visProp['fillColor'] = o.point.fillColor;
248             p.visProp['highlightFillColor'] = o.point.highlightFillColor;
249             p.visProp['strokeColor'] = o.point.strokeColor;
250             p.visProp['highlightStrokeColor'] = o.point.highlightStrokeColor;
251         }
252         else if(p.elementClass == JXG.OBJECT_CLASS_LINE) {
253             p.visProp['fillColor'] = o.line.fillColor;
254             p.visProp['highlightFillColor'] = o.line.highlightFillColor;
255             p.visProp['strokeColor'] = o.line.strokeColor;
256             p.visProp['highlightStrokeColor'] = o.line.highlightStrokeColor;
257             for(t in p.ticks) {
258                 t.majorTicks = o.line.ticks.majorTicks;
259                 t.minTicksDistance = o.line.ticks.minTicksDistance;
260                 t.minorHeight = o.line.ticks.minorHeight;
261                 t.majorHeight = o.line.ticks.majorHeight;
262             }
263         }
264         else if(p.elementClass == JXG.OBJECT_CLASS_CIRCLE) {
265             p.visProp['fillColor'] = o.circle.fillColor;
266             p.visProp['highlightFillColor'] = o.circle.highlightFillColor;
267             p.visProp['strokeColor'] = o.circle.strokeColor;
268             p.visProp['highlightStrokeColor'] = o.circle.highlightStrokeColor;
269         }
270         else if(p.type == JXG.OBJECT_TYPE_ANGLE) {
271             p.visProp['fillColor'] = o.angle.fillColor;
272             p.visProp['highlightFillColor'] = o.angle.highlightFillColor;
273             p.visProp['strokeColor'] = o.angle.strokeColor;
274         }
275         else if(p.type == JXG.OBJECT_TYPE_ARC) {
276             p.visProp['fillColor'] = o.arc.fillColor;
277             p.visProp['highlightFillColor'] = o.arc.highlightFillColor;
278             p.visProp['strokeColor'] = o.arc.strokeColor;
279             p.visProp['highlightStrokeColor'] = o.arc.highlightStrokeColor;
280         }
281         else if(p.type == JXG.OBJECT_TYPE_POLYGON) {
282             p.visProp['fillColor'] = o.polygon.fillColor;
283             p.visProp['highlightFillColor'] = o.polygon.highlightFillColor;
284             p.visProp['fillOpacity'] = o.polygon.fillOpacity;
285             p.visProp['highlightFillOpacity'] = o.polygon.highlightFillOpacity;
286         }
287         else if(p.type == JXG.OBJECT_TYPE_CONIC) {
288             p.visProp['fillColor'] = o.conic.fillColor;
289             p.visProp['highlightFillColor'] = o.conic.highlightFillColor;
290             p.visProp['strokeColor'] = o.conic.strokeColor;
291             p.visProp['highlightStrokeColor'] = o.conic.highlightStrokeColor;
292         }
293         else if(p.type == JXG.OBJECT_TYPE_CURVE) {
294             p.visProp['strokeColor'] = o.curve.strokeColor;
295         }
296     }
297     for(el in board.objects) {
298         p = board.objects[el];
299         if(p.type == JXG.OBJECT_TYPE_SECTOR) {
300             p.arc.visProp['fillColor'] = o.sector.fillColor;
301             p.arc.visProp['highlightFillColor'] = o.sector.highlightFillColor;
302             p.arc.visProp['fillOpacity'] = o.sector.fillOpacity;
303             p.arc.visProp['highlightFillOpacity'] = o.sector.highlightFillOpacity;
304         }
305     }
306 
307     board.fullUpdate();
308     if(boardHadGrid && board.hasGrid) {
309         board.renderer.removeGrid(board);
310         board.renderer.drawGrid(board);
311     } else if(boardHadGrid && !board.hasGrid) {
312         board.renderer.removeGrid(board);
313     } else if(!boardHadGrid && board.hasGrid) {
314         board.renderer.drawGrid(board);
315     }
316 };

357 /**
358  * Decolorizes the given color.
359  * @param {String} color HTML string containing the HTML color code.
360  * @type String
361  * @return Returns a HTML color string
362  */
363 JXG.rgb2bw = function(color) {
364     if(color == 'none') {
365         return color;
366     }
367     var x, HexChars="0123456789ABCDEF", tmp, arr;
368     arr = JXG.rgbParser(color);
369     x = 0.3*arr[0] + 0.59*arr[1] + 0.11*arr[2];
370     tmp = HexChars.charAt((x>>4)&0xf)+HexChars.charAt(x&0xf);
371     color = "#" + tmp + "" + tmp + "" + tmp;
372     return color;
373 };

547 /**

551  * @param board {JXG.Board} The board the options should be applied to.

553 JXG.loadOptionsFromFile = function(fileurl, applyTo, board) {
554    this.cbp = function(t) {
555       this.parseString(t, applyTo, board);

558

562 /**
563  * Apply options given as a string to a board.
564  * @param text {String} Options given as a string in .json-Format
565  * @param apply {bool} <tt>true</tt> if the options should be applied to all objects on the board.
566  * @param board {JXG.Board} The board the options should be applied to.
567  */
568 JXG.parseOptionsString = function(text, applyTo, board) {
569    var newOptions = '';
570 
571    if(text != '') {
572       newOptions = eval("(" + text + ")");
573    }
574    else
575       return;
576 
577    var maxDepth = 10;
578    var applyOption = function (base, option, depth) {
579       if(depth==10)
580          return;
581       depth++;
582 
583       for(var key in option) {
584          if((JXG.isNumber(option[key])) || (JXG.isArray(option[key])) || (JXG.isString(option[key])) || (option[key]==true) || (option[key]==false)) {
585             base[key] = option[key];
586          }
587          else {
588             applyOption(base[key], option[key], depth);
589          }
590       }
591    };
592 
593    applyOption(this, newOptions, 0);
594 
595    if(applyTo && typeof board != 'undefined') {
596        JXG.useStandardOptions(board);
597    }
598 };
