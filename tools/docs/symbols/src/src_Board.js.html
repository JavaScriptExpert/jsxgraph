531 /**
532  * Generates an unique name for the given object. The result depends on object.type, if the object is a {@link JXG.Point}, capital characters are used, if it is
533  * of type {@link JXG.Line} only lower case characters are used. If object is of type {@link JXG.Polygon}, a bunch of lower case characters prefixed
534  * with P_ are used. If object is of type {@link JXG.Circle} the name is generated using lower case characters. prefixed with k_ is used. In any
535  * other case, lower case chars prefixed with s_ is used.
536  * @param {String,Object} object Reference or id or name of an geometry object that is to be named.
537  * @returns {String} Unique name for the object.
538  * @private
539  */
540 JXG.Board.prototype.generateName = function(object) {
541     if(object.type == JXG.OBJECT_TYPE_TICKS) {
542         return '';
543     }
544 
545     var possibleNames;
546     if(object.elementClass == JXG.OBJECT_CLASS_POINT) {
547         // points have capital letters
548         possibleNames = ['', 'A', 'B', 'C', 'D', 'E', 'F', 'G', 'H', 'I', 'J', 'K', 'L', 'M', 'N', 'O',
549                                   'P', 'Q', 'R', 'S', 'T', 'U', 'V', 'W', 'X', 'Y', 'Z'];
550     }
551     else {
552         // all other elements get lowercase labels
553         possibleNames = ['', 'a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'i', 'j', 'k', 'l', 'm', 'n', 'o',
554                                   'p', 'q', 'r', 's', 't', 'u', 'v', 'w', 'x', 'y', 'z'];
555     }
556 
557     // maximum length of a name
558     var maxNameLength = 3;
559     var pre = '';
560     var nameBase = '';
561     var post = '';
562 
563     if(object.elementClass == JXG.OBJECT_CLASS_POINT || object.elementClass == JXG.OBJECT_CLASS_LINE) {
564     }
565     else {
566         if(object.type == JXG.OBJECT_TYPE_POLYGON) {
567             pre = 'P_{';
568             post = '}';
569         }
570         else if(object.type == JXG.OBJECT_TYPE_CIRCLE) {
571             pre = 'k_{';
572             post = '}';
573         }
574         else if(object.type == JXG.OBJECT_TYPE_ANGLE) {
575             pre = 'W_{';
576             post = '}';
577         }
578         else {
579             pre = 's_{';
580             post = '}';
581         }
582     }
583     var indices = [];
584     var name = '';
585     var tmp = '';
586 
587     var i = 0;
588     var j = 0;
589 
590     for(i=0; i<maxNameLength; i++) {
591         indices[i] = 0;
592     }
593 
594     while (indices[maxNameLength-1] < possibleNames.length) {
595         for(indices[0]=1; indices[0]<possibleNames.length; indices[0]++) {
596             name = pre;
597 
598             for(i=maxNameLength; i>0; i--) {
599                 name += possibleNames[indices[i-1]];
600             }
601 
602             if (this.elementsByName[name+post] == null) {
603                 return name+post;
604             }
605 
606         }
607         indices[0] = possibleNames.length;
608         for(i=1; i<maxNameLength; i++) {
609             if(indices[i-1] == possibleNames.length) {
610                 indices[i-1] = 1;
611                 indices[i]++;
612             }
613         }
614     }
615 
616     return '';
617 };

619 /**
620  * Generates unique id for a board. The result is randomly generated and prefixed with 'gxtBoard'.
621  * @returns {String} Unique id for a board.
622  * @private
623  */
624 JXG.Board.prototype.generateId = function () {
625     var r = 1;
626 
627     // as long as we don't have an unique id generate a new one
628     while(JXG.JSXGraph.boards['gxtBoard' + r] != null) {
629         r = Math.round(Math.random()*33);
630     }
631 
632     return ('gxtBoard' + r);
633 };

635 /**
636  * Composes the unique id for a board. If the ID is empty ('' or null) a new ID is generated, depending on the object type.
637  * Additionally, the id of the label is set. As a side effect {@link JXG.Board#numObjects} is updated.
638  * @param {Object} obj Reference of an geometry object that is to be named.
639  * @param {int} type Type of the object.
640  * @returns {String} Unique id for a board.
641  * @private
642  */
643 JXG.Board.prototype.setId = function (obj, type) {
644     var num = this.numObjects,
645         elId = obj.id;
646     this.numObjects++;
647 
648     // Falls Id nicht vorgegeben, eine Neue generieren:
649     if((elId == '') || (elId == null)) {
650         elId = this.id + type + num;
651     }
652     // Objekt an den Renderer zum Zeichnen uebergeben
653     obj.id = elId;
654     // Objekt in das assoziative Array einfuegen
655     this.objects[elId] = obj;
656 
657     if(obj.hasLabel) {
658         obj.label.content.id = elId+"Label";
659         this.addText(obj.label.content);
660     }
661     return elId;
662 };

664 /**
665  * Calculates mouse coordinates relative to the boards container.
666  * @param {Event} Event The browsers event object.
667  * @return {Array} Array of coordinates relative the boards container top left corner.
668  * @private
669  */
670 JXG.Board.prototype.getRelativeMouseCoordinates = function (Event) {
671     var pCont = this.containerObj,
672         cPos = JXG.getOffset(pCont),
673         n; //Element.cumulativeOffset(pCont);
674 
675     // add border width
676     n = parseInt(JXG.getStyle(pCont,'borderLeftWidth'));
677     if (isNaN(n)) n = 0; // IE problem if border-width not set explicitly
678     cPos[0] += n;
679 
680     n = parseInt(JXG.getStyle(pCont,'borderTopWidth'));
681     if (isNaN(n)) n = 0;
682     cPos[1] += n;
683 
684     // add padding
685     n = parseInt(JXG.getStyle(pCont,'paddingLeft'));
686     if (isNaN(n)) n = 0;
687     cPos[0] += n;
688 
689     n = parseInt(JXG.getStyle(pCont,'paddingTop'));
690     if (isNaN(n)) n = 0;
691     cPos[1] += n;
692 
693     return cPos;
694 };
